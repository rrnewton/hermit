16// This test races and it is agnostic to which direction the race goes.  That is, it won't
17// actually fail with a nonzero exit code when the race goes badly. The point is to assert
18// determinism.  Or the caller can use chaos to ensure that both passing and reaching
19// schedules are found.
20fn run_test() {
21    let dir = tempdir().unwrap();
22    let path = dir.path().join("sock");
23    let server_path = path.clone();
24    let client_path = path.clone();
25    let (client_result_tx, client_result_rx) = mpsc::channel();
26
27    let server = thread::spawn(move || {
28        let _listener = UnixListener::bind(server_path).expect("bind to succeed");
29
30        if client_result_rx
31            .recv()
32            .expect("client result channel closed")
33        {
34            println!("Server: got client");
35        }
36    });
37    let client = thread::spawn(move || match UnixStream::connect(client_path) {
38        Ok(_stream) => {
39            eprintln!("Client: connection succeeded..");
40            true
41        }
42        Err(_) => {
43            eprintln!("Client: connection failed.");
44            false
45        }
46    });
47
48    let connected = client.join().expect("client to be ok");
49    client_result_tx
50        .send(connected)
51        .expect("server result channel closed");
52    server.join().expect("server to be ok");
53}
