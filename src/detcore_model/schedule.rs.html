27/// A scheduled action by one thread in the system.  This can be recorded, or replayed to guide the
28/// schedule.
29#[derive(PartialEq, Debug, Eq, Clone, Hash, Serialize, Deserialize)]
30pub struct SchedEvent {
31    /// The thread that originated the event.
32    pub dettid: DetTid,
33    /// The operation performed by the thread.
34    pub op: Op,
35    /// The consecutive count of that same operation (run length encoding).
36    pub count: u32,
37    /// The instruction pointer before this batch of operations.
38    pub start_rip: Option<InstructionPointer>,
39    /// The instruction pointer after this batch of operations.
40    pub end_rip: Option<InstructionPointer>,
41    /// An optional snapshot of the thread logical time at this point.
42    /// This includes time waiting on the global scheduler.
43    pub end_time: Option<LogicalTime>,
44}
45
46/// A more compact printing.
47impl fmt::Display for SchedEvent {
48    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
49        write!(f, "(tid{}", self.dettid)?;
50        if self.count > 1 {
51            write!(f, " cnt={}", self.count)?;
52        }
53        if let Some(srip) = self.start_rip {
54            write!(f, " strt={:#x}", srip)?;
55        }
56        if let Some(erip) = self.end_rip {
57            write!(f, " end={:#x}", erip)?;
58        }
59        if let Some(time) = self.end_time {
60            write!(f, " time={}", time)?;
61        }
62        write!(f, " {:?})", self.op)?;
63        Ok(())
64    }
65}

67impl SchedEvent {
68    /// Add a syscall to the global scheduling history.  This takes the instruction pointer of the
69    /// syscall itself.
70    pub fn syscall(dettid: DetTid, sysno: Sysno, phase: SyscallPhase) -> SchedEvent {
71        SchedEvent {
72            dettid,
73            op: Op::Syscall(sysno, phase),
74            count: 1,
75            start_rip: None,
76            end_rip: None,
77            end_time: None,
78        }
79    }
80
81    /// Add a batch of branches to the global scheduling history.
82    pub fn branches(dettid: DetTid, count: u32) -> SchedEvent {
83        SchedEvent {
84            dettid,
85            op: Op::Branch,
86            count,
87            start_rip: None, // TODO: track the start of the interval as well.
88            end_rip: None,
89            end_time: None,
90        }
91    }
92
93    /// Set the logical time directly.
94    pub fn with_time(mut self, time: LogicalDuration) -> SchedEvent {
95        self.end_time = Some(time);
96        self
97    }
98
99    /// Correctly set logical time based on the threads current time.
100    pub fn with_dettime(mut self, dt: &DetTime) -> SchedEvent {
101        self.end_time = Some(dt.without_starting());
102        self
103    }
104
105    /// Set the start_rip field.  The instruction pointer before the event began executing.
106    pub fn with_start_rip(mut self, start_rip: InstructionPointer) -> Self {
107        self.start_rip = Some(start_rip);
108        self
109    }
110
111    /// Set the end_rip field.  The instruction pointer after the event completed.
112    pub fn with_end_rip(mut self, end_rip: InstructionPointer) -> Self {
113        self.end_rip = Some(end_rip);
114        self
115    }
116}

121/// Which phase of the syscall did we observe on a given event: the prehook or the posthook.
122#[derive(PartialEq, Debug, Eq, Copy, Clone, Hash, Serialize, Deserialize)]
123pub enum SyscallPhase {
124    /// The event was recorded before physically beginning the syscall.
125    Prehook,
126
127    /// An internal (nonblocking) retry of the syscall to check if its done yet (but it wasn't).
128    Polling,
129
130    /// The event was recorded after the syscall logically completed.
131    Posthook,
132}

134/// Simply a type to hang Serialize/Deserialize instances off of.
135#[derive(PartialEq, Debug, Eq, Clone, Copy, Hash)]
136pub struct SigWrapper(pub Signal);
137
138impl Serialize for SigWrapper {
139    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
140    where
141        S: Serializer,
142    {
143        serializer.serialize_str(self.0.as_str())
144    }
145}
146
147impl<'de> de::Deserialize<'de> for SigWrapper {
148    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
149    where
150        D: de::Deserializer<'de>,
151    {
152        struct SignalVisitor;
153        impl<'de> de::Visitor<'de> for SignalVisitor {
154            type Value = Signal;
155
156            fn expecting(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
157                write!(f, "string representing a Signal")
158            }
159
160            fn visit_str<E>(self, v: &str) -> Result<Self::Value, E>
161            where
162                E: serde::de::Error,
163            {
164                let sig: Result<Signal, String> =
165                    FromStr::from_str(v).map_err(|e: nix::errno::Errno| e.to_string());
166                sig.map_err(serde::de::Error::custom)
167            }
168        }
169
170        let sig = deserializer.deserialize_str(SignalVisitor)?;
171        Ok(SigWrapper(sig))
172    }
173}
174
175impl FromStr for SigWrapper {
176    type Err = anyhow::Error;
177
178    fn from_str(s: &str) -> anyhow::Result<Self> {
179        Ok(SigWrapper(Signal::from_str(s)?))
180    }
181}

183impl From<Signal> for SigWrapper {
184    fn from(signal: Signal) -> Self {
185        Self(signal)
186    }
187}

189/// NOTE [Event Semantics]
190///
191/// The observable operations that happen on a guest thread.
192///
193/// Each Op event has a beginning and an end, containing an interval of zero or more instructions
194/// inbetween. Each beginning and end point in time can be thought of as an imaginary marker between
195/// two instructions.  Start/end RIP values, if present in the containing `SchedEvent`, correspond
196/// to those beginning/end points and always point to the *next* instruction to execute.
197///
198/// If we speak of an event as an instantaneous thing, we're usually thinking of it as its end
199/// marker. Which is as follows for each:
200///
201/// - Branches: after the  branch instruction has retired
202/// - Syscall prehooks: just before the syscall instruction, after whatever came before
203/// - Syscall posthooks: just after the syscall instruction completes
204/// - Rdtsc/Cpuid: just after the designated instruction
205/// - OtherInstructions: just after the region of zero or more branch-free, non-interceptable
206///   instructions.
207/// - SignalReceived: just after the last regular, pre-signal guest instruction, and just before the
208///   first instruction of the signal handler.
209///
210/// If we view each event as a series of instructions contained between its start/end markers, then
211/// the pattern of instructions for each would be as follows.  Here we use simple regular
212/// expressions with "B" standing for branch instructions, "S" for syscall instructions, "R" for
213/// RDTSC, "C" for CPUID, and "O" for all other instructions.
214///
215/// - Branch "O*B"
216/// - Syscall prehook: ""
217/// - Syscall posthook: "S"
218/// - Rdtsc: "R"
219/// - Cpuid: "C"
220/// - OtherInstructions: "O*"
221/// - SignalReceived: ""
222///
223/// A few observations about the above:
224///
225/// - Some events always correspond to zero instructions.
226/// - OtherInstructions are omnipresent "dark matter" that we cannot intercept or count, so are
227///   implicitly present between other events.
228/// - Therefore the OtherInstructions event itself is only interesting insofar as it signals the
229///   absence of other events.
230/// - Branches include an implicit prefix of OtherInstructions.  This is because for a branch count
231///   greater than 1 to make sense, we need to include the full between-branches O's: "..BO*B..". We
232///   could change this design by going to either extreme. (1) removing implicit O's and changing the
233///   count mechanism to allow repetition of entire sequences "(O*B)^3" instead of "B^3".  Or (2),
234///   including implicit O's in all event types, and not recording them explicitly.
235#[derive(PartialEq, Debug, Eq, Copy, Clone, Hash, Serialize, Deserialize)]
236pub enum Op {
237    /// A single retired conditional branch, corresponding to one increment of the RCB counter.
238    Branch,
239
240    /// A nondeterministic rdtsc instruction.
241    Rdtsc,
242
243    /// A nondeterministic cpuid instruction.
244    Cpuid,
245
246    /// A system call performed by the thread.  The bool is set to true when this is a syscall
247    /// PREHOOKh event, which is recorded BEFORE the syscall instruction executes, rather than
248    /// after.
249    Syscall(Sysno, SyscallPhase),
250
251    /// An unknown number of other instructions that occured BETWEEN hermit-interceptable events.
252    /// The only way to preempt in between these is expensive single-stepping.
253    OtherInstructions,
254
255    /// The point a signal handler is received, just after whatever regular user instruction
256    /// preceeded it, and just before the first instruction of the signal handler.
257    SignalReceived(SigWrapper),
258}
