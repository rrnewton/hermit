19/// Verification utility for replaying preemptions under hermit
20/// This utility runs a guest program under "hermit run" and records schedules and preemptions. On the second run those recorded schedules are replayed via "hermit run" subcommand
21/// The outputs (stdout, stderr, log file, schedules) are captured and placed in a temporary directory allowing further inspection (if --keep-temp-dir is provided)
22#[derive(Parser, Debug)]
23pub struct ChaosReplayOpts {
24    /// Whether to keep temp directory created for each container run. This directory contains (stdout, stderr, log file, etc) of the container process
25    /// This allows manual inspection of container outputs in cases when a cause of failure is unclear
26    #[clap(long)]
27    keep_temp_dir: bool,
28
29    /// Underlying hermit container will receive "isolated" workdir
30    #[clap(long)]
31    isolate_workdir: bool,
32
33    /// Additional arguments for hermit run subcommand
34    #[clap(long)]
35    hermit_arg: Vec<String>,
36
37    /// Path to a guest program
38    #[clap(value_name = "PROGRAM")]
39    guest_program: PathBuf,
40    /// Arguments for a guest program
41    #[clap(value_name = "ARGS")]
42    args: Vec<String>,
43}
44
45impl UseCase for ChaosReplayOpts {
46    fn build_temp_env(
47        &self,
48        _common_args: &CommonOpts,
49    ) -> crate::common::TemporaryEnvironmentBuilder {
50        TemporaryEnvironmentBuilder::new()
51            .persist_temp_dir(self.keep_temp_dir)
52            .run_count(2)
53    }
54
55    fn build_first_hermit_args(
56        &self,
57        temp_env: &TemporaryEnvironment,
58        current_run: &crate::common::RunEnvironment,
59    ) -> Vec<String> {
60        Hermit::new()
61            .log_level(tracing::Level::TRACE)
62            .log_file(current_run.log_file_path.clone())
63            .run(self.guest_program.clone(), self.args.clone())
64            .hermit_args(self.hermit_arg.clone())
65            .bind(temp_env.path().to_owned())
66            .workdir_isolate(current_run.workdir.clone(), self.isolate_workdir)
67            .record_preemptions_to(current_run.schedule_file.clone())
68            .into_args()
69    }
70
71    fn build_next_hermit_args(
72        &self,
73        _run_no: usize,
74        temp_env: &TemporaryEnvironment,
75        prev_run: &crate::common::RunEnvironment,
76        current_run: &crate::common::RunEnvironment,
77    ) -> Vec<String> {
78        Hermit::new()
79            .log_level(tracing::Level::TRACE)
80            .log_file(current_run.log_file_path.clone())
81            .run(self.guest_program.clone(), self.args.clone())
82            .hermit_args(self.hermit_arg.clone())
83            .bind(temp_env.path().to_owned())
84            .workdir_isolate(current_run.workdir.clone(), self.isolate_workdir)
85            .replay_preemptions_from(prev_run.schedule_file.clone())
86            .record_preemptions_to(current_run.schedule_file.clone())
87            .into_args()
88    }
89
90    fn options(&self) -> crate::use_case::UseCaseOptions {
91        crate::use_case::UseCaseOptions {
92            output_stdout: true,
93            output_stderr: true,
94            verify_stdout: true,
95            verify_stderr: true,
96            verify_detlog_syscalls: true,
97            verify_detlog_syscall_results: true,
98            verify_detlog_others: true,
99            verify_commits: false,
100            verify_exit_statuses: true,
101            verify_desync: true,
102            verify_schedules: true,
103            ignore_lines: vec![String::from("CHAOSRAND")],
104        }
105    }
106}
