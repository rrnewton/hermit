20fn compare_results<T: UseCase>(
21    use_case: &T,
22    common_args: &CommonOpts,
23    left: &RunEnvironment,
24    right: &RunEnvironment,
25) -> anyhow::Result<bool> {
26    let options = use_case.options();
27    let verify = Verify::new(common_args.hermit_bin.as_path());
28    let mut result = true;
29
30    if options.verify_stdout {
31        result &= verify.verify_stdout(left, right)?;
32    }
33
34    if options.verify_stderr {
35        result &= verify.verify_stderr(left, right)?;
36    }
37
38    if options.should_log_diff() {
39        result &= verify.verify_logs(
40            left,
41            right,
42            LogDiffOptions {
43                ignore_lines: options.ignore_lines,
44                skip_commits: !options.verify_commits,
45                skip_detlog_others: !options.verify_detlog_others,
46                skip_detlog_syscalls: !options.verify_detlog_syscalls,
47                skip_detlog_syscall_results: !options.verify_detlog_syscall_results,
48                syscall_history: 5,
49            },
50        )?;
51    }
52
53    if options.verify_exit_statuses {
54        result &= verify.verify_exit_statuses(left, right)?;
55    }
56
57    if options.verify_desync {
58        result &= verify.verify_desync(right)?;
59    }
60
61    if options.verify_schedules {
62        result &= verify.verify_schedules(left, right)?;
63    }
64
65    Ok(result)
66}

68pub fn run_use_case<T: UseCase>(use_case: T, common_args: &CommonOpts) -> anyhow::Result<bool> {
69    let temp_env = use_case.build_temp_env(common_args).build()?;
70    let mut iter = temp_env.runs().iter();
71    let mut current = iter.next().expect("use case must contain at least one run");
72    let mut counter = 0;
73
74    let options = use_case.options();
75
76    println!("{}", format!(":: Run #{}...", counter + 1).yellow().bold());
77
78    use_case.exec_first_run(common_args, &temp_env, current)?;
79
80    let mut result = true;
81    // Compare each run to the one before it (rather than comparing all to the first):
82    loop {
83        counter += 1;
84        if let Some(next) = iter.next() {
85            println!("{}", format!(":: Run #{}...", counter + 1).yellow().bold());
86
87            use_case.exec_next_run(common_args, counter + 1, &temp_env, current, next)?;
88
89            if !compare_results(&use_case, common_args, current, next)? {
90                result = false;
91                break;
92            } else {
93                // during verification we'll get a diff view so we're skipping printing here
94                // it only makes sense to output the stdout/stderr for the first run (if any)
95                if counter == 1 {
96                    if options.output_stdout {
97                        output_stdout(current)?;
98                    }
99                    if options.output_stderr {
100                        output_stderr(current)?;
101                    }
102                }
103            }
104            current = next;
105        } else {
106            break;
107        }
108    }
109
110    if !common_args.ignore_test_result_artifacts_dir
111        && let Some(path) = &common_args.test_result_artifact_dir
112    {
113        println!(
114            "{}",
115            format!(":: Copy artifacts to {}", path.display())
116                .yellow()
117                .bold()
118        );
119        let test_artifacts = TestArtifacts::new(path.to_owned());
120        test_artifacts.copy_run_results(&temp_env)?;
121    }
122    Ok(result)
123}
