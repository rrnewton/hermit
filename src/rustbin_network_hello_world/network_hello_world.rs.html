19use tempfile::tempdir;

21const BUFFER_SIZE: usize = 128;
22
23type ServerReadyPair = Arc<(Mutex<bool>, Condvar)>;
24
25fn main() {
26    if matches!(std::env::var("HERMIT_MODE"), Ok(mode) if mode == "record") {
27        // TODO: Record mode currently hangs for this test
28        eprintln!("Skipping test in record mode.");
29        return;
30    }
31
32    let temp_dir = tempdir().unwrap();
33    let socket_path = temp_dir.path().join("net-hello-world.sock");
34    let socket_path2 = socket_path.clone();
35
36    #[allow(clippy::mutex_atomic)]
37    let server_ready_pair = Arc::new((Mutex::new(false), Condvar::new()));
38    let server_ready_pair2 = Arc::clone(&server_ready_pair);
39
40    let server_thread = run_server(socket_path, server_ready_pair);
41
42    // Wait for the server to be ready
43    {
44        let (lock, cvar) = &*server_ready_pair2;
45        let mut started = lock.lock().unwrap();
46        while !*started {
47            started = cvar.wait(started).unwrap();
48        }
49    }
50
51    let client_thread = run_client(socket_path2);
52
53    client_thread.join().expect("client to be ok");
54    server_thread.join().expect("server to be ok");
55}
56
57fn run_server(socket_path: PathBuf, server_ready_pair: ServerReadyPair) -> thread::JoinHandle<()> {
58    thread::spawn(move || {
59        let listener = UnixListener::bind(&socket_path).unwrap();
60
61        {
62            let (lock, cvar) = &*server_ready_pair;
63            let mut started = lock.lock().unwrap();
64            *started = true;
65            cvar.notify_one();
66        }
67
68        println!("[Server] Listening...");
69        let mut incoming = listener.incoming().next().unwrap().unwrap();
70
71        let mut buffer = [0; BUFFER_SIZE];
72        let num_bytes_read = incoming.read(&mut buffer).unwrap();
73        let request = String::from_utf8(buffer[..num_bytes_read].to_vec()).unwrap();
74        println!("[Server] Got request: {}", request);
75
76        println!("[Server] Sending response...");
77        incoming.write_all(b"Goodbye, moon!").unwrap();
78    })
79}
80
81fn run_client(socket_path: PathBuf) -> thread::JoinHandle<()> {
82    thread::spawn(move || {
83        println!("[Client] Connecting...");
84        let mut stream = UnixStream::connect(socket_path).unwrap();
85
86        println!("[Client] Sending request...");
87        stream.write_all(b"Hello, world!").unwrap();
88
89        let mut buffer = [0; BUFFER_SIZE];
90        let num_bytes_read = stream.read(&mut buffer).unwrap();
91        let response = String::from_utf8(buffer[..num_bytes_read].to_vec()).unwrap();
92        println!("[Client] Got response: {}", response);
93    })
94}
