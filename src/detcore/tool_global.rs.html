81async fn yield_once() {
82    let mut yielded = false;
83    std::future::poll_fn(|context| {
84        if yielded {
85            Poll::Ready(())
86        } else {
87            yielded = true;
88            context.waker().wake_by_ref();
89            Poll::Pending
90        }
91    })
92    .await;
93}

110/// Everything the global scheduler needs to register a new child thread. A
111/// normal clone is registered by the parent; a `CLONE_VFORK` child registers
112/// itself (with `parent_is_kernel_blocked` set) because its parent is blocked
113/// inside the kernel until the child execs or exits.

116    parent_detpid: DetPid,
117    child_dettid: DetTid,

130impl InodePool {
131    fn new() -> Self {
132        InodePool {
133            inodes: HashMap::new(),
134            detinodes_info: HashMap::new(),
135            next_inode: 1,
136        }
137    }
138
139    // Allocate the next deterministic inode.  This takes the raw-inode and
140    // can return an existing mapping or extend the mapping by creating a
141    // new deterministic inode. The returned inode is strictly increasing
142    // to avoid inode re-use issue in some filesystem like ext4.
143    fn add_inode(&mut self, raw_inode: RawInode, mtime: LogicalTime) -> (DetInode, LogicalTime) {
144        match self.inodes.get(&raw_inode) {
145            None => {
146                let new = self.next_inode;
147                self.next_inode += 1;
148                assert!(self.inodes.insert(raw_inode, new).is_none());
149                let prev = self.detinodes_info.insert(
150                    new,
151                    DetInodeInfo {
152                        raw: raw_inode,
153                        mtime,
154                    },
155                );
156                assert!(prev.is_none()); // Should not have been previously used.
157                (new, mtime)
158            }
159            Some(dino) => {
160                let info = self
161                    .detinodes_info
162                    .get(dino)
163                    .expect("Internal invariant broken, det_ino missing entry");
164                (*dino, info.mtime)
165            }
166        }
167    }
168
169    // remove a det inode
170    fn remove_inode(&mut self, det_inode: DetInode) {
171        if let Some(info) = self.detinodes_info.remove(&det_inode) {
172            self.inodes.remove(&info.raw);
173        }
174    }
175}

179/// This is a singleton, and the one object of this type lives inside a central

261            past_first_execve: AtomicBool::new(false),

286    /// Unrecoverable fatal erorr. Bring things to a close cleanly, but as quickly as
287    /// possible.
288    pub fn force_shutdown_with_error(&self) {
289        let start = std::time::Instant::now();
290        let sched = loop {
291            if start.elapsed().as_millis() > 1000 {
292                eprintln!(
293                    "Could not acquire scheduler lock during forced shutdown (timeout)... proceeding anyway."
294                );
295                return;
296            }
297            match self.sched.try_lock() {
298                Ok(guard) => {
299                    break guard;
300                }
301                Err(std::sync::TryLockError::WouldBlock) => {
302                    std::thread::yield_now();
303                    continue;
304                }
305                Err(e) => {
306                    eprintln!(
307                        "Could not acquire scheduler lock during forced shutdown ({})... proceeding anyway.",
308                        e
309                    );
310                    return;
311                }
312            }
313        };
314        info!("Scheduler state at exit:\n{}", sched.full_summary());
315    }

317    /// Shut down anything running, in particular wait on the scheduler.
318    ///
319    /// This is basically the destructor for the global state, but is here rather than in the
320    /// Drop instance so that it can be async, and is more explicitly sequenced in the program.
321    ///
322    /// Print a summary of the execution, typically called when it is complete.
323    ///
324    /// If the boolean argument is true, print to stderr, otherwise only print the summary
325    /// to the log.
326    pub async fn clean_up(mut self, to_stderr: bool, print_summary_to_json_file: &Option<PathBuf>) {
327        if let Some(handle) = self.sched_handle.take() {
328            debug!("Global state cleanup, confirming scheduler has shut down...");
329            handle.await.expect("Global scheduler clean shutdown");
330            debug!("Global state cleanup, continuing...");
331        }
332        let banner =
333            "  ------------------------------ hermit run report ------------------------------";
334        let mut summary = self.into_run_summary().unwrap();
335
336        // Print machine-readable summary:
337        if let Some(path) = print_summary_to_json_file {
338            let json = serde_json::to_string_pretty(&summary).unwrap();
339            fs::write(path, json + "\n").unwrap();
340        }
341
342        // Print human-readable summary:
343        if to_stderr {
344            // In this case, print summary irrespective of logging level.
345            // TODO: output summary in machine-readable, JSON form.
346            eprint!("{}\n{}", banner, summary);
347        } else {
348            // Separate out the nondeterministic bits and print them at debug level:
349            let rt = summary.realtime_elapsed.take();
350            info!("\n{}\n{}", banner, summary);
351            if let Some(x) = rt {
352                debug!("Nondeterministic realtime elapsed: {:?}", x);
353            }
354        }
355    }

357    fn into_run_summary(self) -> anyhow::Result<RunSummary> {
358        // First, the scheduler can generate part of the summary
359        let mut summary = {
360            let mut sched = self.sched.lock().unwrap();
361            sched.generate_partial_run_summary(self.cfg.record_preemptions_to.as_ref())?
362        };
363        // Second, we fill in the rest based on global state.
364        //
365        // Real time report:
366        // N.B.: We don't have a job-level exit hook atm (T76248597), so we use the
367        // CURRENT time -- that we are calling summarize -- as the end time:
368        summary.realtime_elapsed = Some(self.realtime_start.elapsed()?);
369
370        if self.cfg.virtualize_time {
371            let final_time = self.global_time.lock().unwrap();
372            let final_time_ns = final_time.as_nanos();
373            let nanos = self
374                .cfg
375                .epoch
376                .timestamp_nanos_opt()
377                .expect("epoch cannot be represented in a timestamp with nanosecond precision")
378                as u64;
379            let epoch_ns = LogicalTime::from_nanos(nanos);
380            summary.virttime_final = final_time_ns.as_nanos();
381            summary.virttime_elapsed = if final_time_ns.as_nanos() >= epoch_ns.as_nanos() {
382                (final_time_ns - epoch_ns).as_nanos()
383            } else {
384                bail!(
385                    "Internal invariant violated! Global time is before epoch start {}",
386                    epoch_ns
387                );
388            }
389        }
390
391        Ok(summary)
392    }

399    /// A request asks the scheduler to perform an RPC, which includes multiple kinds of
400    /// actions, and, most importantly, permission to acquire resources and run the guest thread.
401    ///
402    /// Irrespective of which method we execute, we can "tick" our local component of the
403    /// global time in the process.
404    type Request = (DetTime, GlobalRequest);
405
406    /// Response from the global portion of the Detcore instrumentation tool.
407    /// The exact form of the response depends on which method was executed.
408    ///
409    /// Irrespective of which method was called, the global handling may have consumed
410    /// logical time, in which case the scheduler can send a new thread-local time back to
411    /// the caller.  Unfortunately, information is lost as this is collapsed to a flat
412    /// scalar instead of a rich `DetTime`.
413    type Response = (Option<LogicalTime>, GlobalResponse);
414
415    /// Called once during startup.
416    async fn init_global_state(cfg: &Config) -> GlobalState {
417        GlobalState::initialize(cfg, true)
418    }

420    async fn receive_rpc(&self, from: Tid, gr: Self::Request) -> Self::Response {
421        type R = GlobalResponse;
422        let dtid = DetTid::from_raw(from.into()); // TODO(T78538674): FIXME
423        let time_from_guest = gr.0.as_nanos();
424
425        // This portion of the time updates "asynchronously", and we can tick it on every rpc:
426        // TODO: eventually the vector clock should be in shared memory, and
427        // the local clocks should update truly asynchrously.  Therefore it
428        // SHOULD be safe to always push through this update on any rpc,
429        // removing the conditional above.
430        self.global_time
431            .lock()
432            .unwrap()
433            .update_global_time(dtid, time_from_guest);
434
435        // RPC boilerplate. (Hard to generate systematically now though, because of the
436        // time payload piggy-backing on each rpc. Maybe eventually once ticking a
437        // threads' own clock happens through shared memory.)
438        #[allow(clippy::unit_arg)]
439        let resp = match gr.1 {
440            GlobalRequest::RequestResources(rs, pid) => {
441                let (response, _endtime) = self.recv_request_resources(from, pid, rs).await;
442                R::RequestResources(response)
443            }
444            GlobalRequest::ReleaseResources(rs) => {
445                R::ReleaseResources(self.recv_release_resources(from, rs).await)
446            }
447            GlobalRequest::ReleaseAllResources => {
448                R::ReleaseAllResources(self.recv_release_all_resources(from).await)
449            }
450            GlobalRequest::MarkPastFirstExecve => {
451                self.past_first_execve.store(true, SeqCst);
452                R::MarkPastFirstExecve(())
453            }
454            // Requested by the parent thread:
455            GlobalRequest::CreateChildThread(dettid, parent_detpid, ctid, flags, priority) => {
456                R::CreateChildThread(
457                    self.recv_create_child_thread(
458                        from,
459                        ChildRegistration {
460                            parent_dettid: DetTid::from_raw(from.into()),
461                            parent_detpid,
462                            child_dettid: dettid,
463                            child_tid_addr: ctid,
464                            flags,
465                            maybe_priority: priority,
466                            parent_is_kernel_blocked: false,
467                        },
468                    )
469                    .await,
470                )
471            }
472            // Requested by the vfork child on behalf of its kernel-blocked parent:
473            GlobalRequest::CreateVforkChildThread(
474                parent_dettid,
475                parent_detpid,
476                child_dettid,
477                ctid,
478                flags,
479                priority,
480            ) => R::CreateChildThread(
481                self.recv_create_child_thread(
482                    from,
483                    ChildRegistration {
484                        parent_dettid,
485                        parent_detpid,
486                        child_dettid,
487                        child_tid_addr: ctid,
488                        flags: Some(flags),
489                        maybe_priority: priority,
490                        parent_is_kernel_blocked: true,
491                    },
492                )
493                .await,
494            ),
495            // Requested by the child thread itself:
496            GlobalRequest::StartNewThread(dettid, detpid) => {
497                R::StartNewThread(self.recv_start_new_thread(from, dettid, detpid).await)
498            }
499            GlobalRequest::DeregisterThread(dettid, detpid, mm, timeslice_stats) => {
500                R::DeregisterThread(
501                    self.recv_deregister_thread(from, dettid, detpid, mm, timeslice_stats)
502                        .await,
503                )
504            }
505            GlobalRequest::FutexAction(dettid, action, futexid, init_read, mask) => R::FutexAction(
506                self.recv_futex_action(from, dettid, action, futexid, init_read, mask)
507                    .await,
508            ),
509            GlobalRequest::DeterminizeInode(ino) => {
510                R::DeterminizeInode(self.recv_determinize_inode(from, ino).await)
511            }
512            GlobalRequest::UnlinkInode(d_ino) => {
513                R::UnlinkInode(self.recv_unlink_inode(from, d_ino).await)
514            }
515            GlobalRequest::TouchFile(ino) => R::TouchFile(self.recv_touch_file(from, ino).await),
516            GlobalRequest::GlobalTimeLowerBound => {
517                let ns = self.global_time.lock().unwrap().as_nanos();
518                R::GlobalTimeLowerBound(ns)
519            }
520            GlobalRequest::TraceSchedEvent(ev, detpid) => {
521                let res = self.recv_trace_schedevent(ev, detpid).await;
522                R::TraceSchedEvent(res)
523            }
524            GlobalRequest::RegisterAlarm(dpid, dtid, secs, sig) => {
525                let remaining = self.recv_register_alarm(dpid, dtid, secs, sig).await;
526                R::RegisterAlarm(remaining)
527            }
528            GlobalRequest::UnrecoverableShutdown => {
529                self.force_shutdown_with_error();
530                R::UnrecoverableShutdown(())
531            }
532            GlobalRequest::RequestPort(open_file_id) => {
533                let mut mut_used_ports = self.used_ports.lock().unwrap();
534                self.update_port_range();
535                let total_available =
536                    self.port_end_range.load(SeqCst) - self.port_start_range.load(SeqCst);
537                let mut index = 0;
538                while (*mut_used_ports).contains(&self.next_port.load(SeqCst))
539                    && index < total_available
540                {
541                    self.next_port.fetch_add(1, SeqCst);
542                    if self.next_port.load(SeqCst) > self.port_end_range.load(SeqCst) {
543                        self.next_port
544                            .store(self.port_start_range.load(SeqCst), SeqCst);
545                    }
546                    index += 1;
547                }
548                if index == total_available {
549                    R::PortFull
550                } else {
551                    (*mut_used_ports).insert(self.next_port.load(SeqCst));
552                    let mut open_file_to_port = self.open_file_to_port.lock().unwrap();
553                    open_file_to_port.insert(open_file_id, self.next_port.load(SeqCst));
554                    R::RequestPort(self.next_port.load(SeqCst))
555                }
556            }
557            GlobalRequest::AddUsedPort(port, open_file_id) => {
558                let mut used_ports = self.used_ports.lock().unwrap();
559                used_ports.insert(port);
560                let mut open_file_to_port = self.open_file_to_port.lock().unwrap();
561                open_file_to_port.insert(open_file_id, port);
562                R::AddUsedPort
563            }
564            GlobalRequest::ReleasePort(open_file_id) => {
565                let mut used_ports = self.used_ports.lock().unwrap();
566                let mut open_file_to_port = self.open_file_to_port.lock().unwrap();
567                let port = open_file_to_port.remove(&open_file_id);
568                if let Some(port) = port {
569                    used_ports.remove(&port);
570                }
571                R::ReleasePort(port)
572            }
573        };
574
575        let time_from_sched = self.global_time.lock().unwrap().threads_time(dtid);
576        let time_update = match time_from_sched.cmp(&time_from_guest) {
577            Ordering::Equal => None,
578            Ordering::Less => {
579                panic!(
580                    "internal error: thread time should never go down, only monotonically up: time in sched {}, thread local time was {}",
581                    time_from_sched, time_from_guest
582                )
583            }
584            Ordering::Greater => Some(time_from_sched),
585        };
586        (time_update, resp)
587    }
588}

590impl GlobalState {
591    async fn recv_request_resources(
592        &self,
593        from: Tid,
594        detpid: DetPid,
595        rs: Resources,
596    ) -> (ResumeStatus, Option<LogicalTime>) {
597        let dettid = DetTid::from_raw(from.into()); // TODO(T78538674): FIXME
598
599        let resp2 = {
600            let mut sched = self.sched.lock().unwrap();
601            let nextturn = sched.next_turns.get(&dettid).unwrap_or_else(|| {
602                panic!("Detcore internal error: no entry for dettid {} in next_turns during resource request.", dettid)
603            }).clone();
604            trace!(
605                "[detcore, dtid {}] ResourceRequest, filling request into {}",
606                &dettid, &nextturn.req
607            );
608            sched.request_put(&nextturn.req, rs.clone(), &self.global_time);
609            nextturn.resp
610        };
611        trace!(
612            "[detcore, dtid {}] waiting on {} for resources: {:?}",
613            dettid, &resp2, rs
614        );
615        let answer = resp2.get().await; // Block on the scheduler allowing our guest to proceed.
616        if let Some((true, process, mm)) = rs.exit_identity() {
617            info!(
618                "Scheduler authorized an exit-group scenario, from dettid {} / detpid {}",
619                dettid, detpid
620            );
621            // Before allowing an `exit_group` to physically proceed, we
622            // deregister the other threads in the thread group to reflect the
623            // fact that they will not receive further logical turns.
624            //
625            // We trust the kernel to physically kill them irrespective of what they're
626            // blocked on, including us having blocked them in the `futex_waiters` list.
627            {
628                let mut sched = self.sched.lock().unwrap();
629                for tid in sched.thread_tree.my_thread_group(&dettid) {
630                    // We don't need to do anything extra for our own thread. That can use the
631                    // same mechanics as a normal exit:
632                    if tid != dettid {
633                        sched.logically_kill_thread(&tid, &process, mm);
634                    }
635                }
636            }
637        }
638
639        match answer {
640            // In this context, SchedValue
641            SchedResponse::Go(Some(schedval)) => {
642                trace!(
643                    "[dtid {}] resources granted, resuming normally: {:?}",
644                    dettid, rs
645                );
646
647                let endtime_update = match schedval {
648                    // Only syscalls timeout, and they don't need to update guest timeslice end.

681    /// Global portion of parent-forking-child protocol.  Called by the parent
682    /// thread for an ordinary clone, or by the child itself for a vfork whose
683    /// parent is blocked inside the kernel (`parent_is_kernel_blocked`).
684    async fn recv_create_child_thread(&self, rpc_sender: Tid, registration: ChildRegistration) {
685        let ChildRegistration {
686            parent_dettid,
687            parent_detpid,
688            child_dettid,
689            child_tid_addr: ctid,
690            flags,
691            maybe_priority,
692            parent_is_kernel_blocked,
693        } = registration;
694        let initial_priority = if let Some(pr) = &self.preemptions_to_replay {
695            assert!(maybe_priority.is_none());
696            let prio = pr
697                .thread_initial_priority(&child_dettid)
698                .unwrap_or_else(|| {
699                    warn!(
700                        "Child thread {} not found in preemption history to replay",
701                        child_dettid
702                    );
703                    DEFAULT_PRIORITY
704                });
705            if !is_ordinary_priority(prio) {
706                panic!(
707                    "Read a bad initial_prority from file: {}\nFull file: {}",
708                    prio,
709                    pr.load_all(),
710                );
711            }
712            prio
713        } else {
714            let prio = maybe_priority.expect(
715                "create_child_thread must take an initial priority unless replaying preemptions",
716            );
717            if !is_ordinary_priority(prio) {
718                panic!(
719                    "recv_create_child_thread received a bad prority argument : {}",
720                    prio,
721                );
722            }
723            prio
724        };
725
726        {
727            let mut sched = self.sched.lock().unwrap();
728
729            // Don't fill in the request, as the child will do it:
730            let _entry = sched
731                .next_turns
732                .entry(child_dettid)
733                .or_insert_with(|| ThreadNextTurn {
734                    dettid: child_dettid,
735                    child_tid_addr: ctid,
736                    req: Ivar::new(),
737                    resp: Ivar::new(),
738                });
739
740            {
741                let is_group_leader = if let Some(f) = flags {
742                    !f.contains(CloneFlags::CLONE_THREAD)
743                } else {
744                    true // root thread
745                };
746                sched
747                    .thread_tree
748                    .add_child(parent_dettid, child_dettid, is_group_leader);
749            }
750
751            if self.cfg.replay_schedule_from.is_none() {
752                // Give the thread an initial priority
753                let old_prio = sched.priorities.insert(child_dettid, initial_priority);
754                assert!(old_prio.is_none());
755            } else {
756                // In replay mode, the context switch point will already have initialized the priority.
757                // UNLESS this is the root thread, in which case we need to fill it in:
758                if let std::collections::btree_map::Entry::Vacant(entry) =
759                    sched.priorities.entry(child_dettid)
760                {
761                    assert_eq!(parent_detpid, ROOT_DETPID);
762                    entry.insert(initial_priority);
763                }
764            }
765
766            if let Some(pr) = &mut sched.preemption_writer {
767                pr.register_thread(child_dettid, initial_priority);
768            }
769
770            let child_first = self.cfg.sequentialize_threads
771                && !parent_is_kernel_blocked
772                && sched.child_runs_first_post_fork(self.cfg.runs_post_fork);
773            let pos = if child_first {
774                sched.runqueue_push_front(child_dettid)
775            } else {
776                sched.runqueue_push_back(child_dettid)
777            };
778            debug!(
779                "[detcore] CreateChildThread with dtid {}: Added child to {} of priority band, position {}.",
780                child_dettid,
781                if child_first { "front" } else { "back" },
782                pos,
783            );
784            sched.started_up.try_put(());
785        }
786        // The child queue position above determines which equal-priority side
787        // gets the first turn when the parent requests ParentContinue.
788        // A vfork parent is already blocked by the kernel and is not in the run
789        // queue, so it must not issue a ParentContinue request here.
790        if self.cfg.sequentialize_threads && !parent_is_kernel_blocked {
791            let mut rs = Resources::new(parent_detpid);
792            rs.insert(
793                ResourceID::ParentContinue {
794                    parent: parent_dettid,
795                    child: child_dettid,
796                },
797                Permission::W,
798            );
799            self.recv_request_resources(rpc_sender, parent_detpid, rs)
800                .await;
801        }
802    }

804    /// Called by the child thread upon startup.
805    /// Returns a thread-preemption history for the new guest thread (if --replay-preemptions-from
806    /// is used).
807    async fn recv_start_new_thread(
808        &self,
809        from: Tid,
810        dettid: DetTid,
811        detpid: DetPid,
812    ) -> Option<ThreadHistory> {
813        let mut tries: u64 = 0;
814        // TODO: eliminate this loop. Could instead signal with an ivar.
815        let response_ivar = loop {
816            yield_once().await;
817            let mut sched = self.sched.lock().unwrap();
818            // The resources that must be held for the fresh thread to run:
819            let rsrcs = {
820                let mut s = HashMap::new();
821                s.insert(ResourceID::MemAddrSpace(detpid), Permission::RW); // TODO(T78055411): track mem aliasing.
822                Resources {
823                    tid: dettid,
824                    resources: s,
825                    poll_attempt: 0,
826                    fyi: String::new(),
827                }
828            };
829            let nextturn = match sched.next_turns.entry(dettid) {
830                Entry::Vacant(_entry) => {
831                    // CreateChildThread on the parent hasn't run yet.
832
833                    // TODO: We could try to populate the entry since we get here
834                    // first, but currently we lack the information right here to
835                    // populate the child_tid_addr field.
836                    if tries == 0 {
837                        trace!(
838                            "[detcore, dtid {}] thread showed up early, no queue entry yet.  Waiting...",
839                            dettid
840                        );
841                    }
842                    tries += 1;
843                    continue;
844                }
845                Entry::Occupied(entry) => {
846                    trace!(
847                        "[detcore, dtid {}] handling StartNewThread rpc.  Found next_turns entry (after {} tries)",
848                        from, tries
849                    );
850                    entry.get().clone()
851                }
852            };
853            sched.request_put(&nextturn.req, rsrcs, &self.global_time);
854            break nextturn.resp;
855        };
856        debug!(
857            "[detcore, dtid {}] New thread will now wait for response on {}...",
858            &dettid, &response_ivar
859        );
860        let _answer = response_ivar.get().await;
861        info!(
862            "[detcore, dtid {}] New thread given go-ahead to proceed via {}",
863            &dettid, &response_ivar
864        );
865        if let Some(pr) = &self.preemptions_to_replay {
866            let history = pr.extract_thread_record(&dettid).unwrap_or_else(|| {
867                warn!(
868                    "Replaying preemptions, but no record found for thread {}",
869                    dettid
870                );
871                ThreadHistory::new()
872            });
873            let old_prio = self
874                .sched
875                .lock()
876                .unwrap()
877                .priorities
878                .insert(dettid, history.initial_priority());
879            debug!(
880                "[replay-preemption] Enqueing new thread at priority {:?} (changed from {:?})",
881                history.initial_priority(),
882                old_prio,
883            );
884            Some(history)
885        } else {
886            None
887        }
888    }

913        &self,
914        _from: Tid,
915        dettid: DetTid,
916        action: FutexAction,
917        futexid: FutexID,
918        _init_read: i32,
919        mask: u32,
920    ) -> Option<SchedValue> {
921        trace!("[detcore, dtid {}] Futex action: {:?}", &dettid, action);
922        let response_iv = {
923            let mut sched = self.sched.lock().unwrap();
924            let resp_iv = sched
925                .next_turns
926                .get(&dettid)
927                .expect("Missing next_turns entry")
928                .resp
929                .clone();
930            match action {
931                FutexAction::WaitRequest(maybe_timeout) => {
932                    sched.sleep_futex_waiter(&dettid, futexid, maybe_timeout, mask);
933                    // block on ivar, below
934                }
935                FutexAction::WaitFinished => {
936                    return None;
937                }
938                FutexAction::WakeRequest(num_threads) => {
939                    let num = sched.wake_futex_waiters(dettid, futexid, num_threads, mask);
940                    return Some(SchedValue::Value(num));
941                }
942                FutexAction::WakeFinished(_num_threads) => {
943                    return None;
944                }
945            }
946            // Blocking on the FUTEX_WAIT here, remove ourselves:
947            assert!(sched.run_queue.remove_tid(dettid));
948            resp_iv
949        };
950        // Wait for wake+scheduler response.
951        match response_iv.get().await {
952            SchedResponse::Go(answer) => {
953                trace!(
954                    "[detcore, dtid {}] Unblocked from futex_wait! ({})",
955                    &dettid, &response_iv
956                );
957                answer
958            }
959            SchedResponse::Signaled() => Some(SchedValue::Value(nix::errno::Errno::EINTR as u64)),
960        }
961    }

1289/// Global method RPC to request to control a resource.
1290///
1291/// Blocking: future returns only when resources are fully acquired.
1292pub async fn resource_request<G, T>(guest: &mut G, r: Resources) -> ResumeStatus
1293where
1294    G: Guest<Detcore<T>>,
1295    T: RecordOrReplay,
1296{
1297    if guest.config().sequentialize_threads {

1301            "[detcore, dtid {}] BLOCKING on resource_request rpc... {:?}",
1302            &dettid, r
1303        );

1305            send_and_update_time(guest, GlobalRequest::RequestResources(r.clone(), detpid)).await;
1306        match resp.1 {
1307            GlobalResponse::RequestResources(x) => {
1308                trace!(
1309                    "[detcore, dtid {}] UNBLOCKED, acquired resources: {:?}",
1310                    &dettid, r
1311                );
1312                x
1313            }
1314            _ => unreachable!(),
1315        }
1316    } else {
1317        ResumeStatus::Normal
1318    }

1370/// Global method RPC for the parent to add a child-thread to the round-robin pool.
1371///
1372/// Nonblocking: future returning does not guarantee anything about the central scheduler,
1373/// except that it will eventually give a slot to the child.  Then the protocol is that
1374/// child will subsequently make a `thread_start_request` to gate the start of its execution.
1375pub async fn create_child_thread<G, T>(
1376    guest: &mut G,
1377    child_dettid: DetTid,
1378    ctid: usize,
1379    flags: Option<CloneFlags>,
1380) where
1381    G: Guest<Detcore<T>>,
1382    T: RecordOrReplay,
1383{
1384    // Random (or replayed) starting priority in chaos mode, constant priority otherwise.
1385    let starting_priority = if guest.config().replay_preemptions_from.is_some() {
1386        // In preemption replay mode, the initial priority is set on the other
1387        // side of the rpc, in recv_create_child_thread.
1388        None
1389    } else if guest.config().replay_schedule_from.is_some() {
1390        // FIXME!  Find a cleaner way to make the root thread start off high-priority:
1391        if child_dettid <= DetTid::from_raw(3) {
1392            Some(REPLAY_FOREGROUND_PRIORITY)
1393        } else {
1394            Some(REPLAY_DEFERRED_PRIORITY)
1395        }
1396    } else if guest.config().chaos {
1397        let entropy = guest
1398            .thread_state_mut()
1399            .chaos_prng_next_u64("child_priority");
1400        if guest.config().chaos_target_races {
1401            // Targeted chaos: bias a freshly created child to an extreme priority
1402            // so it either runs before the parent resumes or strictly after it,
1403            // instead of landing at a uniformly random priority. This maximizes
1404            // parent/child ordering divergence to surface fork/exec races.
1405            // Reproducible under `--fuzz-seed`/`--sched-seed`.
1406            if entropy.is_multiple_of(2) {
1407                Some(FIRST_PRIORITY)
1408            } else {
1409                Some(LAST_PRIORITY)
1410            }
1411        } else {
1412            Some(entropy_to_priority(entropy))
1413        }
1414    } else {
1415        Some(DEFAULT_PRIORITY)
1416    };
1417
1418    let detpid = guest.thread_state().detpid.expect("detpid unset");
1419
1420    let resp = send_and_update_time(
1421        guest,
1422        GlobalRequest::CreateChildThread(child_dettid, detpid, ctid, flags, starting_priority),
1423    )
1424    .await;
1425    match resp.1 {
1426        GlobalResponse::CreateChildThread(x) => x,
1427        _ => unreachable!(),
1428    }
1429}

1431/// Register a vfork child while its parent is blocked inside `clone(2)`.
1432///
1433/// Unlike an ordinary clone, the parent cannot perform this registration
1434/// because the kernel has blocked it until the child execs or exits. The child
1435/// therefore registers itself, carrying the inherited parent identity, flags,
1436/// and starting priority. The starting priority is derived the same way as an
1437/// ordinary clone so that chaos and replay scheduling stay deterministic.
1438pub async fn create_vfork_child_thread<G, T>(
1439    guest: &mut G,
1440    child_dettid: DetTid,
1441    vfork: crate::tool_local::PendingVfork,
1442) where
1443    G: Guest<Detcore<T>>,
1444    T: RecordOrReplay,
1445{
1446    let starting_priority = if guest.config().replay_preemptions_from.is_some() {
1447        None
1448    } else if guest.config().replay_schedule_from.is_some() {
1449        Some(if child_dettid <= DetTid::from_raw(3) {
1450            REPLAY_FOREGROUND_PRIORITY
1451        } else {
1452            REPLAY_DEFERRED_PRIORITY
1453        })
1454    } else if guest.config().chaos {
1455        Some(entropy_to_priority(vfork.child_priority_entropy.expect(
1456            "vfork child priority entropy missing in chaos mode",
1457        )))
1458    } else {
1459        // POSIX vfork suspends the parent until the child execs or _exits. Give
1460        // the vfork child a strictly higher priority (lower number) than the
1461        // parent's DEFAULT_PRIORITY so the deterministic scheduler always runs
1462        // the child first, rather than round-robining the parent and child at
1463        // equal priority (which leaves fork/exec ordering nondeterministic).
1464        Some(DEFAULT_PRIORITY - 1)
1465    };
1466
1467    let resp = send_and_update_time(
1468        guest,
1469        GlobalRequest::CreateVforkChildThread(
1470            vfork.parent_dettid,
1471            vfork.parent_detpid,
1472            child_dettid,
1473            vfork.child_tid_addr,
1474            vfork.flags,
1475            starting_priority,
1476        ),
1477    )
1478    .await;
1479    match resp.1 {
1480        GlobalResponse::CreateChildThread(x) => x,
1481        _ => unreachable!(),
1482    }
1483}

1653/// Record an event in the schedule trace, OR check the event on replay.
1654/// This also prints the backtrace of the schedevent, if indicated.
1655///
1656/// Arguments:
1657/// - tag_end_rip: read the current guest registers to fill in the `end_rip` on the event with the
1658///   current instruction pointer.
1659pub async fn trace_schedevent<G, T>(guest: &mut G, ev: SchedEvent, tag_end_rip: bool)
1660where
1661    G: Guest<Detcore<T>>,
1662    T: RecordOrReplay,
1663{
1664    assert!(guest.config().sequentialize_threads);
1665
1666    // trace_schedevent is called AFTER the event is complete, and the rip is resting just after it.
1667    let ev = if tag_end_rip {
1668        let end_rip = if let Some(r) = ev.end_rip {
1669            r
1670        } else {
1671            let regs = guest.regs().await;
1672            NonZeroUsize::new(regs.rip.try_into().unwrap()).unwrap()
1673        };
1674        SchedEvent {
1675            end_rip: Some(end_rip),
1676            ..ev
1677        }
1678    } else {
1679        ev
1680    };
1681
1682    if let Some(rip) = ev.end_rip {
1683        let rip_addr = AddrMut::<u16>::from_raw(rip.into()).unwrap();
1684        let rip_contents: u16 = guest
1685            .memory()
1686            .read_value(rip_addr)
1687            .expect("memory read succeeds");
1688        trace!(
1689            "Tracing sched event, after which rip is {}, next two instruction bytes {:#06x}",
1690            rip, rip_contents
1691        );
1692    }
1693
1694    let detpid = guest.thread_state().detpid.expect("detpid unset");
1695    let resp = send_and_update_time(guest, GlobalRequest::TraceSchedEvent(ev, detpid)).await;
1696
1697    trace!("trace_schedevent result: {:?}", resp);
1698    match resp {
1699        (
1700            _,
1701            GlobalResponse::TraceSchedEvent(TraceSchedEventResponse {
1702                print_stack_strace,
1703                timeslice,
1704            }),
1705        ) => {
1706            if let Some(m_path) = print_stack_strace {
1707                trace!("[trace_schedevent] writing stacktrace via Reverie...");
1708                write_backtrace(guest, m_path.as_ref());
1709            }
1710
1711            if let Some(timeslice) = timeslice
1712                && guest.thread_state().past_global_first_execve
1713            {
1714                let end_of_timeslice =
1715                    guest.thread_state().thread_logical_time.as_nanos() + timeslice;
1716                trace!(
1717                    "[detcore][dettid {}] setting end_of_timeslice to {:?} as instructed by replayer",
1718                    guest.thread_state().dettid,
1719                    end_of_timeslice
1720                );
1721                guest.thread_state_mut().end_of_timeslice = Some(end_of_timeslice);
1722            }
1723        }
1724        _ => {
1725            unreachable!()
1726        }
1727    }
1728}
