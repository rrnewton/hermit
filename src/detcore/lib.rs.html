9//! Detcore is a Reverie tool that determinizes the execution of a process.
10//!
11//! # Backend-abstraction commandment
12//!
13//! Detcore is a *tool* written against Reverie's **abstract** instrumentation
14//! interface (the `reverie` crate). It depends only on those traits and types
15//! and is deliberately ignorant of how a guest is actually traced.
16//!
17//! Detcore MUST NEVER depend on or import a concrete Reverie backend --
18//! `reverie-ptrace`, `reverie-dbi`, or `reverie-kvm`. Choosing and
19//! instantiating a backend, and running a detcore tool against it, is the sole
20//! responsibility of the `hermit-cli` package. There are no backend-specific
21//! hacks in detcore: any tracing-mechanism-specific behavior belongs behind the
22//! Reverie abstraction, not here.
23//!
24//! Why: Hermit follows Reverie's abstract model. A backend dependency in
25//! detcore would couple the determinism engine to one tracing mechanism and
26//! break the clean abstraction boundary that lets the same tool run over any
27//! backend.
28//!
29//! The one allowed exception is test-only: detcore's own integration tests
30//! (under `detcore/tests/`, wired via the `reverie-ptrace` **dev-dependency**)
31//! drive a real tracer to exercise the tool. That coupling never reaches the
32//! shipped library. This invariant is enforced in CI by
33//! `scripts/check-detcore-backend-abstraction.sh`.

168    /// Defense-in-depth determinism for the registers the syscall instruction
169    /// clobbers.
170    ///
171    /// On x86-64 the `syscall` instruction destroys `%rcx` (which the CPU loads
172    /// with the return instruction pointer) and `%r11` (the saved `RFLAGS`).
173    /// After a syscall these are architecturally "undefined", so hermit must not
174    /// assume a well-behaved guest ignores them: a misbehaving guest that reads
175    /// `%rcx`/`%r11` must still observe deterministic values. Reverie's
176    /// injected-syscall path can otherwise leave its *private trampoline page's*
177    /// RIP/RFLAGS in these registers, which is both nondeterministic and an
178    /// information leak of tracer internals.
179    ///
180    /// This forces both registers to the guest's own (deterministic) RIP and
181    /// RFLAGS, which is exactly what a faithful `SYSRET` would leave there. It is
182    /// a no-op when they already hold the canonical values (the common path), so
183    /// it only writes registers when something diverged. Register-preserved
184    /// arguments (`%rdi`..`%r9`, callee-saved) are deliberately left untouched:
185    /// the Linux ABI preserves them, so zeroing them would break faithful,
186    /// well-behaved programs.
187    #[cfg(target_arch = "x86_64")]
188    async fn canonicalize_syscall_clobbers<G: Guest<Self>>(&self, guest: &mut G) {
189        let mut regs = guest.regs().await;
190        // A faithful SYSRET leaves the return RIP in %rcx and RFLAGS in %r11.
191        if regs.rcx != regs.rip || regs.r11 != regs.eflags {
192            regs.rcx = regs.rip;
193            regs.r11 = regs.eflags;
194            if let Err(err) = guest.set_regs(regs).await {
195                // Best-effort: some backends cannot write registers. Do not fail
196                // the syscall over a defense-in-depth hardening step.
197                debug!(
198                    "canonicalize_syscall_clobbers: set_regs unsupported/failed: {}",
199                    err
200                );
201            }
202        }
203    }

209    /// Update logical thread time with any outstanding ticks of the Reverie clock.  Returns a list
210    /// of corresponding Branch/OtherInstructions events if schedule recording is enabled.
211    ///
212    /// # Arguments
213    ///
214    /// * `precise_branch`: if true, there were no non-branch instructions since the last recorded branch instruction.
215    async fn update_logical_time_rcbs<G: Guest<Self>>(
216        &self,
217        guest: &mut G,
218        precise_branch: bool,
219    ) -> Option<Vec<SchedEvent>> {
220        if self.cfg.use_rcb_time() {
221            let precise_timers = !guest.config().imprecise_timers;
222            // TODO(T86591083): we might need to not always increment as a hack fix
223            // for deterministic virtual time without sequentialize threads.
224            let clock_value = guest.read_clock().expect("Couldn't read clock");
225            // N.B. clock_value does not yet include any updates for the inbound
226            // syscall/instruction because this function is the very first thing that
227            // happens in each type of handler.
228            let thread_state = guest.thread_state_mut();
229            let dettid = thread_state.dettid;
230            assert!(thread_state.committed_clock_value <= clock_value);
231            let delta_rcbs: u64 = clock_value - thread_state.committed_clock_value;
232            thread_state.thread_logical_time.add_rcbs(delta_rcbs);
233            thread_state.committed_clock_value = clock_value;
234
235            if thread_state.end_of_timeslice.is_some() {
236                if let Some(last_timer) = thread_state.last_rcb_timer
237                    && delta_rcbs > last_timer
238                    && precise_timers
239                {
240                    panic!(
241                        "prehook: Missed expected preemption! Clock_value: {}. Stepped forward {:?} RCBs, but should have trapped at {:?}",
242                        clock_value, delta_rcbs, last_timer
243                    );
244                    // TODO: turn the above panic into a warning again if we see any weirdness
245                    // on certain platforms.
246                    // We can repair the state and keep going, using the below:
247                    /*
248                    thread_state.end_of_timeslice = None;
249                    thread_state.last_rcb_timer = None;
250                    thread_state.next_timeslice(&self.cfg);
251                    */
252                }
253                // Otherwise we're very early, at the prehook of handle_thread_start.
254            } else {
255                panic!(
256                    "Invariant violation: end_of_timeslice is None during update_logical_time_rcbs..."
257                )
258            }
259
260            trace!(
261                "[dtid {}] updated rcb clock, new logical time: {:?}, i.e. {}, timeslice end: {}, local rcb clock_value {:?}",
262                dettid,
263                &thread_state.thread_logical_time,
264                thread_state.thread_logical_time.as_nanos(),
265                thread_state
266                    .end_of_timeslice
267                    .map_or_else(|| "".to_string(), |x| format!("{}", x)),
268                clock_value,
269            );
270            if self.cfg.should_trace_schedevent() {
271                let mut vec = Vec::new();
272                let ev = with_guest_time(
273                    guest,
274                    SchedEvent::branches(
275                        dettid,
276                        delta_rcbs
277                            .try_into()
278                            .expect("should not have more than 2^32 branches at once"),
279                    ),
280                );
281                let ev = if precise_branch {
282                    with_guest_rip(guest, ev).await
283                } else {
284                    ev
285                };
286
287                if delta_rcbs > 0 {
288                    // We don't fill the end_rip here, because the current rip is NOT precisely the
289                    // end of this block of branch events.  Other instructions may have occured
290                    // since the last branch.
291                    vec.push(ev)
292                } else {
293                    trace!(
294                        "[detcore, dtid {}] Refusing to record zero-braches event: {:?}",
295                        &ev.dettid, ev
296                    );
297                }
298                if !precise_branch {
299                    // This will ALWAYS record, even if the branches above are zero.
300                    let ev2 = with_guest_time(
301                        guest,
302                        SchedEvent {
303                            dettid,
304                            op: Op::OtherInstructions,
305                            count: 1,
306                            start_rip: None,
307                            end_rip: None,
308                            end_time: None,
309                        },
310                    );
311                    // Fill in end_rip because current rip represents the end of this event.
312                    let ev2 = with_guest_rip(guest, ev2).await;
313                    vec.push(ev2);
314                }
315                Some(vec)
316            } else {
317                None
318            }
319        } else {
320            None
321        }
322    }

324    /// A common hook called at the start of *every* handler, just after we receive
325    /// control from the guest.
326    async fn pre_handler_hook<G: Guest<Self>>(&self, guest: &mut G, precise_branch: bool) {
327        let dettid = guest.thread_state().dettid;
328        let evs = self.update_logical_time_rcbs(guest, precise_branch).await;
329
330        if guest.thread_state().guest_past_first_execve() {
331            detlog_debug!(
332                "(pre) registers [dtid {}][rcbs {}]. {}",
333                dettid,
334                guest.thread_state().thread_logical_time.rcbs(),
335                guest.regs().await.display()
336            );
337        }
338        trace!(
339            "prehook [dtid {}] Updating rcbs and checking time remaining.",
340            dettid
341        );
342        let thread_state = guest.thread_state();
343        if let Some(slice_end) = thread_state.end_of_timeslice {
344            let now_ns = thread_state.thread_logical_time.as_nanos();
345            if slice_end < now_ns {
346                trace!(
347                    "prehook [dtid {}] Time {} is beyond end of slice {}",
348                    dettid, now_ns, slice_end
349                );
350                self.end_timeslice(guest).await;
351            }
352        }
353
354        if let Some(vec) = evs {
355            for ev in vec {
356                trace_schedevent(guest, ev, false).await;
357            }
358        }
359    }

361    /// A common hook called at the end of *every* handler, just before returning control
362    /// to the guest. This resets the preemption timeout for the next timeslice.
363    ///
364    /// However, note that the thread's timeslice (turn) may have expired DURING this handler.
365    /// Therefore the timeslice can end in the posthook as well as in the prehook.
366    async fn post_handler_hook<G: Guest<Self>>(&self, guest: &mut G) {
367        let dettid = guest.thread_state().dettid;
368        let timeout_delta_ns = guest.config().preemption_timeout;
369        let current_time = guest.thread_state_mut().thread_logical_time.as_nanos();
370
371        // If the preemption timeout is set, then end_of_timeslice must always be.
372        if let Some(slice_end) = guest.thread_state().end_of_timeslice {
373            let mut slice_end = slice_end;
374            assert!(timeout_delta_ns.is_some());
375            // TODO: get rid of fractional NANOS_PER_RCB so it's clear that this does not lose precision:
376            let epsilon = Duration::from_nanos(NANOS_PER_RCB as u64);
377
378            // If there is less than a single RCB worth of time left... we call it.
379            if current_time + epsilon > slice_end {
380                // Our timeslice wasn't over yet at the start of the handler.  But the
381                // elapse of time during the handler (e.g. executing a logical syscall)
382                // has pushed us over the threshold to where our time is used up.
383                // Therefore, we don't let the guest/mutator proceed executing more code yet.
384                // Rather, we go back to the scheduler and wait for our turn again.
385                trace!(
386                    "posthook [dtid {}] Time {} beyond (or close enough to) end of slice {}! Ending slice.",
387                    dettid, current_time, slice_end
388                );
389                let current_slice = guest.thread_state().stats.timeslice_count;
390                self.end_timeslice(guest).await;
391                // After ending time slice, need to reread this:
392                slice_end = guest
393                    .thread_state()
394                    .end_of_timeslice
395                    .expect("internal invariant");
396                let current_time = guest.thread_state().thread_logical_time.as_nanos();
397                trace!(
398                    "posthook [dtid {}] after ending timeslice T{}, next end is {}, current time {}",
399                    dettid, current_slice, slice_end, current_time,
400                );
401            }
402            if current_time + epsilon > slice_end {
403                panic!(
404                    "Unhandled! Ended time slice, but current time {} is still beyond (or too near) timeslice end {}",
405                    current_time, slice_end
406                );
407            }
408
409            let ns_remaining = slice_end - current_time;
410            let mut rcbs_remaining = ns_remaining.into_rcbs();
411
412            if !guest.thread_state().interrupt_at.is_empty() {
413                let current_rcbs = guest.thread_state().thread_logical_time.rcbs();
414
415                if let Some(next_interrupt) = guest
416                    .thread_state()
417                    .interrupt_at
418                    .range((current_rcbs + 1)..)
419                    .next()
420                {
421                    rcbs_remaining = next_interrupt - current_rcbs;
422
423                    debug!(
424                        "[dtid: {}] current rcbs: {}, next interrupt_at: {}",
425                        dettid, current_rcbs, next_interrupt
426                    )
427                }
428            }
429
430            trace!(
431                "posthook [dtid {}] After time consumed by handler/injection, {} remaining in slice ({} rcbs).",
432                dettid, ns_remaining, rcbs_remaining,
433            );
434
435            if ns_remaining.is_zero() {
436                panic!(
437                    "Timer invariant broken: we should not exit a handler with 0 timeslice remaining."
438                );
439            }
440            assert!(rcbs_remaining > 0);
441            trace!(
442                "posthook [dtid {}] Resetting timer to {:?} RCBs in the future (current {})",
443                dettid,
444                rcbs_remaining,
445                guest.thread_state().thread_logical_time.rcbs()
446            );
447            guest.thread_state_mut().last_rcb_timer = Some(rcbs_remaining);
448
449            if guest.config().imprecise_timers {
450                guest
451                    .set_timer(TimerSchedule::Rcbs(rcbs_remaining))
452                    .expect("Failed to set timer");
453            } else {
454                guest
455                    .set_timer_precise(TimerSchedule::Rcbs(rcbs_remaining))
456                    .expect("Failed to set timer");
457            }
458        } else {
459            assert!(timeout_delta_ns.is_none());
460        }
461
462        if guest.thread_state().guest_past_first_execve() {
463            detlog_debug!(
464                "(post) registers [dtid {}][rcbs {}]. {}",
465                dettid,
466                guest.thread_state().thread_logical_time.rcbs(),
467                guest.regs().await.display(),
468            );
469        }
470    }

485    async fn end_timeslice_with_sched_yield<G: Guest<Self>>(
486        &self,
487        guest: &mut G,
488        explicit_sched_yield: bool,
489    ) {
490        let thread_state = guest.thread_state();
491        let dettid = thread_state.dettid;
492        let chaos = guest.config().chaos;
493        let end_time = thread_state.thread_logical_time.as_nanos();
494        info!(
495            "[detcore, dtid {}] ending timeslice T{}. {} syscalls and {} signals this timeslice.",
496            dettid,
497            thread_state.stats.timeslice_count,
498            thread_state.stats.timeslice_syscall_count,
499            thread_state.stats.timeslice_signal_count,
500        );
501        let maybe_prio = guest.thread_state_mut().next_timeslice(&self.cfg); // Reset end_of_timeslice
502
503        // Depending on chaos mode, a received timer event is either a preemption or a changepoint
504        let req = if let Some(prio) = maybe_prio {
505            Self::priority_changepoint_request(guest, end_time, prio)
506        } else if chaos {
507            Self::random_priority_changepoint_request(guest, end_time)
508        } else if explicit_sched_yield && self.cfg.replay_schedule_from.is_none() {
509            Self::sched_yield_request(guest)
510        } else {
511            Self::yield_request(guest)
512        };
513        resource_request(guest, req).await;
514    }

516    fn detlog_memory_maps<G: Guest<Self>>(&self, guest: &mut G) -> Result<(), reverie::Error> {
517        if !(self.cfg.detlog_stack || self.cfg.detlog_heap) {
518            // Don't incur the *significant* performance penalty for reading
519            // /proc/maps unless one of these flags is enabled.
520            return Ok(());
521        }
522        for mmap in procmaps::from_pid(guest.pid(), |map| match map.pathname {
523            procmaps::MMapPath::Stack if self.cfg.detlog_stack => true,
524            procmaps::MMapPath::Heap if self.cfg.detlog_heap => true,
525            _ => false,
526        })? {
527            let dettid = guest.thread_state().dettid;
528            detlog!(
529                "[memory][dtid {}] {}->{}",
530                dettid,
531                procmaps::display(&mmap),
532                procmaps::compute_hash(guest, &mmap)?
533            )
534        }
535        Ok(())
536    }

881    fn init_thread_state(
882        &self,
883        tid: Tid,
884        parent: Option<(Tid, &Self::ThreadState)>,
885    ) -> Self::ThreadState {
886        trace!("[tid {}] detcore init new thread state", tid);
887
888        let record_or_replay = self
889            .record_or_replay
890            .init_thread_state(tid, parent.map(|(ptid, ts)| (ptid, ts.as_ref())));
891
892        // TODO(T78538674): virtualize tid, extend tid<=>dettid mapping here.
893        match parent {
894            None => ThreadState::new(DetPid::from_raw(tid.into()), &self.cfg, record_or_replay),
895            Some(pts) => {
896                let clone_flags = pts
897                    .1
898                    .clone_flags
899                    .expect("clone_flags must be set by parent");
900                let dettid = DetPid::from_raw(tid.into());
901
902                // If we had mutable access to the parent state, we could update it here, but
903                // instead we leave that to the clone/fork handling.
904                let (child_pedigree, _parent) = pts.1.pedigree.fork();
905
906                ThreadState {
907                    dettid,
908                    detpid: None, // Initialized later.
909                    mm_id: MmId::for_clone(
910                        pts.1.mm_id,
911                        dettid,
912                        clone_flags.contains(CloneFlags::CLONE_VM),
913                    ),
914                    memory_metadata: if clone_flags.contains(CloneFlags::CLONE_VM) {
915                        Arc::clone(&pts.1.memory_metadata)
916                    } else {
917                        Arc::new(Mutex::new(
918                            pts.1
919                                .memory_metadata
920                                .lock()
921                                .expect("memory metadata mutex poisoned")
922                                .clone(),
923                        ))
924                    },
925                    pedigree: child_pedigree,
926                    stats: ThreadStats::new(),
927                    file_metadata: {
928                        debug!(
929                            "[init_thread-state, parent dtid = {}] child thread {}, clone_flags = {:x?}",
930                            pts.0, tid, clone_flags
931                        );
932                        if clone_flags.contains(CloneFlags::CLONE_FILES) {
933                            pts.1.file_metadata.clone()
934                        } else {
935                            Arc::new(Mutex::new(
936                                pts.1.file_metadata.lock().unwrap().fork_for(dettid),
937                            ))
938                        }
939                    },
940                    // POSIX timers are shared among threads of a process but are
941                    // NOT inherited across fork(2). Share the table for a new
942                    // thread (CLONE_THREAD); give a new process a fresh, empty
943                    // one.
944                    posix_timers: if clone_flags.contains(CloneFlags::CLONE_THREAD) {
945                        Arc::clone(&pts.1.posix_timers)
946                    } else {
947                        Arc::new(Mutex::new(PosixTimers::default()))
948                    },
949                    // Resource limits are process state: threads share them,
950                    // while a forked process inherits a snapshot.
951                    resource_limits: if clone_flags.contains(CloneFlags::CLONE_THREAD) {
952                        Arc::clone(&pts.1.resource_limits)
953                    } else {
954                        Arc::new(Mutex::new(
955                            pts.1
956                                .resource_limits
957                                .lock()
958                                .expect("resource limits mutex poisoned")
959                                .clone(),
960                        ))
961                    },
962                    clone_flags: None,
963                    pending_vfork: pts.1.pending_vfork.clone(),
964
965                    // For a child thread, we use the parent to initialize our rng state:
966                    prng: thread_rng_from_parent("USER RAND", &pts.1.prng, dettid),
967                    chaos_prng: thread_rng_from_parent("CHAOSRAND", &pts.1.chaos_prng, dettid),
968
969                    // For comparing progress to other threads, it is important that our
970                    // child thread start at a sensible place, rather than starting back
971                    // at zero:
972                    thread_logical_time: pts.1.thread_logical_time.clone(),
973                    // A new thread gets a new clock, so we've committed 0 ticks
974                    committed_clock_value: 0,
975
976                    end_of_timeslice: None,
977                    last_rcb_timer: None,
978
979                    record_or_replay,
980                    preemption_points: None,
981
982                    // We only get to the point of creating child threads if we're past the first execve.
983                    past_global_first_execve: true,
984                    interrupt_at: self.cfg.interrupts_for_thread(dettid),
985                }
986            }
987        }
988    }

990    async fn handle_thread_start<G: Guest<Self>>(&self, guest: &mut G) -> Result<(), Error> {
991        let detpid = DetPid::from_raw(guest.pid().into());
992        trace!(
993            "[tid {}] detcore handle_thread_start, pid={}",
994            guest.tid(),
995            detpid
996        );
997
998        // Delayed initialization of thread_state for this new thread:
999        guest.thread_state_mut().detpid = Some(detpid);
1000
1001        let new_dettid = DetTid::from_raw(guest.tid().into()); // TODO(T78538674): virtualize pid/tid:
1002        assert_eq!(new_dettid, guest.thread_state().dettid);
1003
1004        if guest.is_root_thread() {
1005            // There is no fork event to catch for the root thread.
1006            debug!(
1007                "[detcore, dtid {}] root thread start, scheduling.. full config:\n {:?}",
1008                &new_dettid,
1009                guest.config()
1010            );
1011            create_child_thread(guest, new_dettid, 0, None).await;
1012        } else if let Some(vfork) = guest.thread_state_mut().pending_vfork.take() {
1013            create_vfork_child_thread(guest, new_dettid, vfork).await;
1014        }
1015
1016        // Except for the root task, let's block until it's our turn to go:
1017        let th = tool_global::thread_start_request(&self.cfg, guest, self.detpid).await;
1018
1019        // Finish the delayed initialization of the full threadstate:
1020        {
1021            let ts = guest.thread_state_mut();
1022            ts.preemption_points = th.map(|x| x.into_iter());
1023            ts.next_timeslice(&self.cfg); // Must be after preemption_points is set.
1024        }
1025
1026        // The prehook is a noop for a thread just starting.  Can't end the timeslice.  There's no
1027        // RCB progress to record.  However, we call it for consistency with all the other handlers.
1028        self.pre_handler_hook(guest, true).await;
1029        // ^ precise_branch=true: There should have been ZERO prior instructions before this,
1030        // because the thread hasn't done anything yet.
1031
1032        self.record_or_replay
1033            .handle_thread_start(&mut guest.into_guest())
1034            .await?;
1035
1036        self.post_handler_hook(guest).await;
1037        Ok(())
1038    }

1040    async fn handle_post_exec<G: Guest<Self>>(&self, guest: &mut G) -> Result<(), Errno> {
1041        guest.thread_state_mut().past_global_first_execve = true;
1042        tool_global::mark_past_first_execve(guest).await;
1043        self.pre_handler_hook(guest, false).await;
1044
1045        if let Some(ptr) = guest.auxv().at_random() {
1046            // It is safe to mutate this address since libc has not yet had a
1047            // chance to modify or copy the auxv table.
1048            let bytes: [u8; 16] = guest.thread_state_mut().thread_prng().random();
1049            detlog!(
1050                "[post_exec, dtid {}] init auxv AT_RANDOM value to {:?}",
1051                guest.thread_state().dettid,
1052                bytes
1053            );
1054            let ptr = unsafe { ptr.into_mut() };
1055            guest.memory().write_value(ptr, &bytes)?;
1056        }
1057
1058        self.post_handler_hook(guest).await;
1059        Ok(())
1060    }

1101    async fn handle_syscall_event<G: Guest<Self>>(
1102        &self,
1103        guest: &mut G,
1104        call: Syscall,
1105    ) -> Result<i64, Error> {
1106        self.pre_handler_hook(guest, false).await;
1107
1108        let dettid = guest.thread_state().dettid;
1109
1110        if guest.thread_state().guest_past_first_execve() {
1111            detlog!(
1112                "[syscall][detcore, dtid {}] inbound syscall: {} = ?",
1113                dettid,
1114                call.display(&guest.memory())
1115            );
1116        }
1117
1118        let config = guest.config().clone(); // TODO/FIXME: this is an inefficient and unnecessary copy
1119
1120        if config.sequentialize_threads && self.cfg.should_trace_schedevent() {
1121            trace_schedevent(
1122                guest,
1123                with_guest_time(
1124                    guest,
1125                    SchedEvent::syscall(dettid, call.number(), SyscallPhase::Prehook),
1126                ),
1127                true,
1128            )
1129            .await;
1130        }
1131
1132        let virtualize_time = config.virtualize_time;
1133        let new_count = {
1134            // which results from not being able to borrow guest twice.
1135            let thread_state = guest.thread_state_mut();
1136            thread_state.stats.count_syscall();
1137
1138            // Add the syscall to our thread's logical progress, advancing logical time.
1139            if config.sequentialize_threads {
1140                thread_state.thread_logical_time.add_syscall();
1141            } else {
1142                let bumpit = matches!(
1143                    call,
1144                    Syscall::Gettimeofday(_)
1145                        | Syscall::Time(_)
1146                        | Syscall::ClockGettime(_)
1147                        | Syscall::Write(_)
1148                        | Syscall::Writev(_)
1149                );
1150                // TODO(T86591083): remove this conditional to bump logical time unconditionally:
1151                if bumpit && virtualize_time {
1152                    thread_state.thread_logical_time.add_syscall();
1153                }
1154            }
1155            thread_state.stats.syscall_count
1156        };
1157
1158        let res = match classify_syscall(call.number()) {
1159            // Rseq is not type-safe in the pinned Reverie revision. Dispatch by Sysno so a
1160            // future typed representation preserves this explicit policy.
1161            SyscallClassification::Determinized if call.number() == Sysno::rseq => {
1162                if config.panic_on_unsupported_syscalls {
1163                    Err(Error::Errno(Errno::ENOSYS))
1164                } else {
1165                    self.passthrough(guest, call).await
1166                }
1167            }

1169                Syscall::Write(w) => self.handle_write(guest, w).await,
1170                // AUTONOMOUS-BOT-IMPLEMENTED
1171                // TODO-HUMAN-REVIEW(#547)
1172                Syscall::Writev(w) => self.handle_writev(guest, w).await,
1173                Syscall::Openat(o) => self.handle_openat(guest, o).await,
1174                Syscall::Open(o) => self.handle_openat(guest, o.into()).await,
1175                Syscall::Creat(o) => self.handle_openat(guest, o.into()).await,
1176                Syscall::Close(s) => self.handle_close(guest, s).await,
1177                Syscall::Read(s) => self.handle_read(guest, s).await,
1178                Syscall::Pread64(s) => self.handle_pread64(guest, s).await,
1179                // This syscall is advisory; fixed success preserves its API contract.
1180                Syscall::Fadvise64(_) => Ok(0),
1181                Syscall::Mmap(s) => self.handle_mmap(guest, s).await,
1182                Syscall::Madvise(s) => self.handle_madvise(guest, s).await,
1183                Syscall::Munmap(s) => self.handle_munmap(guest, s).await,
1184                Syscall::Mremap(s) => self.handle_mremap(guest, s).await,
1185                Syscall::Stat(s) => self.handle_stat_family(guest, s.into()).await,
1186                Syscall::Lstat(s) => self.handle_stat_family(guest, s.into()).await,
1187                Syscall::Fstat(s) => self.handle_stat_family(guest, s.into()).await,
1188                Syscall::Newfstatat(s) => self.handle_stat_family(guest, s.into()).await,
1189                Syscall::Statx(s) => self.handle_statx(guest, s).await,
1190                Syscall::Fcntl(s) => self.handle_fcntl(guest, s).await,
1191                Syscall::Ioctl(s) => self.handle_ioctl(guest, s).await,
1192                Syscall::Futex(s) => self.handle_futex(guest, s).await,
1193
1194                Syscall::Clone(s) => self.handle_clone_family(guest, s.into()).await,
1195                Syscall::Clone3(s) => self.handle_clone_family(guest, s.into()).await,
1196                Syscall::Fork(s) => self.handle_clone_family(guest, s.into()).await,
1197
1198                // Forward vfork as vfork (rather than rewriting to fork) so the
1199                // kernel enforces the CLONE_VFORK parent-blocking contract while the
1200                // child registers itself and runs to exec/exit.
1201                Syscall::Vfork(s) => self.handle_clone_family(guest, s.into()).await,
1202                Syscall::Wait4(s) => self.handle_wait4(guest, s).await,
1203                Syscall::Waitid(s) => self.handle_waitid(guest, s).await,
1204
1205                Syscall::Setsid(s) => self.handle_setsid(guest, s).await,
1206                Syscall::Gettimeofday(s) => {
1207                    if virtualize_time {
1208                        self.handle_gettimeofday(guest, s).await
1209                    } else {
1210                        self.handle_unclassified_syscall(
1211                            guest,
1212                            call,
1213                            dettid,
1214                            config.panic_on_unsupported_syscalls,
1215                        )
1216                        .await
1217                    }
1218                }
1219                Syscall::Time(s) => {
1220                    if virtualize_time {
1221                        self.handle_time(guest, s).await
1222                    } else {
1223                        self.handle_unclassified_syscall(
1224                            guest,
1225                            call,
1226                            dettid,
1227                            config.panic_on_unsupported_syscalls,
1228                        )
1229                        .await
1230                    }
1231                }
1232                Syscall::ClockGettime(s) => {
1233                    if virtualize_time {
1234                        self.handle_clock_gettime(guest, s).await
1235                    } else {
1236                        self.handle_unclassified_syscall(
1237                            guest,
1238                            call,
1239                            dettid,
1240                            config.panic_on_unsupported_syscalls,
1241                        )
1242                        .await
1243                    }

1490    async fn on_exit_thread<G: GlobalRPC<Self::GlobalState>>(
1491        &self,
1492        tid: Tid,
1493        global_state: &G,
1494        mut thread_state: Self::ThreadState,
1495        exit_status: ExitStatus,
1496    ) -> Result<(), Error> {
1497        let dettid = thread_state.dettid;
1498        info!(
1499            "[detcore, dtid {}] thread exit hook, deregistering from scheduler.",
1500            dettid
1501        );
1502        // Close the final in-progress timeslice so this thread contributes its
1503        // last (partial) slice to the run report, even if it never exhausted a
1504        // full slice.
1505        let now = thread_state.thread_logical_time.as_nanos();
1506        thread_state.stats.close_final_timeslice(now);
1507        let detpid = thread_state.detpid.expect("Missing DetPid");
1508        let mm_id = thread_state.mm_id;
1509        deregister_thread(
1510            dettid,
1511            thread_state.thread_logical_time.clone(),
1512            &self.cfg,
1513            global_state,
1514            detpid,
1515            mm_id,
1516            thread_state.stats.timeslice_stats,
1517        )
1518        .await;
1519
1520        self.record_or_replay
1521            .on_exit_thread(
1522                tid,
1523                global_state,
1524                thread_state.record_or_replay,
1525                exit_status,
1526            )
1527            .await?;
1528
1529        Ok(())
1530    }
