31/// A record of all the preemptions and other scheduling events that occur during execution.
32#[derive(PartialEq, Default, Debug, Eq, Clone, Hash, Serialize, Deserialize)]
33pub struct PreemptionRecord {
34    // TODO(T106838933): switch the keys from DetTid to Pedigree.
35    /// A sorted list of end-of-timeslice preemption times for each thread.
36    per_thread: BTreeMap<DetTid, ThreadHistory>,
37    global: Vec<SchedEvent>,
38}
39
40impl std::fmt::Display for PreemptionRecord {
41    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
42        let str = serde_json::to_string(&self).unwrap();
43        write!(f, "{}", str)
44    }
45}

47impl PreemptionRecord {
48    /// TODO: This is not ideal because PreemptionRecord shouldn't be doing double duty as an eventlog.
49    pub fn from_sched_events(events: Vec<SchedEvent>) -> Self {
50        Self {
51            per_thread: Default::default(),
52            global: events,
53        }
54    }
55
56    /// Make a copy of everything in the `PreemptionRecord` in a public format.
57    pub fn extract_all(&self) -> BTreeMap<DetTid, ThreadHistory> {
58        self.per_thread.clone()
59    }
60
61    /// Inverse of from_vecs.
62    pub fn as_vecs(&self) -> BTreeMap<DetTid, Vec<(LogicalTime, Priority)>> {
63        let mut bt = BTreeMap::new();
64        for (tid, th) in &self.per_thread {
65            bt.insert(*tid, th.as_vec());
66        }
67        bt
68    }
69    /// TODO: provide a better way to create an empty PreemptionRecord but with entries for a given set of tids.
70    pub fn strip_contents(mut self) -> Self {
71        for history in self.per_thread.values_mut() {
72            history.prio_changes = Vec::new();
73            history.final_prio = 1000;
74        }
75        self.global = Vec::new();
76        self
77    }
78
79    /// Iterates over each sched event and applies function event -> *event yielding resulting iterator as new events
80    /// This effectively moves content of global allowing to split or delete events based on the function provided
81    pub fn split_map<F, R>(&mut self, splitter: F)
82    where
83        R: IntoIterator<Item = SchedEvent>,
84        F: Fn(SchedEvent, &ReplayCursor<SchedEvent>) -> R,
85    {
86        let mut result = Vec::new();
87        let global = std::mem::take(&mut self.global);
88        let mut cursor = ReplayCursor::from_iter(global);
89
90        while let Some(event) = cursor.next() {
91            for new_event in splitter(event, &cursor) {
92                result.push(new_event);
93            }
94        }
95        self.global = result;
96    }
97
98    /// Iterate over the schedevents.
99    pub fn schedevents_iter_mut(&mut self) -> std::slice::IterMut<'_, SchedEvent> {
100        self.global.iter_mut()
101    }
102
103    /// Leave only the per-thread preemption records, not the global schedule.
104    pub fn preemptions_only(&mut self) {
105        self.global.clear();
106    }
107
108    /// Make a copy just of the preemptions (cheap) rather than the whole schedule trace (expensive)
109    pub fn clone_preemptions_only(&self) -> Self {
110        PreemptionRecord {
111            per_thread: self.per_thread.clone(),
112            global: Vec::new(),
113        }
114    }
115
116    /// Read-only access to the raw event schedule.
117    pub fn schedevents(&self) -> &Vec<SchedEvent> {
118        &self.global
119    }
120
121    /// Returns true if the schedule trace (SchedEvent) record is nonempty.
122    pub fn contains_schedevents(&self) -> bool {
123        !self.global.is_empty()
124    }

126    /// Convert from a flat vector representation (Time,Priority_AFTER_Time) into the internal representation.
127    /// The very first timeslice should have a zero timestamp, but it is ignored if nonzero.
128    pub fn from_vecs(bt: &BTreeMap<DetTid, Vec<(LogicalTime, Priority)>>) -> Self {
129        let mut bt2 = BTreeMap::new();
130        for (tid, vec) in bt {
131            let th = if vec.is_empty() {
132                ThreadHistory {
133                    final_prio: DEFAULT_PRIORITY,
134                    prio_changes: Vec::new(),
135                }
136            } else {
137                // We could insist on this invariant, but it's a little more flexible not to:
138                // assert!(vec.get(0).unwrap().per_thread.is_zero());
139                let (_final_end, final_prio) = vec.last().unwrap();
140                let final_prio = *final_prio;
141
142                let mut prio_changes = Vec::new();
143                let mut it = vec.iter().peekable();
144                while let Some((_this_ns, this_p)) = it.next() {
145                    if let Some((next_ns, _next_p)) = it.peek() {
146                        prio_changes.push((*next_ns, *this_p));
147                    } else {
148                        // Don't need the last one, because we've already used it's
149                        // timestamp, and retrieved final_prio.
150                        break;
151                    }
152                }
153                assert!(final_prio >= FIRST_PRIORITY);
154                assert!(final_prio <= LAST_PRIORITY);
155                ThreadHistory {
156                    final_prio,
157                    prio_changes,
158                }
159            };
160            bt2.insert(*tid, th);
161        }
162        PreemptionRecord {
163            per_thread: bt2,
164            global: Vec::new(),
165        }
166    }

175        let mut str: String = self.to_string();
176        str.push('\n');

178            Ok(mut file) => match std::io::Write::write_all(&mut file, str.as_bytes()) {
179                Ok(_) => Ok(()),
180                Err(err) => Err(format!(
181                    "Failed to write preemption record to file {:?}, error: {}",
182                    path, err
183                )),
184            },
185            Err(err) => Err(format!(
186                "Failed to create file for preemption record {:?}, error: {}",
187                path, err
188            )),
189        }

192    /// Perform internal invariant checks on the PreemptionRecord and return an error if
193    /// it is not well formed.
194    pub fn validate(&self) -> Result<(), String> {
195        for (tid, history) in &self.per_thread {
196            if !is_ordinary_priority(history.final_prio) {
197                return Err(format!(
198                    "final priority for thread {} invalid: {}",
199                    tid, history.final_prio
200                ));
201            }
202            {
203                let mut time_last = None;
204                for (count, (ns, prio)) in history.prio_changes.iter().enumerate() {
205                    if let Some(last) = time_last {
206                        if !is_ordinary_priority(*prio) {
207                            return Err(format!(
208                                "preemption priority #{} for thread {} invalid: {}",
209                                count, tid, history.final_prio
210                            ));
211                        }
212                        if *ns <= last {
213                            return Err(format!(
214                                "Timestamps failed to monotonically increase ({}), in series:\n {:?}",
215                                ns, history.prio_changes
216                            ));
217                        }
218                    }
219                    time_last = Some(*ns);
220                }
221            }
222        }
223        Ok(())
224    }

226    /// Normalize priorities for readibility.
227    pub fn normalize(&self) -> PreemptionRecord {
228        let mut clone = self.clone();
229
230        let mut priomap: BTreeMap<Priority, Priority> = BTreeMap::new();
231        for history in clone.per_thread.values_mut() {
232            let _ = priomap.insert(history.final_prio, 0);
233            for (_ns, prio) in &history.prio_changes {
234                let _ = priomap.insert(*prio, 0);
235            }
236        }
237
238        for (cur_prio, val) in (DEFAULT_PRIORITY..).zip(priomap.values_mut()) {
239            assert!(cur_prio <= LAST_PRIORITY);
240            *val = cur_prio;
241        }
242        for history in clone.per_thread.values_mut() {
243            history.final_prio = *priomap.get(&history.final_prio).unwrap();
244            for (_ns, prio) in &mut history.prio_changes {
245                *prio = *priomap.get(prio).unwrap();
246            }
247        }
248
249        // One more pass to remove anything that just has default priority its whole lifetime.
250        let mut finalmap = BTreeMap::new();
251        for (tid, history) in clone.per_thread.into_iter() {
252            if history.final_prio != DEFAULT_PRIORITY || !history.prio_changes.is_empty() {
253                assert!(finalmap.insert(tid, history).is_none());
254            }
255        }
256        clone.per_thread = finalmap;
257        clone
258    }
259
260    /// Remove whichever is the latest preemption across all thread histories. If
261    /// there are no priority changes in any thread history, the entire last thread
262    /// (highest thread number) is completely removed.
263    /// Note this function assumes the last priority change in the list has the latest
264    /// preempt time.
265    pub fn with_latest_preempt_removed(&self) -> PreemptionRecord {
266        let mut preempts_latest_prio_changes: Vec<(DetTid, LogicalTime)> = self
267            .per_thread
268            .clone()
269            .into_iter()
270            .map(|(tid, th)| {
271                (
272                    tid,
273                    th.prio_changes
274                        .last() // Assume last prio change in the list has the latest preempt time
275                        .map_or(LogicalTime::ZERO, |prio_change| prio_change.0),
276                )
277            })
278            .collect();
279        preempts_latest_prio_changes.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());
280
281        let tid_with_preempt_to_drop = preempts_latest_prio_changes
282            .into_iter()
283            .map(|(tid, _th)| tid)
284            .next();
285
286        let mut clone = self.clone();
287        if let Some(tid) = tid_with_preempt_to_drop {
288            let thread_history = &mut clone.per_thread.get_mut(&tid).unwrap();
289            if !thread_history.prio_changes.is_empty() {
290                // There were prio changes for at least one of the threads, remove that
291                // preemption as the one to drop
292                let removed_prio = thread_history.prio_changes.pop().unwrap().1;
293                thread_history.final_prio = removed_prio;
294            } else {
295                // There were no prio changes for any threads, so just remove the last
296                // thread completely as the drop
297                clone.per_thread.pop_last();
298            }
299        }
300        clone
301    }
302}

304/// The record of priorities and preemptions for a particular thread.
305#[derive(
306    PartialEq, // Silly protection from rustfmt disagreements.
307    Debug,
308    Eq,
309    Clone,
310    Hash,
311    Serialize,
312    Deserialize,
313)]
314pub struct ThreadHistory {
315    /// The priority of the last timeslice, after the last preemption.
316    /// (This is also the initial priority if there are no preemptions.)
317    pub final_prio: Priority,
318
319    /// The preemption points and priority just BEFORE the preemption occurred.  It's
320    /// private, and thus you must go through the iterator, as this will change in the
321    /// future.
322    prio_changes: Vec<(LogicalTime, Priority)>,
323}
324
325impl ThreadHistory {
326    /// An empty thread history, with default priority only.
327    pub fn new() -> Self {
328        ThreadHistory {
329            final_prio: DEFAULT_PRIORITY,
330            prio_changes: Vec::new(),
331        }
332    }
333
334    /// Turns the thread history into an iterator that produces a stream of preemption events.
335    #[allow(clippy::should_implement_trait)]
336    pub fn into_iter(self) -> ThreadHistoryIterator {
337        ThreadHistoryIterator {
338            full_history: self,
339            ix: 0,
340        }
341    }
342
343    /// Flatten the time series into a uniform representation.
344    /// This is a `(Time, Priority_AFTER_Time)` representation.
345    /// The very first timeslice will have a zero timestamp.
346    // Note that this is different than the internal representation which groups together
347    // `Priority_UNTIL_Time, Time`, i.e. shifted by one.
348    pub fn as_vec(&self) -> Vec<(LogicalTime, Priority)> {
349        let mut vec = Vec::new();
350        let mut time0 = LogicalTime::from_nanos(0);
351
352        for (time1, prio) in &self.prio_changes {
353            vec.push((time0, *prio));
354            time0 = *time1;
355        }
356        vec.push((time0, self.final_prio));
357        vec
358    }
359
360    /// Return the priority of the initial timeslice after the thread starts running.
361    pub fn initial_priority(&self) -> Priority {
362        if let Some((_ns, pr)) = self.prio_changes.first() {
363            *pr
364        } else {
365            self.final_prio
366        }
367    }
368}
369
370impl Default for ThreadHistory {
371    fn default() -> Self {
372        Self::new()
373    }
374}

376/// An iterator over the preemption history of a thread.  Does not include the final
377/// timeslice, which does not have an ending time.
378// TODO: in the future this will perform "lazy IO" and pull from disk in batches.
379#[derive(PartialEq, Eq, Clone, Serialize, Deserialize)]
380pub struct ThreadHistoryIterator {
381    /// For now we store the full history in memory:
382    full_history: ThreadHistory,
383    /// Index that tracks our position.
384    ix: usize,
385}
386
387impl ThreadHistoryIterator {
388    /// Return the priority of the initial slice after the thread starts running.
389    pub fn initial_priority(&self) -> Priority {
390        self.full_history.initial_priority()
391    }
392
393    /// Return the priority of the last slice after the last preemption.
394    pub fn final_priority(&self) -> Priority {
395        self.full_history.final_prio
396    }
397}
398
399impl Iterator for ThreadHistoryIterator {
400    type Item = (LogicalTime, Priority);
401
402    fn next(&mut self) -> Option<Self::Item> {
403        let vec = &self.full_history.prio_changes;
404        if vec.len() > self.ix {
405            let elt = vec[self.ix];
406            self.ix += 1;
407            Some(elt)
408        } else {
409            None
410        }
411    }
412}

655/// A writer for a stream of preemption events.
656#[derive(Debug)]
657pub struct PreemptionWriter {
658    inner: PreemptionRecord,
659    dest: Option<PathBuf>,
660    flushed: bool,
661}
662
663impl PreemptionWriter {
664    /// A new, empty record of preemptions, with an optional location on disk that it will be
665    /// written to.  If no path is supplied, the results will accumulate in memory only.
666    pub fn new(path: Option<PathBuf>) -> Self {
667        PreemptionWriter {
668            inner: Default::default(),
669            dest: path,
670            flushed: false,
671        }
672    }
673
674    /// Does the record have zero entries?
675    pub fn is_empty(&self) -> bool {
676        self.inner.per_thread.is_empty()
677    }
678
679    /// Total number of elements in the record, across all threads.
680    pub fn len(&self) -> usize {
681        let mut count = 0;
682        for v in self.inner.per_thread.values() {
683            count += v.prio_changes.len();
684        }
685        count
686    }
687
688    /// Register a thread with its default/final priority.
689    pub fn register_thread(&mut self, tid: DetTid, prio: Priority) {
690        if self
691            .inner
692            .per_thread
693            .insert(
694                tid,
695                ThreadHistory {
696                    final_prio: prio,
697                    prio_changes: Vec::new(),
698                },
699            )
700            .is_some()
701        {
702            panic!(
703                "PreemptionRecord: error, cannot re-register thread id already registered: {}",
704                tid
705            )
706        }
707    }
708
709    /// Insert a new preemption point for the given thread.
710    /// It must monotonically increase in time.
711    ///
712    /// For now it (redundantly) takes both the priority of the timeslice just finished,
713    /// and the next one coming up.  This serves as a sanity check.
714    pub fn insert_reprioritization(
715        &mut self,
716        tid: DetTid,
717        time: LogicalTime,
718        prior_prio: Priority,
719        next_prio: Priority,
720    ) {
721        let history = self.inner.per_thread.get_mut(&tid).unwrap_or_else(|| {
722            panic!(
723                "PreemptionRecord: Cannot insert a preemption before registering thread {}",
724                tid
725            )
726        });
727        assert_eq!(history.final_prio, prior_prio);
728
729        if let Some((last, _prio)) = history.prio_changes.last() {
730            assert!(&time > last);
731        }
732        history.prio_changes.push((time, prior_prio));
733        history.final_prio = next_prio;
734    }
735
736    /// Add a SchedEvent to the global log of thread behavior.
737    pub fn insert_schedevent(&mut self, ev: SchedEvent) {
738        if ev.count > 0 {
739            self.inner.global.push(ev)
740            // TODO, aggregation: possibly check if the last event was the SAME, and combine them,
741            // increasing the count.
742        } else {
743            trace!("NOT recording scheduled event with zero count!");
744        }
745    }
746
747    /// Set the priority of the current thread after a change.
748    pub fn set_current(&mut self, tid: DetTid, new_prio: Priority) {
749        let history = self.inner.per_thread.get_mut(&tid).unwrap_or_else(|| {
750            panic!(
751                "PreemptionRecord: Cannot set current priority before registering thread {}",
752                tid
753            )
754        });
755        history.final_prio = new_prio;
756    }
757
758    /// Abort writing to disk and instead gather the output thusfar into a string.
759    pub fn into_string(mut self) -> String {
760        self.flushed = true;
761        self.inner.to_string()
762    }
763
764    /// Finish any IO necessary to flush the record to disk.
765    /// This will also happen automatically on drop.
766    pub fn flush(mut self) -> Result<(), String> {
767        self.flushed = true;
768        self.write_to_disk()
769    }
770
771    fn write_to_disk(&mut self) -> Result<(), String> {
772        if let Some(path) = &self.dest {
773            self.inner.write_to_disk(path)
774        } else {
775            Err(
776                "Cannot write_to_disk because this PreemptionWriter was created without a backing file.".to_string()
777            )
778        }
779    }
780}
781
782impl Drop for PreemptionWriter {
783    fn drop(&mut self) {
784        if !self.flushed
785            && self.dest.is_some()
786            && let Err(e) = self.write_to_disk()
787        {
788            panic!("Error while dropping PreemptionWriter: {}", e);
789        }
790    }
791}

793/// A reader for a stream of preemption events.
794#[derive(Debug)]
795pub struct PreemptionReader {
796    inner: PreemptionRecord,
797}
798
799// Deprecated: use schedevents()
800/// Read a full trace from disk.  Panic if it doesn't load.
801pub fn read_trace(path: &Path) -> Vec<SchedEvent> {
802    let pr = read_preemption_record(path);
803    pr.global
804}
805
806// TODO: we should implement streaming and not read this all at once.
807fn read_preemption_record(path: &Path) -> PreemptionRecord {
808    let string = std::fs::read_to_string(path)
809        .unwrap_or_else(|e| panic!("Error reading file {:?}:\n {}", path, e));
810    let pr: PreemptionRecord = serde_json::from_str(&string).unwrap_or_else(|e| {
811        panic!(
812            "Error parsing PreemptionRecord from JSON: {}\nJSON contents:\n{}",
813            e, string
814        )
815    });
816    if let Err(e) = pr.validate() {
817        panic!(
818            "Invalid PreemptionRecord when loading from path {}. Error:\n {}",
819            path.display(),
820            e
821        );
822    }
823    pr
824}
825
826// TODO: eventually this should do streaming IO, and abstract it behind this interface.
827impl PreemptionReader {
828    /// Access the stored `PreemptionRecord`, and eagerly or lazily load its data from disk.
829    pub fn new(path: &Path) -> Self {
830        let pr = read_preemption_record(path);
831        PreemptionReader { inner: pr }
832    }
833
834    /// Gets the inner `PreemptionRecord`.
835    pub fn into_inner(self) -> PreemptionRecord {
836        self.inner
837    }
838
839    /// Extract a copy of all the entries for a particular thread.  Note that this returns None if
840    /// there is NO record for the thread (not registered).  Otherwise, it returns a copy of the
841    /// `ThreadHistory`, which may be read into memory or not.
842    pub fn extract_thread_record(&self, tid: &DetTid) -> Option<ThreadHistory> {
843        self.inner.per_thread.get(tid).cloned()
844    }
845
846    /// Return the initial priority for a thread, if it is recorded in the record.
847    pub fn thread_initial_priority(&self, tid: &DetTid) -> Option<Priority> {
848        self.inner.per_thread.get(tid).map(|x| x.initial_priority())
849    }
850
851    /// Return all the threads in this record.
852    pub fn all_threads(&self) -> Vec<DetTid> {
853        // TODO(T110956298): have this return an iterator.
854        self.inner.per_thread.keys().copied().collect()
855    }
856
857    /// Load the full record of preemptions into memory.
858    pub fn load_all(&self) -> PreemptionRecord {
859        self.inner.clone()
860    }
861
862    /// Size in number of preemptions, across all threads.
863    pub fn size(&self) -> usize {
864        let mut sum = 0;
865        for th in self.inner.per_thread.values() {
866            sum += th.prio_changes.len()
867        }
868        sum
869    }
870}

872/// Take a file containing a PreemptionRecord and strip the times from all the events.
873/// Optionally take a destination path, otherwise create a destination file in the same directory
874/// with our own naming convention.
875pub fn strip_times_from_events_file(
876    sched_path: &Path,
877    dest: Option<PathBuf>,
878) -> anyhow::Result<PathBuf> {
879    let new_path = dest.unwrap_or_else(|| sched_path.with_extension("notimes"));
880    let mut preemptions = PreemptionReader::new(sched_path).into_inner();
881    for se in preemptions.schedevents_iter_mut() {
882        se.end_time = None;
883    }
884    preemptions
885        .write_to_disk(new_path.as_ref())
886        .map_err(anyhow::Error::msg)?;
887    Ok(new_path)
888}
