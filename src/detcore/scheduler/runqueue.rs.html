9//! The main queue for runnable tasks.
10//!
11//! Tasks are selected from the queue based on 2 factors:
12//! 1. Their priority
13//! 2. Their round-robin order.
14//!
15//! The factors are compared in that order. Round-robin orders monotonically increase across
16//! the entire queue; a task is assigned an order at insertion time.
17//!
18//! Round-robin orders can also be negative when the `push_front` method is
19//! used. This "skips the line" within the priority level. This is mostly
20//! relevant in non-chaos modes, where all threads have the same priority. In
21//! this case, time-based events should skip the line and end-up in the front of
22//! the queue.
23//!
24//! In addition, some priority values are reserved, such as a high priority
25//! for IO eager polling.
26//!
27//! # Polling strategy
28//!
29//! We employ a polling strategy for guest threads that *would* blocked, but where we
30//! don't model precisely what conditions they're waiting for.  Anywhere we have a precise
31//! model of inter-thread dependencies (e.g. futexes), we can sleep a thread until we
32//! encounter the matching event that will wake it.  But for Linux features that we don't
33//! model 100% precisely, polling is a way to remain agnostic as to the exact
34//! dependencies, but still support these blocking behaviors deterministically.
35//!
36//! The older DetTrace system used polling, but it would poll every time through the
37//! round robin queue, which can create extremely bad performance with many threads
38//! polling an unbounded number of times.  We can greatly improve the performance by
39//! polling only at less-frequent, but still deterministically-defined intervals, such
40//! as when we think we're out of "productive" work to do.
41//!
42//! Thererefore we have special handling for scheduling polling tasks. When
43//! initially queued, there is exponential backoff in priority with the number
44//! of attempts. After enough queueing operations are performed, however,
45//! polling tasks are upgraded to their original priority to prevent complete
46//! starvation. The frequency of upgrades is controlled by `POLLING_UPGRADE_INTERVAL`.
47

62/// The user-accessible priority of a thread. Lowest runs first.
63pub type Priority = u64;
64
65const EAGER_IO_REPOLL_PRIORITY: Priority = Priority::MIN;
66
67/// The lowest/highest priority a thread can have.
68pub const FIRST_PRIORITY: Priority = EAGER_IO_REPOLL_PRIORITY + 1;
69
70/// The last/lowest (numerically largest) priority a thread can have.
71pub const LAST_PRIORITY: Priority = 10000;
72
73/// A high priority for threads the replayer DOES want to run.
74pub const REPLAY_FOREGROUND_PRIORITY: Priority = FIRST_PRIORITY;
75
76/// A low priority given to threads the replayer does NOT want to run.
77pub const REPLAY_DEFERRED_PRIORITY: Priority = LAST_PRIORITY - 1;
78
79/// The default priority for a thread. If chaos mode is not enabled, all threads have this
80/// priority.
81pub const DEFAULT_PRIORITY: Priority = 1000;
82
83/// Whether the priority is user-accessible. We use some values for special
84/// purposes; these shouldn't be set by the user.
85pub fn is_ordinary_priority(prio: Priority) -> bool {
86    (FIRST_PRIORITY..=LAST_PRIORITY).contains(&prio)
87}
88
89/// Deterministically transform 64 bits of entropy into a random user-settable
90/// priority.
91pub fn entropy_to_priority(entropy: u64) -> Priority {
92    let range = LAST_PRIORITY - FIRST_PRIORITY + 1;
93    let offset = entropy % range;
94    FIRST_PRIORITY + offset
95}

97/// The round robin turn of threads within a given priority level. Lowest runs
98/// first. Both negative and positive values are used to allow insertion of a
99/// thread at both the "front" and "back" of a priority level.
100type RoundRobinTurn = i64;
101
102/// The key into the priority queue that uniquely determines what to run next.
103/// Priorities that compare lower run first.
104#[derive(Debug, Copy, Clone)]
105pub struct PrioritizedOrder {
106    priority: Priority,
107    turn: RoundRobinTurn,
108}
109
110// These match the derived definitions, but clearly display our intention:
111impl Ord for PrioritizedOrder {
112    fn cmp(&self, other: &Self) -> Ordering {
113        self.priority
114            .cmp(&other.priority)
115            .then(self.turn.cmp(&other.turn))
116    }
117}
118impl PartialOrd for PrioritizedOrder {
119    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
120        Some(self.cmp(other))
121    }
122}
123impl PartialEq for PrioritizedOrder {
124    fn eq(&self, other: &Self) -> bool {
125        self.cmp(other).is_eq()
126    }
127}
128impl Eq for PrioritizedOrder {}

130impl fmt::Display for PrioritizedOrder {
131    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
132        write!(f, "(p: {}, t: {})", self.priority, self.turn)
133    }

136/// After queueing this many new tasks, perform a "poll upgrade," in which
137/// we upgrade all outstanding polling tasks their original priority levels,
138/// temporarily negating backoff behavior.
139const POLLING_UPGRADE_INTERVAL: u64 = 200;
140
141#[derive(Debug, Copy, Clone)]
142struct QueueValue {
143    tid: DetTid,
144    /// Upgrade to this priority during polling upgrades
145    poll_upgrade: Option<Priority>,
146}
147
148#[derive(Debug, Clone)]
149pub struct RunQueue {
150    /// We use a "flattened" queue (rather than a Priority -> Vec<DetTid> map)
151    /// to simplify peek/pop logic: there's no need to ignore clear empty
152    /// from unused priority levels vectors. This could also reduce allocator
153    /// pressure. Also, each thread having a clear global key makes it easier to
154    /// change their priorities after they are in the queue.
155    ///
156    /// Additionally, we use a TreeMap rather than a Heap to ease removing /
157    /// inserting random values for poll upgrades. std::BinaryHeap would require
158    /// destroying/re-allocating the entire structure to do this.
159    queue: BTreeMap<PrioritizedOrder, QueueValue>,
160
161    // We use global turn counters across all priority levels. This foregoes
162    // the need for an extra data structure to track them, while also ensuring
163    // unique keys for every insertion. Because of this, we never need to alter
164    // a turn value when altering the priority level of a thread.
165    last_back_turn: RoundRobinTurn,
166    last_front_turn: RoundRobinTurn,
167
168    /// Used to lock the queue from other changes while we are tentatively popping from it, and also
169    /// cache the result.
170    tentative_selection: Option<DetTid>,
171
172    /// A thread that explicitly yielded must not be selected again until some
173    /// other runnable thread receives a turn.
174    yielded_skip: Option<DetTid>,
175
176    // TODO: The following fields need to be properly abstracted into separate types of run queues.
177    /// Which scheduling strategy shall we use.
178    sched_strategy: SchedHeuristic,
179    prng: Pcg64Mcg,
180
181    sticky_random_param: f64,
182    sticky_random_selection: Option<DetTid>,
183}

185/// A multi-line print of the runqueue.
186impl fmt::Display for RunQueue {
187    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
188        writeln!(
189            f,
190            "Run queue, size={}, last_back_turn={}, last_front_turn={}:",
191            self.queue.len(),
192            self.last_back_turn,
193            self.last_front_turn,
194        )?;
195        for x in self.queue.iter() {
196            writeln!(f, "    {:.500?}", x)?;
197        }
198        Ok(())
199    }
200}

203    /// Create a new RunQueue.
204    pub fn new(ss: SchedHeuristic, seed: u64, srp: f64) -> Self {
205        detlog!("SCHEDRAND: seeding scheduler runqueue with seed {}", seed);
206        Self {
207            queue: BTreeMap::new(),
208            // For clarity, 0 is unused so that positive/negative == back/front:
209            last_back_turn: 0,
210            last_front_turn: 0,
211            sched_strategy: ss,
212            tentative_selection: None,
213            yielded_skip: None,
214            prng: Pcg64Mcg::seed_from_u64(seed),
215            sticky_random_param: srp,
216            sticky_random_selection: None,
217        }
218    }

220    fn push_safety_check(&self, tid: DetTid) {
221        if cfg!(debug_assertions) {
222            // Expensive.
223            for qv in self.queue.values() {
224                if qv.tid == tid {
225                    panic!(
226                        "Invariant violation! Tried to add {} to runqueue, but it's already present:\n {:?}",
227                        tid, self
228                    );
229                }
230            }
231        }
232    }

234    // Return the numerically least Priority value in the run_queue, or None if the queue is empty.
235    pub fn first_priority(&self) -> Option<Priority> {
236        let (k, _) = self.queue.first_key_value()?;
237        Some(k.priority)
238    }
239
240    /// Push a thread to the back of the specified priority. Return the
241    /// resulting overall position in the queue.
242    ///
243    /// Mutating operation: this will error if a tentative_pop/commit transaction is underway.
244    pub fn push_back(&mut self, tid: DetTid, priority: Priority) -> PrioritizedOrder {
245        assert!(self.tentative_selection.is_none());
246        self.push_safety_check(tid);
247        if !is_ordinary_priority(priority) {
248            panic!("This is not an acceptable priority value: {}", priority);
249        }
250        self.push_back_inner(tid, priority, None)
251    }
252
253    /// Requeue an explicitly yielding thread at its persistent priority while
254    /// excluding it from the next selection. The exclusion, rather than the
255    /// queue key, makes this a one-turn operation under every heuristic.
256    pub fn push_yielded(&mut self, tid: DetTid, priority: Priority) -> PrioritizedOrder {
257        assert!(self.yielded_skip.is_none());
258        self.yielded_skip = Some(tid);
259        self.push_back(tid, priority)
260    }
261
262    /// Push a polling thread. The priority level is an exponential backoff from
263    /// the given `normal_priority` value. The pushed thread will also
264    /// participatein "poll upgrades" in which periodically polling threads are
265    /// re-boosted to their original `normal_priority` values.
266    ///
267    /// Mutating operation: this will error if a tentative_pop/commit transaction is underway.
268    pub fn push_poller(
269        &mut self,
270        tid: DetTid,
271        normal_priority: Priority,
272        poll_attempt: u32,
273    ) -> PrioritizedOrder {
274        assert!(self.tentative_selection.is_none());
275        self.push_safety_check(tid);
276        // Exponential backoff in priority, up to LAST_PRIORITY:
277        let priority = 1u64
278            .checked_shl(poll_attempt)
279            .and_then(|f| f.checked_mul(normal_priority))
280            .unwrap_or(Priority::MAX)
281            .min(LAST_PRIORITY);
282        // Upgrade back to original priority:
283        self.push_back_inner(tid, priority, Some(normal_priority))
284    }
285
286    fn push_back_inner(
287        &mut self,
288        tid: DetTid,
289        priority: Priority,
290        poll_upgrade: Option<Priority>,
291    ) -> PrioritizedOrder {
292        self.last_back_turn += 1;
293        let turn = self.last_back_turn;
294        let prio = PrioritizedOrder { priority, turn };
295        self.push_inner(tid, prio, poll_upgrade)
296    }
297
298    /// Push a thread to the front of the specified priority. `push_back` should
299    /// be used unless special circumstances call for `push_front`. Return the
300    /// resulting overall position in the queue.
301    ///
302    /// Mutating operation: this will error if a tentative_pop/commit transaction is underway.
303    pub fn push_front(&mut self, tid: DetTid, priority: Priority) -> PrioritizedOrder {
304        assert!(self.tentative_selection.is_none());
305        self.push_safety_check(tid);
306        assert!(is_ordinary_priority(priority));
307        self.push_front_inner(tid, priority, None)
308    }
309
310    /// Workaround for eager io repolling: this will send the thread to the
311    /// absolute front of the queue. Return the resulting overall position in
312    /// the queue.
313    ///
314    /// Mutating operation: this will error if a tentative_pop/commit transaction is underway.
315    pub fn push_eager_io_repoll(&mut self, tid: DetTid) -> PrioritizedOrder {
316        assert!(self.tentative_selection.is_none());
317        self.push_safety_check(tid);
318        let priority = EAGER_IO_REPOLL_PRIORITY;
319        self.push_front_inner(tid, priority, None)
320    }
321
322    fn push_front_inner(
323        &mut self,
324        tid: DetTid,
325        priority: Priority,
326        poll_upgrade: Option<Priority>,
327    ) -> PrioritizedOrder {
328        self.last_front_turn -= 1;
329        let turn = self.last_front_turn;
330        let prio = PrioritizedOrder { priority, turn };
331        self.push_inner(tid, prio, poll_upgrade)
332    }
333
334    fn push_inner(
335        &mut self,
336        tid: DetTid,
337        prio: PrioritizedOrder,
338        poll_upgrade: Option<Priority>,
339    ) -> PrioritizedOrder {
340        let qval = QueueValue { tid, poll_upgrade };
341        let old = self.queue.insert(prio, qval);
342        assert!(old.is_none()); // last_*_turn should be monotonic
343        self.check_poll_upgrade();
344        prio
345    }

363    pub fn contains_tid(&self, tid: DetTid) -> bool {

367    /// Remove `tid` from the queue, returning true if removal ocurred.
368    /// Mutating operation: this will error if a tentative_pop/commit transaction is underway.
369    pub fn remove_tid(&mut self, tid: DetTid) -> bool {
370        assert!(self.tentative_selection.is_none());
371
372        // This is O(N), but could be faster if we also stored a thread -> priority mapping.
373        let mut kept_all = true;
374        self.queue.retain(|_k, v| {
375            let ret = v.tid != tid;
376            kept_all = kept_all && ret;
377            ret
378        });
379        if self.yielded_skip == Some(tid) {
380            self.yielded_skip = None;
381        }
382        !kept_all
383    }

385    // Helper function for logging purposes.
386    fn random_range<T>(&mut self, start: T, end: T) -> T
387    where
388        T: SampleUniform + Display + PartialOrd + Copy,
389    {
390        let r = self.prng.random_range(start..end);
391        detlog!("SCHEDRAND: [{},{}) => {}", start, end, r);
392        r
393    }

395    /// Begin, but do not complete, a pop_transaction.  This can be committed or undone later.  But
396    /// one of those must happen before other modification operations can occur on the RunQueue.
397    ///
398    /// Postcondition: if return a `Some` value, the RunQueue enters a *locked* state where
399    /// commit or undo must happen before any other mutations to the structure.
400    pub fn tentative_pop_next(&mut self) -> Option<DetTid> {
401        let skip = self
402            .yielded_skip
403            .filter(|tid| self.queue.len() > 1 && self.contains_tid(*tid));
404        self.tentative_selection = match self.sched_strategy {
405            SchedHeuristic::None | SchedHeuristic::ConnectBind => self
406                .queue
407                .values()
408                .find(|value| Some(value.tid) != skip)
409                .map(|value| value.tid),
410            SchedHeuristic::Random => {
411                if self.queue.is_empty() {
412                    return None;
413                }
414
415                // If there is not Tid picked from a previous operation, let's pick one now.
416                if self.tentative_selection.is_none() {
417                    let eligible = self.queue.len() - usize::from(skip.is_some());
418                    let random_idx = self.random_range(0, eligible);
419                    self.tentative_selection = self
420                        .queue
421                        .values()
422                        .filter(|value| Some(value.tid) != skip)
423                        .nth(random_idx)
424                        .map(|value| value.tid);
425                };
426
427                self.tentative_selection
428            }
429            SchedHeuristic::StickyRandom => {
430                if self.queue.is_empty() {
431                    return None;
432                }
433
434                if self.sticky_random_selection == skip {
435                    self.sticky_random_selection = None;
436                }
437                if self.sticky_random_selection.is_none()
438                    || !self.contains_tid(self.sticky_random_selection.unwrap())
439                {
440                    let eligible = self.queue.len() - usize::from(skip.is_some());
441                    let random_idx = self.random_range(0, eligible);
442                    self.sticky_random_selection = self
443                        .queue
444                        .values()
445                        .filter(|value| Some(value.tid) != skip)
446                        .nth(random_idx)
447                        .map(|value| value.tid);
448                }
449
450                self.sticky_random_selection
451            }
452        };
453
454        self.tentative_selection
455    }

457    /// Complete the tentative pop operation, readying the RunQueue for future operations.  This
458    /// operation is only permissible when the queue is locked, i.e. the tentative_pop has
459    /// previously returned `Some`.
460    pub fn commit_tentative_pop(&mut self) -> DetTid {
461        // Check that queue is locked and unlock it.
462        let tentative_selection = self
463            .tentative_selection
464            .take()
465            .expect("tentative_pop to already returned a `Some`");
466
467        let ret = match self.sched_strategy {
468            SchedHeuristic::None | SchedHeuristic::ConnectBind | SchedHeuristic::Random => {
469                let key = *self
470                    .queue
471                    .iter()
472                    .find(|(_k, v)| v.tid == tentative_selection)
473                    .map(|(k, _v)| k)
474                    .unwrap();
475                self.queue.remove(&key).map(|v| v.tid)
476            }
477            SchedHeuristic::StickyRandom => {
478                let tid = self.sticky_random_selection.unwrap();
479                // Probability of staying to our current thread on the next round.
480                // If the generated random number is smaller than what we set, we switch threads.
481                if self.random_range(0f64, 1f64) <= 1.0 - self.sticky_random_param {
482                    self.sticky_random_selection = None;
483                }
484
485                let key = *self
486                    .queue
487                    .iter()
488                    .find(|(_k, v)| v.tid == tid)
489                    .map(|(k, _v)| k)
490                    .unwrap();
491
492                self.queue.remove(&key).map(|v| v.tid)
493            }
494        }
495        .expect("to always return a DetTid");
496        // The above should always return a DetTid as we peeked right before.
497        // If this invariant is violated, then it's a bug, or the queue is modified
498        // between the peek and tentative_pop and commit_tentative_pop.
499        debug_assert!(ret == tentative_selection);
500        ret
501    }

503    /// Commit a tentative pop for a guest turn that will actually run. A
504    /// yielded thread's one-turn exclusion is consumed only here, not by
505    /// scheduler bookkeeping turns that never unblock a guest.
506    pub fn commit_tentative_pop_completed_turn(&mut self) -> DetTid {
507        let tid = self.commit_tentative_pop();
508        self.consume_yield_exclusion();
509        tid
510    }
511
512    /// Mark that a different guest received execution after an explicit yield.
513    pub fn consume_yield_exclusion(&mut self) {
514        self.yielded_skip = None;
515    }
516
517    /// Forget the tentative pop as though it never happened.
518    pub fn undo_tentative_pop(&mut self) {
519        assert!(self.tentative_selection.is_some());
520        self.tentative_selection = None;
521    }
522
523    /// Return how many things have been queued.
524    fn turn_counter(&self) -> u64 {
525        debug_assert!(self.last_back_turn >= 0);
526        debug_assert!(self.last_front_turn <= 0);
527        self.last_back_turn as u64 + self.last_front_turn.unsigned_abs()
528    }
529
530    fn check_poll_upgrade(&mut self) {
531        if self.turn_counter().is_multiple_of(POLLING_UPGRADE_INTERVAL) {
532            self.do_poll_upgrade()
533        }
534    }
535
536    /// Upgrade polled tasks to their specified normal priority.
537    #[cold]
538    fn do_poll_upgrade(&mut self) {
539        let upgrades = self
540            .queue
541            .iter()
542            .filter_map(|(k, v)| v.poll_upgrade.map(|upgd| (*k, upgd)))
543            .collect::<Vec<(PrioritizedOrder, Priority)>>();
544        // TODO(T100400409): if all polling threads are below a certain priority, this
545        // can use a range query rather than iterating over all threads in the
546        // run queue:
547        for (key, upgrade_prio) in upgrades {
548            let mut new_key = key;
549            new_key.priority = upgrade_prio;
550            let mut qval = self.queue.remove(&key).unwrap();
551            qval.poll_upgrade = None; // there's no need to upgrade to the same priority twice
552            let old = self.queue.insert(new_key, qval);
553            assert!(old.is_none()); // round robin turns should ensure uniqueness
554        }
555    }
556}

568    #[test]
569    fn yielded_thread_cedes_exactly_one_turn_under_every_heuristic() {
570        for strategy in [
571            SchedHeuristic::None,
572            SchedHeuristic::ConnectBind,
573            SchedHeuristic::Random,
574            SchedHeuristic::StickyRandom,
575        ] {
576            let yielding = DetTid::from_raw(1);
577            let peer = DetTid::from_raw(2);
578            let mut queue = RunQueue::new(strategy, 0, 1.0);
579
580            queue.push_back(yielding, DEFAULT_PRIORITY - 1);
581            assert_eq!(queue.tentative_pop_next(), Some(yielding));
582            assert_eq!(queue.commit_tentative_pop(), yielding);
583
584            queue.push_yielded(yielding, DEFAULT_PRIORITY - 1);
585            queue.push_back(peer, LAST_PRIORITY);
586            assert_eq!(queue.tentative_pop_next(), Some(peer), "{strategy:?}");
587            assert_eq!(
588                queue.commit_tentative_pop_completed_turn(),
589                peer,
590                "{strategy:?}"
591            );
592
593            assert_eq!(queue.yielded_skip, None, "{strategy:?}");
594            let restored_priority = queue
595                .queue
596                .iter()
597                .find(|(_key, value)| value.tid == yielding)
598                .map(|(key, _value)| key.priority);
599            assert_eq!(
600                restored_priority,
601                Some(DEFAULT_PRIORITY - 1),
602                "{strategy:?}"
603            );
604
605            if matches!(strategy, SchedHeuristic::None | SchedHeuristic::ConnectBind) {
606                queue.push_back(peer, LAST_PRIORITY);
607                assert_eq!(queue.tentative_pop_next(), Some(yielding), "{strategy:?}");
608            }
609        }
610    }

614        let yielding = DetTid::from_raw(1);
615        let peer = DetTid::from_raw(2);
616        let mut queue = RunQueue::default();
617
618        queue.push_back(yielding, DEFAULT_PRIORITY);

620        assert_eq!(queue.commit_tentative_pop(), yielding);

622        queue.push_yielded(yielding, DEFAULT_PRIORITY);
623        queue.push_back(peer, DEFAULT_PRIORITY);
