24/// Replayer
25#[derive(Debug)]
26pub struct Replayer {
27    /// A cursor that holds our place in the global total order of events being replayed.
28    pub cursor: ReplayCursor<SchedEvent>,
29    /// Keep track of how many events we have observed during replay.  The current value is the
30    /// event number of the NEXT event to replay.
31    pub traced_event_count: u64,
32
33    /// The number of events we've popped from the recording.
34    pub events_popped: u64,
35
36    /// Desync events observed per thread.  Counts different kinds of desyncs.
37    pub desync_counts: BTreeMap<DetTid, DesyncStats>,
38    /// A cached copy of the same (immutable) field in Config.
39    pub replay_exhausted_panic: bool,
40    /// A cached copy of the same (immutable) field in Config.
41    pub die_on_desync: bool,
42}
43
44/// The count of desyncs for a particular thread.
45#[derive(Debug, PartialEq, Eq, Default, Copy, Clone)]
46pub struct DesyncStats {
47    /// Not inclusive of hard desyncs.  You must add soft+hard together to get total desyncs.
48    pub soft: u64,
49    pub hard: u64,
50    /// The very worst kind, where we need to context switch, but we don't have a match on the
51    /// current event. These are a subset of the hard desyncs.
52    pub at_context_switch: u64,
53    /// How many times do we find a match again, after a mismatch, specifically new obserserved
54    /// events that were not expected.
55    pub resync_insertions: u64,
56    /// How many times do we find a match again, after a mismatch, specifically expected events that
57    /// were not observed.
58    pub resync_deletions: u64,
59    /// How many times did we fast-forward to resync.
60    pub fast_forwards: u64,
61
62    /// A bit of statefulness: true when the last event processed was a desync.
63    pub last_was_desync: bool,
64}
65
66impl Add for DesyncStats {
67    type Output = Self;
68
69    fn add(self, rhs: Self) -> Self::Output {
70        DesyncStats {
71            soft: self.soft + rhs.soft,
72            hard: self.hard + rhs.hard,
73            at_context_switch: self.at_context_switch + rhs.at_context_switch,
74            resync_insertions: self.resync_insertions + rhs.resync_insertions,
75            resync_deletions: self.resync_deletions + rhs.resync_deletions,
76            fast_forwards: self.fast_forwards + rhs.fast_forwards,
77
78            // Not commutative!
79            last_was_desync: rhs.last_was_desync,
80        }
81    }
82}

91pub enum ReplayAction {
92    Continue(Option<LogicalTime>),
93    Stop(StopReason),
94    /// The boolean indicates if the context switch should be immediate (or wait till the current
95    /// thread blocks itself and naturally yields control).
96    ContextSwitch(bool, DetTid, Option<LogicalTime>),
97}

99fn compare_desync(observed: &SchedEvent, expected: &SchedEvent) -> String {
100    if !is_desync(observed, expected) {
101        "SYNCED".to_string()
102    } else if observed.op != expected.op {
103        "FULL-OP-DESYNC".to_string()
104    } else {
105        let mut msg = "DESYNC".to_string();
106        if (expected.start_rip.is_some() && observed.start_rip != expected.start_rip)
107            || (expected.end_rip.is_some() && observed.end_rip != expected.end_rip)
108        {
109            msg = "RIP-".to_owned() + &msg;
110        }
111        if observed.end_time != expected.end_time {
112            msg = "TIME-".to_owned() + &msg;
113        }
114        if observed.op == Op::Branch
115            && expected.op == Op::Branch
116            && observed.count != expected.count
117        {
118            msg = "RCB-".to_owned() + &msg;
119        }
120        msg
121    }
122}

124/// Are the two events consistent with eachother, i.e. that they could come from the same execution.
125/// This returns true for an event which is a prefix of another (having a lower count).
126///
127/// This weaker notion always implies `events_match` but not the other way around.
128pub fn events_consistent(observed: &SchedEvent, expected: &SchedEvent) -> bool {
129    !is_hard_desync(observed, expected)
130}
131
132/// Do the events describe the same range of instructions, based on all the information present.
133/// end_time is ignored because it depends on the global context, not just these events.
134pub fn events_match(observed: &SchedEvent, expected: &SchedEvent) -> bool {
135    !is_desync(observed, expected)
136}

138fn is_desync(observed: &SchedEvent, expected: &SchedEvent) -> bool {
139    if (observed.dettid, observed.op, observed.count)
140        != (expected.dettid, expected.op, expected.count)
141    {
142        return true;
143    }
144    // It is ok for the observed event to contain information not present in the replay trace.
145    if expected.end_time.is_some() && expected.end_time != observed.end_time {
146        return true;
147    }
148
149    if expected.start_rip.is_some() && observed.start_rip != expected.start_rip {
150        return true;
151    }
152
153    if expected.end_rip.is_some() && observed.end_rip != expected.end_rip {
154        return true;
155    }
156
157    false
158}

160fn is_hard_desync(observed: &SchedEvent, expected: &SchedEvent) -> bool {
161    let mut strip1 = observed.clone();
162    let mut strip2 = expected.clone();
163    strip1.end_time = None;
164    strip2.end_time = None;
165    strip1.count = 0;
166    strip2.count = 0;
167    strip1 != strip2
168}

170impl Replayer {
171    /// Allocate a new replayer.
172    pub fn new(i: impl IntoIterator<Item = SchedEvent>) -> Replayer {
173        let cursor = ReplayCursor::from_iter(i);
174        Replayer {
175            cursor,
176            desync_counts: BTreeMap::new(),
177            die_on_desync: false,
178            replay_exhausted_panic: false,
179            traced_event_count: 0,
180            events_popped: 0,
181        }
182    }
183
184    /// This function look-aheads into following instruction to determine if
185    /// an explicit timer preemption is required while replaying. This is required when
186    /// the original run schedule was modified before replaying which should be a valid
187    /// usecase for hermit analyze. Returned LogicalTime is RELATIVE to the corresponding thread localtime
188    fn resolve_time_to_run(&self, next: &SchedEvent) -> Option<LogicalTime> {
189        let cursor = &self.cursor;
190
191        if let Some(next_time_slice_in_brances) =
192            match (next, cursor.peek_nth(1), cursor.peek_nth(2)) {
193                (
194                    SchedEvent {
195                        op: Op::Branch,
196                        count,
197                        ..
198                    },
199                    Some(SchedEvent {
200                        op: Op::OtherInstructions,
201                        ..
202                    }),
203                    Some(SchedEvent { op: Op::Branch, .. }),
204                )
205                | (
206                    // this is currently impossible because we always have OtherInstruction
207                    // following a branch but this doesn't have to be the case and it might
208                    // be better to avoid this behavior in the future
209                    SchedEvent {
210                        op: Op::Branch,
211                        count,
212                        ..
213                    },
214                    Some(SchedEvent { op: Op::Branch, .. }),
215                    _,
216                ) => {
217                    trace!("branch count to set the next_timeslice: {}", count);
218                    Some(count)
219                }
220                (
221                    SchedEvent {
222                        op: Op::Branch,
223                        count,
224                        dettid: tid,
225                        ..
226                    },
227                    Some(SchedEvent {
228                        op: Op::OtherInstructions,
229                        ..
230                    }),
231                    Some(SchedEvent {
232                        dettid: next_tid, ..
233                    }),
234                )
235                | (
236                    SchedEvent {
237                        op: Op::Branch,
238                        count,
239                        dettid: tid,
240                        ..
241                    },
242                    Some(SchedEvent {
243                        dettid: next_tid, ..
244                    }),
245                    _,
246                ) if tid != next_tid => {
247                    trace!(
248                        "branch count to set before a context switch event: {}",
249                        count
250                    );
251                    Some(count)
252                }
253
254                _ => None,
255            }
256        {
257            return Some(LogicalTime::from_rcbs(*next_time_slice_in_brances as u64));
258        }
259        None
260    }
261
262    fn pop_next(&mut self) -> Option<SchedEvent> {
263        self.events_popped += 1;
264        self.cursor.next()
265    }
266
267    fn peek_next(&self) -> Option<SchedEvent> {
268        self.cursor.peek().cloned()
269    }
270
271    fn next_replay_matches(&self, observed: &SchedEvent) -> bool {
272        if let Some(expected) = self.cursor.peek() {
273            !is_hard_desync(observed, expected)
274        } else {
275            false
276        }
277    }

279    pub fn observe_event(&mut self, observed: &SchedEvent) -> ReplayAction {
280        let mytid = observed.dettid;
281        let current_ix = self.traced_event_count;
282        self.traced_event_count += 1;
283
284        if let Some(expected) = self.peek_next() {
285            debug!(
286                "[detcore, dtid {}] {}: Ran event #{} {}, current replay event: {}",
287                mytid,
288                compare_desync(observed, &expected),
289                current_ix,
290                observed,
291                expected,
292            );
293            if let Some(nxt) = self.cursor.peek_nth(1) {
294                trace!("[detcore, dtid {}] NEXT event to replay {}", mytid, nxt);
295            }
296            if observed.dettid != expected.dettid {
297                tracing::warn!(
298                    "[tracereplay] expected to be running thread {}, but running {} at event #{}",
299                    expected.dettid,
300                    observed.dettid,
301                    current_ix
302                );
303            }
304
305            let is_desync = is_desync(observed, &expected);
306            let mut in_hard_desync_mode = false;
307            if is_desync {
308                let counts = self.desync_counts.entry(mytid).or_default();
309                if is_hard_desync(observed, &expected) {
310                    if self.die_on_desync {
311                        eprintln!("Replay mode desynchronized from trace, bailing out.");
312                        return ReplayAction::Stop(StopReason::FatalDesync);
313                    }
314                    counts.hard += 1;
315                    in_hard_desync_mode = true;
316                } else {
317                    // TODO: we need to handle a mismatch in counts by doing a partial-pop on the
318                    // replayer state.
319                    counts.soft += 1;
320                }
321            }
322            self.counts_next_round(mytid, is_desync);
323
324            if is_desync && let Some(ix) = self.check_fast_forward(observed) {
325                debug_assert!(ix > 0);
326                // This counts DELETIONS (missing events from observed stream):
327                let counts = self.desync_counts.entry(mytid).or_default();
328                counts.resync_deletions += 1;
329                in_hard_desync_mode = false;
330                // is_desync = false; // Logically correct, but... unused_assignments
331            }
332
333            if !in_hard_desync_mode {
334                // Whether naturally or by fast forward, we should end up in this state before we pop:
335                if !self.next_replay_matches(observed) {
336                    tracing::warn!(
337                        "Expected match before pop, observed: {}\n Cursor at: {}",
338                        observed,
339                        self.cursor.display_first_n(10)
340                    );
341                }
342                let _ = self.pop_next();
343            }
344
345            let action = self.check_context_switch(observed);
346            if let ReplayAction::ContextSwitch(_, next_tid, _) = action
347                && in_hard_desync_mode
348            {
349                tracing::warn!(
350                    "[dtid {}] context switch to {}, but in a desynced state at this event (#{}) in --replay-schedule-from",
351                    &mytid,
352                    next_tid,
353                    current_ix
354                );
355                let counts = self.desync_counts.entry(mytid).or_default();
356                counts.at_context_switch += 1;
357            }
358            action
359        } else if self.replay_exhausted_panic {
360            eprintln!(
361                "[detcore, dtid {}] Replay trace ran out, stopping at unknown event {:?}",
362                mytid, observed
363            );
364            ReplayAction::Stop(StopReason::ReplayExausted)
365        } else {
366            info!(
367                "[detcore, dtid {}] Replay trace ran out, unknown event {:?}",
368                &mytid, observed
369            );
370            ReplayAction::Continue(None)
371        }
372    }
373
374    /// Logically advance the counting machinery to the next round.
375    fn counts_next_round(&mut self, mytid: DetTid, is_desync: bool) {
376        let counts = self.desync_counts.entry(mytid).or_default();
377        if is_desync {
378            // Set it for the next round:
379            counts.last_was_desync = true;
380        } else {
381            if counts.last_was_desync {
382                // This counts INSERTION events (new events in observed stream), because we
383                // previously desynced and left unmatched events on the replay cursor, which are
384                // now matched.
385                counts.resync_insertions += 1;
386            }
387            counts.last_was_desync = false;
388        }
389    }
390
391    /// Return how many to drop to have observed equal to the head of the replay tape,
392    /// i.e. jump past missing or mismatched events on the replay tape.
393    fn check_fast_forward(&mut self, observed: &SchedEvent) -> Option<usize> {
394        let print_win = |sz: usize| -> String {
395            let mut acc = String::new();
396            for i in 0..sz {
397                if let Some(ev) = self.cursor.peek_nth(i) {
398                    acc.push_str(&format!(" {}", ev));
399                }
400            }
401            acc
402        };
403
404        if let Some(ix) = self.cursor.prefix_contains(LOOKAHEAD_WINDOW, |expected| {
405            events_match(observed, expected)
406        }) {
407            let counts = self.desync_counts.entry(observed.dettid).or_default();
408            counts.fast_forwards += 1;
409            info!(
410                "FAST FORWARD by dropping {} events from replay to resync: {}",
411                ix,
412                print_win(ix)
413            );
414            // Drop the corresponding number of of missed/mismatched events so that the observed
415            // matches the head of the replay tape (which is to be popped).
416            self.cursor.drop(ix);
417            Some(ix)
418        } else {
419            trace!(
420                "No fast-forward possible for {} in window{}",
421                observed,
422                print_win(LOOKAHEAD_WINDOW)
423            );
424            None
425        }
426    }
427
428    fn check_context_switch(&mut self, observed: &SchedEvent) -> ReplayAction {
429        // We run after the pop of any matched action, so the next action in replay represents the
430        // FUTURE not the PRESENT/PAST:
431        if let Some(next_ev) = self.cursor.peek() {
432            let mytid = observed.dettid;
433            let time = observed.end_time.expect("timestamps required for now");
434            let is_prehook = matches!(observed.op, Op::Syscall(_, SyscallPhase::Prehook));
435
436            let next_tid = next_ev.dettid;
437            // Checking for an optional timeslice in case of RCB
438            let timeslice = self.resolve_time_to_run(next_ev);
439
440            if next_tid != observed.dettid {
441                if is_prehook {
442                    info!(
443                        "[detcore, dtid {}] CONTEXT SWITCH to {} after this syscall blocks.  Reprioritizing at time {}",
444                        &mytid, next_tid, time
445                    );
446                } else {
447                    info!(
448                        "[detcore, dtid {}] CONTEXT SWITCH to {} after the last event retired.  Reprioritizing at time {}",
449                        &mytid, next_tid, time
450                    );
451                }
452
453                // The *downgrading* of the current thread will be handled by the caller if the
454                // context switch is *now*.  If the last traced event on this thread is instead
455                // a prehook, well we don't deschedule the current thread quite yet.  Rather, we
456                // let the thread plow ahead, and actually block on the syscall, which will have
457                // the effect of descheduling the therad anyway.  After that, the priorities
458                // be set so as to make sure the correct thread (next_tid) runs.
459                //
460                // Note: for *natural* schedules that are recorded from hermit, the only reason
461                // to switch away from the thread after the prehook is if it blocked. But a
462                // synthetic schedule can choose to do this arbitrarily, and we need to assign
463                // semantics to that on replay.  If we have threads A&B, and events "A_prehook,
464                // B_*, A_posthook", then does the effect of A's syscall happen before or after
465                // B's logic?
466                ReplayAction::ContextSwitch(!is_prehook, next_tid, timeslice)
467            } else {
468                // We're still running the next event.
469                ReplayAction::Continue(timeslice)
470            }
471        } else {
472            ReplayAction::Continue(None)
473        }
474    }
