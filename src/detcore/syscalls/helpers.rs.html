44    /// Record or replay a BLOCKING syscall without stalling the current thread (and thus
45    /// deadlocking).  This uses a protocol of an extra resource request before/after the
46    /// syscall to inform the scheduler that the thread is leaving/rejoining the runnable
47    /// threads pool.
48    ///
49    /// This is only valid to use (1) in hermit record/replay modes, or (2)
50    /// when we're in "hermit run", but we're NOT sequentializing threads, because in
51    /// that case it's ok to use the blocking versions of system calls.
52    pub async fn record_or_replay_blocking<G: Guest<Self>>(
53        &self,
54        guest: &mut G,
55        call: Syscall,
56    ) -> Result<i64, Error> {
57        let dettid = guest.thread_state().dettid;
58        let op_id = ExternalOpId::new(dettid, guest.thread_state().stats.syscall_count);
59        // Internal-vs-external fd classification happens at the call sites that hold the
60        // typed, nonblockize-able syscall (see execute_nonblockable_fd_syscall):
61        // container-internal pipes are routed to the InternalIOPolling nonblockize-retry
62        // path and must NOT reach this external-blocking protocol. BlockingExternalIO
63        // deschedules the thread to run in the background and rejoin nondeterministically,
64        // which is unsafe for a pipe whose reader and writer are interdependent -- doing
65        // so is the root cause of the record/replay pipe deadlock. The remaining callers
66        // (external poll, wait4) are external by construction (their fd is not a single
67        // extractable internal pipe). Guard the invariant in debug builds while the
68        // deterministic scheduler is active. With thread sequentialization disabled,
69        // resource requests are no-ops and internal pipes intentionally use a blocking
70        // host syscall, as documented by this method.
71        debug_assert!(
72            !self.cfg.sequentialize_threads || !syscall_targets_internal_fd(guest, call),
73            "record_or_replay_blocking (BlockingExternalIO) reached for an internal pipe fd \
74             on syscall {}; internal fds must use the InternalIOPolling path",
75            call.name()
76        );
77        {
78            let mut rsrcs = Resources::new(dettid);
79            // With sequentialization enabled, only truly EXTERNAL endpoints reach here.
80            // Without it, resource_request is a no-op and internal fds may block directly.
81            rsrcs.insert(ResourceID::BlockingExternalIO(op_id), Permission::RW);
82            rsrcs.fyi(call.name());
83            resource_request(guest, rsrcs).await;
84        }
85        tracing::trace!(
86            "Guest proceeding to execute potentially blocking call {}...",
87            call.name()
88        );
89        let res = self.record_or_replay(guest, call).await;
90        // N.B. BlockingExternalIO is a "oneshot" resource, so no need to release
91        // explicitly here:
92        {
93            let mut rsrcs = Resources::new(dettid);
94            rsrcs.insert(ResourceID::BlockedExternalContinue(op_id), Permission::RW);
95            rsrcs.fyi(call.name());
96            resource_request(guest, rsrcs).await;
97        }
98        Ok(res?)
99    }

101    /// Executes a nonblockable syscall according to the following strategy:
102    /// - Record mode: Execute possibly blocking syscall
103    /// - Run mode: Transform the syscall to nonblocking if required before executing
104    ///
105    /// These are fd-oriented syscalls in the sense that whether they block or not depends
106    /// on whether NONBLOCK was set on the corresponding file descriptor.
107    pub async fn execute_nonblockable_fd_syscall<
108        G: Guest<Self>,
109        C: SyscallInfo + NonblockableSyscall + Into<Syscall>,
110    >(
111        &self,
112        guest: &mut G,
113        call: C,
114    ) -> Result<i64, Error> {
115        let wrapped: Syscall = call.into();
116
117        let action = ioaction_based_on_fd_status(guest, call);
118
119        // Is this operation on a container-INTERNAL fd (currently: pipes)? Internal
120        // pipes are made physically nonblocking even in record/replay (see
121        // handle_pipe2), so they can take the deterministic InternalIOPolling
122        // nonblockize-and-retry path. They must NOT be forced onto the
123        // BlockingExternalIO path in R/R: a pipe reader and its paired writer are not
124        // independent, so descheduling the reader as "external blocking IO" deadlocks
125        // the sequentialized scheduler (the documented R/R pipe hang). Truly external
126        // endpoints (host fds, network sockets) still use BlockingExternalIO. Sockets
127        // are left external for now: there is no internal-vs-external socket detection
128        // yet (see the handle_accept4 comment).
129        let internal_fd = syscall_targets_internal_fd(guest, wrapped);
130
131        if !self.cfg.sequentialize_threads
132            || (self.cfg.recordreplay_modes && !internal_fd)
133            || action == IOAction::Blocking
134        {
135            tracing::trace!(
136                "NonblockableSyscall: executing in blocking mode after all: {}",
137                call.name()
138            );
139            // We let these have nondeterminstic timing in record mode:
140            Ok(self.record_or_replay_blocking(guest, wrapped).await?)
141            // If in the future we want to record EXTERNAL network traffic only, we have a
142            // challenge to overcome.  We don't know if we need to record until after the
143            // accept completes, so we need an API for *post-facto* recording.
144        } else if action == IOAction::NonblockizeRetry {
145            tracing::trace!(
146                "NonblockableSyscall: converting to nonblocking syscall (internal polling): {}",
147                call.name()
148            );
149            let mut rsrc = Resources::new(guest.thread_state().dettid);
150            rsrc.insert(ResourceID::InternalIOPolling, Permission::W);
151            rsrc.fyi(call.name());
152            // In record/replay mode, route an internal-fd (pipe) read/write through the
153            // record/replay subtool so its data is captured on record and reproduced on
154            // replay (see retry_nonblocking_syscall). In plain `hermit run` there is no
155            // recorder, so execute directly (subtool = None).
156            let subtool = (self.cfg.recordreplay_modes && internal_fd).then_some(self);
157            Ok(retry_nonblocking_syscall(guest, call, rsrc, subtool).await?)
158        } else {
159            assert!(action == IOAction::PassThru);
160            tracing::trace!(
161                "NonblockableSyscall: just passing it through: {}",
162                call.name()
163            );
164            // Otherwise, the socket was already nonblocking, so we can safely execute it just once.
165            Ok(self.record_or_replay(guest, wrapped).await?)
166        }
167    }

171    /// Complete a logically blocking pipe writev after Hermit has made the pipe physically
172    /// nonblocking. A positive short write is an implementation artifact here: without
173    /// O_NONBLOCK, Linux blocks until the full vector is written unless a signal or error
174    /// interrupts it. Atomic vectors retain a private iovec snapshot for every retry; larger
175    /// vectors advance a positive short-write remainder through scalar writes.
176    pub async fn execute_blocking_pipe_writev<G: Guest<Self>>(
177        &self,
178        guest: &mut G,
179        call: syscalls::Writev,
180    ) -> Result<i64, Error> {
181        const MAX_IOVECS: usize = 1024;
182        // Linux limits a single vectored transfer to INT_MAX rounded down to a page.
183        const MAX_RW_COUNT: usize = 0x7fff_f000;
184        // Linux guarantees pipe writes through this size are atomic.
185        const PIPE_BUF: usize = 4096;
186
187        let Some(iov_addr) = call.iov() else {
188            return self.execute_nonblockable_fd_syscall(guest, call).await;
189        };
190        if call.len() == 0 || call.len() > MAX_IOVECS {
191            return self.execute_nonblockable_fd_syscall(guest, call).await;
192        }
193
194        let iovecs: Vec<(usize, usize)> = {
195            let mut raw_iovecs = vec![
196                libc::iovec {
197                    iov_base: std::ptr::null_mut(),
198                    iov_len: 0,
199                };
200                call.len()
201            ];
202            guest.memory().read_values(iov_addr, &mut raw_iovecs)?;
203            raw_iovecs
204                .into_iter()
205                .map(|iovec| (iovec.iov_base as usize, iovec.iov_len))
206                .collect()
207        };
208        let requested = iovecs.iter().try_fold(0usize, |total, (_, length)| {
209            total.checked_add(*length).ok_or(Errno::EINVAL)
210        })?;
211        if requested > isize::MAX as usize {
212            return Err(Errno::EINVAL.into());
213        }
214        let target = requested.min(MAX_RW_COUNT);
215        if target == 0 {
216            return self.execute_nonblockable_fd_syscall(guest, call).await;
217        }
218
219        let atomic_pipe_write = target <= PIPE_BUF;
220
221        tracing::trace!(
222            "NonblockableSyscall: converting to nonblocking syscall (internal polling): writev"
223        );
224        let mut resources = Resources::new(guest.thread_state().dettid);
225        resources.insert(ResourceID::InternalIOPolling, Permission::W);
226        resources.fyi(call.name());
227        let subtool = self.cfg.recordreplay_modes.then_some(self);
228        let mut current = Syscall::Writev(call);
229        let mut written_total = 0usize;
230
231        loop {
232            if resources.poll_attempt > 0
233                && resource_request(guest, resources.clone()).await == ResumeStatus::Signaled
234            {
235                break if written_total > 0 {
236                    Ok(written_total as i64)
237                } else {
238                    Err(call.signal_interrupt_errno().into())
239                };
240            }
241
242            let result = if atomic_pipe_write {
243                self.execute_atomic_pipe_writev_attempt(guest, call, &iovecs)
244                    .await
245            } else {
246                match subtool {
247                    Some(detcore) => detcore.record_or_replay(guest, current).await,
248                    None => guest.inject_with_retry(current).await,
249                }
250            };
251            match result {
252                Ok(written) if written > 0 => {
253                    let written = usize::try_from(written).map_err(|_| Errno::EIO)?;
254                    written_total = written_total.checked_add(written).ok_or(Errno::EIO)?;
255                    if written_total >= target {
256                        break Ok(written_total as i64);
257                    }
258                    if atomic_pipe_write {
259                        break Ok(written_total as i64);
260                    }
261                    current = match remaining_writev_segment(
262                        call.fd(),
263                        &iovecs,
264                        written_total,
265                        target - written_total,
266                    ) {
267                        Ok(Some(write)) => Syscall::Write(write),
268                        Ok(None) => break Ok(written_total as i64),
269                        Err(_) => break Ok(written_total as i64),
270                    };
271                }
272                Ok(0) => break Ok(written_total as i64),
273                Err(Errno::EAGAIN) => {
274                    if !atomic_pipe_write && matches!(current, Syscall::Writev(_)) {
275                        current = match remaining_writev_segment(call.fd(), &iovecs, 0, target) {
276                            Ok(Some(write)) => Syscall::Write(write),
277                            Ok(None) => break Ok(0),
278                            Err(error) => break Err(error.into()),
279                        };
280                    }
281                }
282                Err(error) => {
283                    break if written_total > 0 {
284                        Ok(written_total as i64)
285                    } else {
286                        Err(error.into())
287                    };
288                }
289                Ok(_) => break Err(Errno::EIO.into()),
290            }
291
292            resources.poll_attempt += 1;
293            tracing::trace!(
294                "Retry #{} for {}blocking pipe writev after {:?}: {}",
295                resources.poll_attempt,
296                if atomic_pipe_write { "atomic " } else { "" },
297                result,
298                call.display(&guest.memory())
299            );
300            record_retry_event(guest, call).await;
301        }
302    }

304    async fn execute_atomic_pipe_writev_attempt<G: Guest<Self>>(
305        &self,
306        guest: &mut G,
307        call: syscalls::Writev,
308        iovecs: &[(usize, usize)],
309    ) -> Result<i64, Errno> {
310        // Every backend provides at least 512 bytes of tool scratch. Linux's own fast-iovec
311        // path is smaller; this covers common vectors without consuming guest VM mappings.
312        const STACK_IOVECS: usize = 32;
313        if iovecs.len() <= STACK_IOVECS {
314            let mut stack = guest.stack().await;
315            let scratch_array = {
316                let mut raw_iovecs = [libc::iovec {
317                    iov_base: std::ptr::null_mut(),
318                    iov_len: 0,
319                }; STACK_IOVECS];
320                for (raw, (base, length)) in raw_iovecs.iter_mut().zip(iovecs) {
321                    raw.iov_base = *base as *mut libc::c_void;
322                    raw.iov_len = *length;
323                }
324                stack.push(raw_iovecs)
325            };
326            let scratch_iov: Addr<libc::iovec> = scratch_array.cast();
327            let _guard = stack
328                .commit()
329                .unwrap_or_else(|error| panic!("failed to commit atomic writev scratch: {error}"));
330            let scratch_call = call.with_iov(Some(scratch_iov));
331            return if self.cfg.recordreplay_modes {
332                self.record_or_replay(guest, scratch_call).await
333            } else {
334                guest.inject_with_retry(scratch_call).await
335            };
336        }
337
338        let mapping_len = iovecs
339            .len()
340            .checked_mul(std::mem::size_of::<libc::iovec>())
341            .expect("validated iovec count cannot overflow scratch length");
342        let mapped = guest
343            .inject_with_retry(Syscall::Mmap(
344                syscalls::Mmap::new()
345                    .with_addr(None)
346                    .with_len(mapping_len)
347                    .with_prot(ProtFlags::PROT_READ | ProtFlags::PROT_WRITE)
348                    .with_flags(MapFlags::MAP_PRIVATE | MapFlags::MAP_ANONYMOUS)
349                    .with_fd(-1)
350                    .with_offset(0),
351            ))
352            .await
353            .unwrap_or_else(|error| panic!("failed to map atomic writev scratch: {error}"));
354        let mapped = usize::try_from(mapped)
355            .unwrap_or_else(|_| panic!("atomic writev scratch mmap returned {mapped}"));
356        let scratch_iov = Addr::<libc::iovec>::from_raw(mapped)
357            .unwrap_or_else(|| panic!("atomic writev scratch mmap returned a null address"));
358        let mapping_addr: Addr<libc::c_void> = scratch_iov.cast();
359        let write_result = {
360            let raw_iovecs: Vec<libc::iovec> = iovecs
361                .iter()
362                .map(|(base, length)| libc::iovec {
363                    iov_base: *base as *mut libc::c_void,
364                    iov_len: *length,
365                })
366                .collect();
367            // SAFETY: the injected anonymous mapping is exclusively owned scratch space.
368            guest
369                .memory()
370                .write_values(unsafe { scratch_iov.into_mut() }, &raw_iovecs)
371        };
372        if let Err(write_error) = write_result {
373            guest
374                .inject_with_retry(Syscall::Munmap(
375                    syscalls::Munmap::new()
376                        .with_addr(Some(mapping_addr))
377                        .with_len(mapping_len),
378                ))
379                .await
380                .unwrap_or_else(|cleanup_error| {
381                    panic!(
382                        "failed to populate atomic writev scratch ({write_error}); cleanup failed ({cleanup_error})"
383                    )
384                });
385            panic!("failed to populate atomic writev scratch: {write_error}");
386        }
387
388        let scratch_call = call.with_iov(Some(scratch_iov));
389        let result = if self.cfg.recordreplay_modes {
390            self.record_or_replay(guest, scratch_call).await
391        } else {
392            guest.inject_with_retry(scratch_call).await
393        };
394        guest
395            .inject_with_retry(Syscall::Munmap(
396                syscalls::Munmap::new()
397                    .with_addr(Some(mapping_addr))
398                    .with_len(mapping_len),
399            ))
400            .await
401            .unwrap_or_else(|error| panic!("failed to unmap atomic writev scratch: {error}"));
402        result
403    }

406    pub fn maybe_set_nonblocking_fd<G: Guest<Self>>(&self, guest: &G, fd: i32) {
407        if self.cfg.sequentialize_threads && !self.cfg.debug_externalize_sockets {

418fn remaining_writev_segment(
419    fd: i32,
420    iovecs: &[(usize, usize)],
421    mut consumed: usize,
422    remaining_limit: usize,
423) -> Result<Option<syscalls::Write>, Errno> {
424    for (base, length) in iovecs {
425        if consumed >= *length {
426            consumed -= *length;
427            continue;
428        }
429        let base = base.checked_add(consumed).ok_or(Errno::EFAULT)?;
430        let buffer = Addr::<u8>::from_raw(base).ok_or(Errno::EFAULT)?;
431        return Ok(Some(
432            syscalls::Write::new()
433                .with_fd(fd)
434                .with_buf(Some(buffer))
435                .with_len((*length - consumed).min(remaining_limit)),
436        ));
437    }
438    Ok(None)
439}

441/// A blocking syscall that involves a fail descriptor may be handled in these three ways:
442#[derive(PartialEq, Eq, Debug)]
443pub enum IOAction {
444    /// It may physically block and we can't change that.  Treat it as ExternalBlockingIO.
445    Blocking,
446    /// We can nonblockize and retry the call.
447    NonblockizeRetry,
448    /// The call is nonblocking already, and safe to execute.
449    PassThru,
450}
451
452/// Returns strategy based on FD-based call may actually block when executed.
453pub fn ioaction_based_on_fd_status<
454    G: Guest<Detcore<T>>,
455    T: RecordOrReplay,
456    C: SyscallInfo + Into<Syscall>,
457>(
458    guest: &mut G,
459    call: C,
460) -> IOAction {
461    let wrapped: Syscall = call.into();
462    let fd = get_fd(wrapped).unwrap_or_else(|| panic!("Failed to get fd for {}", call.name()));
463    let (phys, virt) = guest
464        .thread_state()
465        .with_detfd(fd, |detfd| {
466            (detfd.physically_nonblocking(), detfd.is_nonblocking())
467        })
468        .unwrap();
469    tracing::trace!(
470        "Checking FD {} for nonblocking: physical {} / virtual {}",
471        fd,
472        phys,
473        virt
474    );
475    if virt && !phys {
476        // TF: simulate nonblocking on top of physically blocking? How?
477        panic!(
478            "Invariant violation, fd {}: we cannot simulate nonblocking behavior when set to blocking mode in the kernel.",
479            fd
480        );
481    } else if !virt && !phys {
482        // FF: logically blocking, physically blocking, this could only work with BlockingExternalIO.
483        IOAction::Blocking
484    } else if virt && phys {
485        // TT: both nonblocking, so firing once is sufficient
486        IOAction::PassThru
487    } else {
488        // FT: Need to simulate blocking on top of nonblocking.
489        IOAction::NonblockizeRetry
490    }
491}

493/// Does this single-fd syscall operate on a container-INTERNAL file descriptor?
494///
495/// Currently this recognizes pipes, whose two endpoints are always both owned by guest
496/// processes inside the deterministic container. Internal pipes are made physically
497/// nonblocking (see `handle_pipe2`) so a potentially-blocking op on them can use the
498/// deterministic `InternalIOPolling` nonblockize-and-retry strategy instead of
499/// `BlockingExternalIO`. Treating an internal pipe as external blocking IO deadlocks
500/// the sequentialized scheduler in record/replay, because a pipe reader and its paired
501/// writer are not independent.
502///
503/// Sockets are intentionally NOT classified as internal here: there is no reliable
504/// internal-vs-external socket detection yet (loopback / AF_UNIX-to-another-guest vs a
505/// real host peer), so sockets conservatively remain external. Syscalls whose fd is not
506/// directly extractable (e.g. poll/ppoll, which carry a pointer to an fd array) return
507/// false and keep their existing handling.
508pub fn syscall_targets_internal_fd<G: Guest<Detcore<T>>, T: RecordOrReplay>(
509    guest: &mut G,
510    call: Syscall,
511) -> bool {
512    match get_fd(call) {
513        Some(fd) => guest
514            .thread_state()
515            .with_detfd(fd, |detfd| matches!(detfd.ty(), FdType::Pipe))
516            .unwrap_or(false),
517        None => false,
518    }
519}

637/// A system call which may or may not block, but which can be MADE nonblocking.
638#[async_trait]
639pub trait NonblockableSyscall: SyscallInfo {
640    /// Convert the system call to a nonblocking version of itself.  Sometimes this means
641    /// setting a zero timeout, and sometimes it means something else.
642    ///
643    /// This may need to stack allocate, so it returns a StackGuard.
644    async fn into_nonblocking<T: RecordOrReplay, G: Guest<Detcore<T>>>(
645        self,
646        guest: &mut G,
647    ) -> (Self, Option<<G::Stack as Stack>::StackGuard>);
648
649    /// Check if the result (in nonblocking mode) is analogous to blocking in blocking mode.
650    /// I.e. the result means "try again".
651    fn syscall_would_have_blocked(&self, res: Result<i64, Errno>) -> bool {
652        res == Ok(0)
653    }
654
655    /// Return the errno used when a signal interrupts this internally polled syscall.
656    /// Most blocking I/O is restartable when its handler uses `SA_RESTART`.
657    fn signal_interrupt_errno(&self) -> Errno {
658        Errno::ERESTARTSYS
659    }
660
661    /// Convert a physical nonblocking completion into the result expected by the guest.
662    /// `retried` is true after a prior result was classified as blocked.
663    fn normalize_nonblocking_result(
664        &self,
665        res: Result<i64, Errno>,
666        _retried: bool,
667    ) -> Result<i64, Errno> {
668        res
669    }
670}

677}

742async fn zero_timespec<'stack, T: RecordOrReplay, G: Guest<Detcore<T>>>(
743    guest: &mut G,
744) -> (Addr<'stack, Timespec>, <G::Stack as Stack>::StackGuard) {
745    let mut stack = guest.stack().await;
746    let tp_val = Timespec {
747        tv_sec: 0,
748        tv_nsec: 0,
749    };
750    let tp = stack.push(tp_val);
751    let guard = stack.commit().expect("stack.commit to succeed");
752    (tp, guard)
753}

880/// A common helper shared among several network syscalls.
881/// We can't actually CONVERT these syscalls into nonblocking, but we can assert that they are by
882/// checking the status of their file descriptor.
883fn network_comm_syscall<T: RecordOrReplay, G: Guest<Detcore<T>>, C: SyscallInfo + Into<Syscall>>(
884    call: C,
885    guest: &mut G,
886) -> (C, Option<<G::Stack as Stack>::StackGuard>) {
887    // Already nonblocking because we've assured the socket is.
888    let fd = get_fd(call.into()).unwrap_or_else(|| {
889        panic!(
890            "network_comm_syscall called on invalid syscall / unknown fd: {}",
891            call.name()
892        );
893    });
894    guest
895        .thread_state()
896        .with_detfd(fd, |detfd| {
897            assert!(
898                detfd.physically_nonblocking(),
899                "expecting sockets/pipes to be physically nonblocking"
900            );
901        })
902        .unwrap();
903    (call, None)
904}

1040/// Transform a syscall to nonblocking, then retry it until it returns a successful result.
1041/// Retry a nonblockizable syscall (e.g. a pipe/socket read or write) until it succeeds.
1042///
1043/// `subtool` selects how each poll iteration executes the underlying syscall. Pass
1044/// `Some(detcore)` in record/replay mode for a container-INTERNAL fd (currently pipes):
1045/// each iteration is then routed through `Detcore::record_or_replay`, so the read's
1046/// bytes (and every intervening `EAGAIN`) are captured in the recording and reproduced
1047/// verbatim on replay. Without this, an internal-pipe read on the InternalIOPolling path
1048/// bypasses the recorder and replay reads live from a pipe whose cross-process writer
1049/// schedule is not reproduced -- the reader sees EOF instead of the recorded data and
1050/// replay desyncs. Pass `None` for plain `hermit run` (no recording) or for external fds.
1051pub async fn retry_nonblocking_syscall<T, G, C>(
1052    guest: &mut G,
1053    call: C,
1054    rsrc: Resources,
1055    subtool: Option<&Detcore<T>>,
1056) -> Result<i64, Error>
1057where
1058    C: NonblockableSyscall + Into<Syscall>,
1059    T: RecordOrReplay,
1060    G: Guest<Detcore<T>>,
1061{
1062    // Bogus 99 return value is dead code below:
1063    retry_nonblocking_syscall_helper(guest, call, rsrc, None, subtool).await
1064}
1065
1066/// Retry a non-blocking syscall until it succeeds. Set the timeout to zero for the actual
1067/// syscalls (retries), while monitoring the clock to see if/when the logical timeout
1068/// should trigger.  Timeout is passed as an ABSOLUTE TIME (not duration).
1069pub async fn retry_nonblocking_syscall_with_timeout<T, G, C>(
1070    guest: &mut G,
1071    call: C,
1072    rsrc: Resources,
1073    // Logical timeout:
1074    maybe_timeout: Option<LogicalTime>,
1075) -> Result<i64, Error>
1076where
1077    C: NonblockableSyscall + TimeoutableSyscall + Into<Syscall>,
1078    T: RecordOrReplay,
1079    G: Guest<Detcore<T>>,
1080{
1081    let maybe_tup = maybe_timeout.map(|t| (t, call.timeout_return_val()));
1082    // poll/epoll_wait/futex/rt_sigtimedwait keep their existing execution (raw
1083    // inject_with_retry): their record/replay handling is out of scope for the internal
1084    // pipe data-ordering fix, and their fds are not necessarily internal pipes.
1085    retry_nonblocking_syscall_helper(guest, call, rsrc, maybe_tup, None).await
1086}
1087
1088// Private helper.
1089async fn retry_nonblocking_syscall_helper<T, G, C>(
1090    guest: &mut G,
1091    call0: C,
1092    rsrc: Resources,
1093    maybe_timeout: Option<(LogicalTime, Result<i64, Errno>)>,
1094    subtool: Option<&Detcore<T>>,
1095) -> Result<i64, Error>
1096where
1097    C: NonblockableSyscall + Into<Syscall>,
1098    T: RecordOrReplay,
1099    G: Guest<Detcore<T>>,
1100{
1101    // The stack-allocated memory here needs to live across the loop, which means
1102    // surviving multiple syscall injections:
1103    let (call, _maybe_stackguard) = call0.into_nonblocking(guest).await;
1104    let mut rsrc = rsrc.clone();
1105
1106    loop {
1107        if resource_request(guest, rsrc.clone()).await == ResumeStatus::Signaled {
1108            let errno = call.signal_interrupt_errno();
1109            tracing::trace!(
1110                "retry_nonblocking_syscall: interrupted by signal before retrying {}: {:?}",
1111                call.display(&guest.memory()),
1112                errno
1113            );
1114            return Err(errno.into());
1115        }
1116        // Route through the record/replay subtool for internal pipes so each poll (an
1117        // EAGAIN, or the final data-bearing read) becomes one recorded event that replay
1118        // reproduces deterministically; otherwise execute the syscall directly.
1119        let res = match subtool {
1120            Some(detcore) => detcore.record_or_replay(guest, call).await,
1121            None => guest.inject_with_retry(call).await,
1122        };
1123        if call.syscall_would_have_blocked(res) {
1124            rsrc.poll_attempt += 1;
1125            if let Some((timeout, timeout_result)) = maybe_timeout {
1126                let new_time = thread_observe_time(guest).await;
1127                if new_time >= timeout {
1128                    tracing::trace!(
1129                        "Timing out syscall after #{} retries: {}",
1130                        rsrc.poll_attempt - 1,
1131                        call.display(&guest.memory())
1132                    );
1133                    return timeout_result.map_err(|e| e.into());
1134                } else {
1135                    tracing::trace!(
1136                        "Retry #{} for syscall due to result {:?}, {} from timeout: {}",
1137                        rsrc.poll_attempt,
1138                        res,
1139                        timeout - new_time,
1140                        call.display(&guest.memory())
1141                    );
1142                    record_retry_event(guest, call).await;
1143                }
1144            } else {
1145                tracing::trace!(
1146                    "Retry #{} for syscall due to result {:?}: {}",
1147                    rsrc.poll_attempt,
1148                    res,
1149                    call.display(&guest.memory())
1150                );
1151                record_retry_event(guest, call).await;
1152            }
1153        } else {
1154            let res = call
1155                .normalize_nonblocking_result(res, rsrc.poll_attempt > 0)
1156                .map_err(|e| e.into());
1157            tracing::trace!(
1158                "retry_nonblocking_syscall: syscall completed after {} retries: {} = {:?}",
1159                rsrc.poll_attempt,
1160                call.display(&guest.memory()),
1161                res
1162            );
1163            return res;
1164        }
1165    }
1166}

1168pub(crate) async fn record_retry_event<G, C, T>(guest: &mut G, call: C)
1169where
1170    C: SyscallInfo,
1171    T: RecordOrReplay,
1172    G: Guest<Detcore<T>>,
1173{

1199// Enrich the event with the RIP register from the current guest state, but only if it is unset.
1200pub async fn with_guest_rip<G, T>(guest: &mut G, mut event: SchedEvent) -> SchedEvent
1201where
1202    G: Guest<Detcore<T>>,
1203    T: RecordOrReplay,

1206

1226// Convert to absolute logical time point for the timeout.
1227// 0 duration means no timeout, and this will return None.
1228pub async fn nanos_duration_to_absolute_timeout<G: Guest<Detcore<T>>, T: RecordOrReplay>(
1229    guest: &mut G,
1230    timeout_nanos: u128,
1231) -> Option<LogicalTime> {
1232    if timeout_nanos > 0 {
1233        let ns_delta = Duration::from_nanos(timeout_nanos as u64);
1234        let base_time = thread_observe_time(guest).await;
1235        let target_time = base_time + ns_delta;
1236        Some(target_time)
1237    } else {
1238        None
1239    }
