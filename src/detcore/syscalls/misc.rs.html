42fn from_str(s: &str) -> [i8; 65] {
43    let mut ret: [i8; 65] = [0; 65];
44    for (i, ch) in s.bytes().take(64).enumerate() {
45        ret[i] = ch as i8;
46    }
47    ret
48}

50const GETRANDOM_ALLOWED_FLAGS: u32 = libc::GRND_NONBLOCK | libc::GRND_RANDOM | libc::GRND_INSECURE;

52// AUTONOMOUS-BOT-IMPLEMENTED
53// TODO-HUMAN-REVIEW(#545): Confirm getrandom flag, stream, and fault semantics.
54fn validate_getrandom_flags(flags: usize) -> Result<(), Errno> {
55    let flags = flags as u32;
56    let random = flags & libc::GRND_RANDOM != 0;
57    let insecure = flags & libc::GRND_INSECURE != 0;
58
59    if flags & !GETRANDOM_ALLOWED_FLAGS != 0 || (random && insecure) {
60        Err(Errno::EINVAL)
61    } else {
62        Ok(())
63    }
64}

66const RANDOM_FILL_CHUNK_BYTES: usize = 4096;
67// Linux's import_ubuf clamps getrandom requests to MAX_RW_COUNT on x86_64.
68const GETRANDOM_MAX_BYTES: usize = (i32::MAX as usize) & !4095;
69
70fn getrandom_request_len(requested: usize) -> usize {
71    requested.min(GETRANDOM_MAX_BYTES)
72}

74fn write_random_chunk(
75    mut memory: impl MemoryAccess,
76    remote_buf: AddrMut<u8>,
77    local_buf: &[u8],
78) -> Result<usize, Errno> {
79    const PTRACE_WORD_SPLIT: usize = std::mem::size_of::<u64>() / 2;
80
81    if local_buf.len() != std::mem::size_of::<u64>() {
82        return memory.write(remote_buf, local_buf);
83    }
84
85    // safeptrace uses PTRACE_POKEDATA for exactly eight bytes, which bypasses guest page
86    // protections. Split that case so getrandom observes the same EFAULT boundary as Linux.
87    let first = memory.write(remote_buf, &local_buf[..PTRACE_WORD_SPLIT])?;
88    if first < PTRACE_WORD_SPLIT {
89        return Ok(first);
90    }
91    let Some(second_buf) = remote_buf
92        .as_raw()
93        .checked_add(PTRACE_WORD_SPLIT)
94        .and_then(AddrMut::<u8>::from_raw)
95    else {
96        return Ok(first);
97    };
98    match memory.write(second_buf, &local_buf[PTRACE_WORD_SPLIT..]) {
99        Ok(second) => Ok(first + second),
100        Err(_) => Ok(first),
101    }
102}

105    fn write_arch_prctl_u64<G: Guest<Self>>(
106        &self,
107        guest: &mut G,
108        raw_addr: usize,
109        value: u64,
110    ) -> Result<i64, Error> {
111        let addr = AddrMut::<u64>::from_raw(raw_addr).ok_or(Errno::EFAULT)?;
112        guest.memory().write_value(addr, &value)?;
113        Ok(0)
114    }

117    // TODO-HUMAN-REVIEW(#539): Confirm the virtual arch_prctl control policy.
118    /// Preserve thread-local bases while hiding host CPU feature controls.
119    pub async fn handle_arch_prctl<G: Guest<Self>>(
120        &self,
121        guest: &mut G,
122        call: syscalls::ArchPrctl,
123    ) -> Result<i64, Error> {
124        let cpuid_uses_backend_policy =
125            self.cfg.virtualize_cpuid && self.cfg.cpuid_virtualized_by_backend;
126        let cpuid_uses_faulting = self.cfg.virtualize_cpuid && guest.has_cpuid_interception();
127        match call.cmd() {
128            ArchPrctlCmd::ARCH_SET_FS(_)
129            | ArchPrctlCmd::ARCH_SET_GS(_)
130            | ArchPrctlCmd::ARCH_GET_FS(_)
131            | ArchPrctlCmd::ARCH_GET_GS(_) => Ok(guest.inject(call).await?),
132
133            // KVM installs a deterministic CPUID table while leaving the instruction enabled.
134            ArchPrctlCmd::ARCH_GET_CPUID(_) if cpuid_uses_backend_policy => Ok(1),
135            ArchPrctlCmd::ARCH_SET_CPUID(value) if cpuid_uses_backend_policy => {
136                if value == 0 {
137                    Err(Errno::EPERM.into())
138                } else {
139                    Ok(0)
140                }
141            }
142
143            // When Reverie successfully disables native CPUID, Detcore answers its fault from a
144            // fixed table. Preserve that backend state and reject attempts to re-enable CPUID.
145            ArchPrctlCmd::ARCH_GET_CPUID(_) if cpuid_uses_faulting => Ok(0),
146            ArchPrctlCmd::ARCH_SET_CPUID(value) if cpuid_uses_faulting => {
147                if value == 0 {
148                    Ok(0)
149                } else {
150                    Err(Errno::EPERM.into())
151                }
152            }
153            // Reverie cannot faithfully deliver a CPUID fault requested by the tracee. In
154            // explicit host-CPUID mode, expose a fixed enabled control state and reject disable.
155            ArchPrctlCmd::ARCH_GET_CPUID(_) if !self.cfg.virtualize_cpuid => Ok(1),
156            ArchPrctlCmd::ARCH_SET_CPUID(value) if !self.cfg.virtualize_cpuid => {
157                if value == 0 {
158                    Err(Errno::EPERM.into())
159                } else {
160                    Ok(0)
161                }
162            }
163
164            // Ptrace hosts without CPUID-faulting support retain the kernel's honest state.
165            ArchPrctlCmd::ARCH_GET_CPUID(_) | ArchPrctlCmd::ARCH_SET_CPUID(_) => {
166                Ok(guest.inject(call).await?)
167            }
168
169            // Expose a conservative virtual CPU with no optional extended-state permissions.
170            ArchPrctlCmd::Other(
171                ARCH_GET_XCOMP_SUPP | ARCH_GET_XCOMP_PERM | ARCH_GET_XCOMP_GUEST_PERM,
172                addr,
173            ) => self.write_arch_prctl_u64(guest, addr, 0),
174            ArchPrctlCmd::Other(ARCH_REQ_XCOMP_PERM | ARCH_REQ_XCOMP_GUEST_PERM, _) => {
175                Err(Errno::EINVAL.into())
176            }
177
178            // Keep shadow stacks disabled in the virtual policy. Disabling an already-disabled
179            // feature is idempotent; enable/lock/unlock requests cannot be honored.
180            ArchPrctlCmd::Other(ARCH_SHSTK_STATUS, addr) => {
181                self.write_arch_prctl_u64(guest, addr, 0)
182            }
183            ArchPrctlCmd::Other(ARCH_SHSTK_DISABLE, features)
184                if features != 0 && features & !ARCH_SHSTK_VALID_MASK == 0 =>
185            {
186                Ok(0)
187            }
188            ArchPrctlCmd::Other(ARCH_SHSTK_DISABLE, _)
189            | ArchPrctlCmd::Other(ARCH_SHSTK_ENABLE | ARCH_SHSTK_LOCK | ARCH_SHSTK_UNLOCK, _) => {
190                Err(Errno::EINVAL.into())
191            }
192
193            ArchPrctlCmd::Other(_, _) => Err(Errno::EINVAL.into()),
194        }
195    }

197    /// Fill guest memory from the deterministic PRNG owned by the current thread.
198    pub(super) fn fill_random_bytes<G: Guest<Self>>(
199        &self,
200        guest: &mut G,
201        remote_buf: AddrMut<u8>,
202        len: usize,
203        source: &str,
204    ) -> Result<usize, Error> {
205        let mut local_words = [0_u64; RANDOM_FILL_CHUNK_BYTES / std::mem::size_of::<u64>()];
206        let mut hasher = DefaultHasher::new();
207        let mut written = 0;
208
209        while written < len {
210            let remote_chunk = match remote_buf
211                .as_raw()
212                .checked_add(written)
213                .and_then(AddrMut::<u8>::from_raw)
214            {
215                Some(address) => address,
216                None if written == 0 => return Err(Errno::EFAULT.into()),
217                None => break,
218            };
219            let chunk_len = (len - written).min(RANDOM_FILL_CHUNK_BYTES);
220            // safeptrace's 8-byte write fast path currently requires an aligned source buffer.
221            let local_buf = unsafe {
222                std::slice::from_raw_parts_mut(local_words.as_mut_ptr().cast::<u8>(), chunk_len)
223            };
224            guest.thread_state_mut().thread_prng().fill(local_buf);
225            let n = match write_random_chunk(guest.memory(), remote_chunk, local_buf) {
226                Ok(n) => n,
227                Err(_) if written > 0 => break,
228                Err(error) => return Err(error.into()),
229            };
230            if n == 0 {
231                if written == 0 {
232                    return Err(Errno::EFAULT.into());
233                }
234                break;
235            }
236            if cfg!(debug_assertions) {
237                Hash::hash_slice(&local_buf[..n], &mut hasher);
238            }
239            written += n;
240            if n < chunk_len {
241                break;
242            }
243        }
244
245        if cfg!(debug_assertions) {
246            detlog!(
247                "[dtid {}] USER RAND [{}] Filled guest memory with {} random bytes, hash of bytes: {}",
248                guest.thread_state().dettid,
249                source,
250                written,
251                hasher.finish()
252            );
253        }
254        Ok(written)
255    }

257    /// uname syscall
258    pub async fn handle_uname<G: Guest<Self>>(
259        &self,
260        guest: &mut G,
261        call: syscalls::Uname,
262    ) -> Result<i64, Error> {
263        let ret = self.record_or_replay(guest, call).await?;
264        if let Some(buf) = call.buf() {
265            let mut un = guest.memory().read_value(buf)?;
266            // Keep this in configured UTC: `Local` initializes libc TLS, which is unavailable
267            // while a DynamoRIO application thread is executing a client callback.
268            let epoch = guest.config().epoch;
269
270            if !guest.config().has_uts_namespace {
271                // FIXME: It should be possible to remove this once all tests
272                // are also using namespaces.
273                un.nodename = from_str(DEFAULT_HOSTNAME);
274                un.domainname = from_str(DEFAULT_HOSTNAME.split('.').next_back().unwrap_or(""));
275            }
276
277            un.release = from_str("5.2.0");
278            un.version = from_str(&format!("#1 SMP {}", epoch.format("%a %b %d %T %Z %Y")));
279            guest.memory().write_value(buf, &un)?;
280        }
281
282        Ok(ret)
283    }

285    /// Fill `getrandom(2)` requests from the current thread's seeded deterministic PRNG.
286    /// Supported blocking/source-selection flags share that always-ready stream; invalid Linux
287    /// flag combinations are rejected before guest memory is touched.
288    pub async fn handle_getrandom<G: Guest<Self>>(
289        &self,
290        guest: &mut G,
291        call: syscalls::Getrandom,
292    ) -> Result<i64, Error> {
293        validate_getrandom_flags(call.flags())?;
294        let len = getrandom_request_len(call.buflen());
295        if len == 0 {
296            return Ok(0);
297        }
298
299        let buf = call.buf().ok_or(Errno::EFAULT)?;
300
301        let n = self.fill_random_bytes(guest, buf, len, "getrandom")?;
302        Ok(n as i64)
303    }

306    pub async fn handle_setsid<G: Guest<Self>>(
307        &self,
308        guest: &mut G,
309        call: syscalls::Setsid,
310    ) -> Result<i64, Error> {
311        let res = guest.inject(call).await?;
312
313        // task is trying to become a daemon process. for more details
314        // see: https://notes.shichao.io/apue/ch13/
315        if guest.config().kill_daemons {
316            guest.daemonize().await;
317        }
318        Ok(res)
319    }

321    /// membarrier (system call).
322    ///
323    /// `membarrier(2)` issues process-wide memory barriers so that userspace can
324    /// use asymmetric fences (e.g. CPython's QSBR, RCU-style reclamation).
325    /// Detcore serializes all guest threads onto a single logical CPU with a
326    /// total memory order, so any requested barrier is *already* satisfied and
327    /// every command is a deterministic no-op. For `MEMBARRIER_CMD_QUERY` we
328    /// report the set of commands we emulate so the guest stays on this
329    /// controlled path instead of a host-dependent fallback; every other command
330    /// returns success without doing anything.
331    pub async fn handle_membarrier<G: Guest<Self>>(
332        &self,
333        guest: &mut G,
334        call: syscalls::Membarrier,
335    ) -> Result<i64, Error> {
336        // Values from <linux/membarrier.h>.
337        const MEMBARRIER_CMD_QUERY: i32 = 0;
338        const MEMBARRIER_CMD_GLOBAL: i32 = 1 << 0;
339        const MEMBARRIER_CMD_GLOBAL_EXPEDITED: i32 = 1 << 1;
340        const MEMBARRIER_CMD_REGISTER_GLOBAL_EXPEDITED: i32 = 1 << 2;
341        const MEMBARRIER_CMD_PRIVATE_EXPEDITED: i32 = 1 << 3;
342        const MEMBARRIER_CMD_REGISTER_PRIVATE_EXPEDITED: i32 = 1 << 4;
343        const SUPPORTED: i32 = MEMBARRIER_CMD_GLOBAL
344            | MEMBARRIER_CMD_GLOBAL_EXPEDITED
345            | MEMBARRIER_CMD_REGISTER_GLOBAL_EXPEDITED
346            | MEMBARRIER_CMD_PRIVATE_EXPEDITED
347            | MEMBARRIER_CMD_REGISTER_PRIVATE_EXPEDITED;
348
349        let cmd = call.cmd();
350        if cmd == MEMBARRIER_CMD_QUERY {
351            detlog!(
352                "[dtid {}] membarrier(QUERY) => reporting emulated commands {:#x}",
353                guest.thread_state().dettid,
354                SUPPORTED,
355            );
356            Ok(SUPPORTED as i64)
357        } else {
358            detlog!(
359                "[dtid {}] membarrier(cmd={}) no-op (threads are serialized on one CPU)",
360                guest.thread_state().dettid,
361                cmd,
362            );
363            Ok(0)
364        }
365    }

367    /// getcpu system call
368    pub async fn handle_getcpu<G: Guest<Self>>(
369        &self,
370        guest: &mut G,
371        call: syscalls::Getcpu,
372    ) -> Result<i64, Error> {
373        // Always set the CPU to 0.
374        if let Some(cpu) = call.cpu() {
375            guest.memory().write_value(cpu, &0)?;
376        }
377
378        // Always set the NUMA node to 0.
379        if let Some(node) = call.node() {
380            guest.memory().write_value(node, &0)?;
381        }
382
383        Ok(0)
384    }

393        for flags in [
394            0,
