39fn time_from_resources(rsrcs: &Resources) -> Option<LogicalTime> {
40    if rsrcs.resources.len() > 1 {
41        panic!(
42            "time_from_resources: multiple resource ids in resource request: {:?}",
43            rsrcs
44        );
45    }
46    for rs in rsrcs.resources.iter() {
47        if let (ResourceID::SleepUntil(tm), _) = rs {
48            return Some(*tm);
49        }
50    }
51    None
52}

54/// Flatten a `timespec` to nanoseconds. Negative fields are not valid for the
55/// timer syscalls we handle; treat them as zero rather than panicking.
56fn timespec_to_ns(ts: libc::timespec) -> u64 {
57    let secs = ts.tv_sec.max(0) as u64;
58    let nsec = ts.tv_nsec.max(0) as u64;
59    secs.saturating_mul(1_000_000_000).saturating_add(nsec)
60}
61
62/// Inverse of [`timespec_to_ns`].
63fn ns_to_timespec(ns: u64) -> libc::timespec {
64    libc::timespec {
65        tv_sec: (ns / 1_000_000_000) as libc::time_t,
66        tv_nsec: (ns % 1_000_000_000) as libc::c_long,
67    }
68}

72    pub async fn sleep_request<G: Guest<Self>>(guest: &mut G, ns_delta: Duration) -> Resources {
73        let base_time = thread_observe_time(guest).await;
74        let target_time = base_time + ns_delta;
75        let resource = ResourceID::SleepUntil(target_time);
76        guest.thread_state().mk_request(resource, Permission::W)
77    }

120    /// gettimeofday
121    pub async fn handle_gettimeofday<G: Guest<Self>>(
122        &self,
123        guest: &mut G,
124        call: syscalls::Gettimeofday,
125    ) -> Result<i64, Error> {
126        let time_ns = thread_observe_time(guest).await;
127
128        let ret = self.record_or_replay(guest, call).await?;
129
130        let mut memory = guest.memory();
131
132        let tv: Timeval = time_ns.into();
133
134        if let Some(tp) = call.tv() {
135            memory.write_value(tp, &tv)?;
136        }
137
138        Ok(ret)
139    }

141    /// time
142    pub async fn handle_time<G: Guest<Self>>(
143        &self,
144        guest: &mut G,
145        call: syscalls::Time,
146    ) -> Result<i64, Error> {
147        let time_ns = thread_observe_time(guest).await;
148        let secs = time_ns.as_secs() as i64;
149
150        if let Some(tloc) = call.tloc() {
151            let mut memory = guest.memory();
152            memory.write_value(tloc, &secs)?;
153        }
154
155        Ok(secs)
156    }

197    /// Helper function to wait a given period, which may either succeed or be interrupted by a signal.
198    /// Return 0 or EINTR respectively.
199    async fn wait_and_return<R: Guest<Self>>(
200        guest: &mut R,
201        request: Resources,
202        call: NanosleepFamily,
203    ) -> Result<i64, Error> {
204        let target_time = time_from_resources(&request).expect("a sleepuntil resource request");
205        match resource_request(guest, request).await {
206            ResumeStatus::Normal => Ok(0),
207            ResumeStatus::Signaled => {
208                let now = thread_observe_time(guest).await;
209                let delta: Duration = target_time.duration_since(now);
210                let addr2 = call.rem();
211                if let Some(addr2) = addr2 {
212                    info!(
213                        "[interrupted] sleep till (until {}), woke up {:?} early, writing into nanosleep rem argument.",
214                        target_time, delta
215                    );
216                    let t = Timespec {
217                        tv_sec: delta.as_secs() as i64,
218                        tv_nsec: delta.subsec_nanos() as i64,
219                    };
220                    guest.memory().write_value(addr2, &t)?;
221                } else {
222                    info!("[interrupted] nanosleep rem argument is null, not writing it.")
223                }
224                Err(reverie::Error::Errno(Errno::EINTR))
225            }
226        }
227    }

229    /// clock_nanosleep and nanosleep
230    pub async fn handle_nanosleep_family<R: Guest<Self>>(
231        &self,
232        guest: &mut R,
233        call: NanosleepFamily,
234    ) -> Result<i64, Error> {
235        if call.flags() > libc::TIMER_ABSTIME {
236            trace!("Unhandled clock_nanosleep flags, letting syscall through...");
237            return Ok(guest.inject(Syscall::from(call)).await?);
238        }
239
240        // TODO: use 2nd, `rem` argument when providing a way for a signal to interrupt the
241        // logical sleep.
242        let addr = call.req().ok_or(Errno::EFAULT)?;
243        let t: Timespec = guest
244            .memory()
245            .read_value(addr)
246            .expect("should be able to read from memory");
247
248        match call.flags() {
249            0 => {
250                if self.cfg.sequentialize_threads {
251                    let time = Duration::from_secs(t.tv_sec as u64)
252                        + Duration::from_nanos(t.tv_nsec as u64);
253                    let request = Self::sleep_request(guest, time).await;
254                    trace!(
255                        "nanosleep adding delta {:?} to yield request {:?}",
256                        time, &request
257                    );
258                    Self::wait_and_return(guest, request, call).await
259                } else {
260                    trace!("Not sequentializing threads, letting nanosleep through...");
261                    Ok(guest.inject(Syscall::from(call)).await?)
262                }
263            }
264            libc::TIMER_ABSTIME => {
265                let target_time = LogicalTime::from_secs(t.tv_sec as u64)
266                    + LogicalTime::from_nanos(t.tv_nsec as u64);
267                if self.cfg.sequentialize_threads {
268                    if self.cfg.virtualize_time {
269                        let request = Self::sleep_request_abs(guest, target_time).await;
270                        trace!(
271                            "nanosleep setting absolute time {:?} to yield request {:?}",
272                            target_time, &request
273                        );
274                        Self::wait_and_return(guest, request, call).await
275                    } else {
276                        // TODO T124594597: Record-replay case here, need better ideas to enable proper handling of this case.
277                        error!(
278                            "Sequentializing but not virtualizing, so can't rely on passed abs time, especially when replaying a recording, just yelding"
279                        );
280                        let request = Self::yield_request(guest);
281                        Self::wait_and_return(guest, request, call).await
282                    }
283                } else if self.cfg.virtualize_time {
284                    trace!(
285                        "Not sequentializing, but virtualizing so calculating relative time and invoking nanosleep..."
286                    );
287                    let relative_ts = Self::relative_time_from_abs_target(guest, target_time).await;
288                    let mut stack = guest.stack().await;
289                    let req = stack.push(relative_ts);
290                    stack.commit()?;
291                    let modified_call = syscalls::Nanosleep::new().with_req(Some(req));
292                    Ok(guest.inject(modified_call).await?)
293                } else {
294                    trace!(
295                        "Not sequentializing threads not virtualizing, letting nanosleep through..."
296                    );
297                    Ok(guest.inject(Syscall::from(call)).await?)
298                }
299            }
300            _ => unreachable!("Unexpected, unhandled flag value"),
301        }
302    }

304    async fn relative_time_from_abs_target<G: Guest<Self>>(
305        guest: &mut G,
306        target_time: LogicalTime,
307    ) -> Timespec {
308        let base_time = thread_observe_time(guest).await;
309        let relative_logical = target_time - base_time;
310
311        Timespec {
312            tv_sec: relative_logical.as_secs() as i64,
313            tv_nsec: relative_logical.subsec_nanos() as i64,
314        }
315    }

317    /// timer_create: allocate a per-process POSIX timer and hand back a
318    /// deterministic id. The timer's arming is tracked (in the process-local
319    /// `PosixTimers` table) but expiration signals are not delivered.
320    pub async fn handle_timer_create<G: Guest<Self>>(
321        &self,
322        guest: &mut G,
323        call: syscalls::TimerCreate,
324    ) -> Result<i64, Error> {
325        // The kernel writes the new timer id here; a null pointer is EFAULT.
326        let timerid_ptr = call.timerid().ok_or(Errno::EFAULT)?;
327        let clockid = call.clockid();
328        let id = {
329            let mut timers = guest.thread_state().posix_timers.lock().unwrap();
330            timers.create()
331        };
332        guest
333            .memory()
334            .write_value(timerid_ptr, &(id as libc::c_int))?;
335        detlog!(
336            "[dtid {}] timer_create(clockid={:?}) => deterministic timer id {} (arming tracked; signal delivery not emulated)",
337            guest.thread_state().dettid,
338            clockid,
339            id,
340        );
341        Ok(0)
342    }

344    /// timer_settime: arm or disarm a timer against the deterministic virtual
345    /// clock. The old arming is reported through `old_value` when requested.
346    pub async fn handle_timer_settime<G: Guest<Self>>(
347        &self,
348        guest: &mut G,
349        call: syscalls::TimerSettime,
350    ) -> Result<i64, Error> {
351        let id = call.timerid();
352        let new_ptr = call.new_value().ok_or(Errno::EINVAL)?;
353        let new: libc::itimerspec = guest.memory().read_value(new_ptr)?;
354        let interval_ns = timespec_to_ns(new.it_interval);
355        let value_ns = timespec_to_ns(new.it_value);
356
357        let now = thread_observe_time(guest).await;
358        let deadline = if value_ns == 0 {
359            None
360        } else if call.flags() & libc::TIMER_ABSTIME != 0 {
361            // Absolute expiration is interpreted against the same virtual clock.
362            Some(LogicalTime::from_nanos(value_ns))
363        } else {
364            Some(now + Duration::from_nanos(value_ns))
365        };
366
367        let old = {
368            let mut timers = guest.thread_state().posix_timers.lock().unwrap();
369            timers.settime(id, interval_ns, deadline, now)
370        };
371        let (old_remaining_ns, old_interval_ns) = old.ok_or(Errno::EINVAL)?;
372
373        if let Some(old_ptr) = call.old_value() {
374            let old_spec = libc::itimerspec {
375                it_interval: ns_to_timespec(old_interval_ns),
376                it_value: ns_to_timespec(old_remaining_ns),
377            };
378            guest.memory().write_value(old_ptr, &old_spec)?;
379        }
380
381        detlog!(
382            "[dtid {}] timer_settime(id={}, interval_ns={}, value_ns={}) armed against virtual clock (not delivered)",
383            guest.thread_state().dettid,
384            id,
385            interval_ns,
386            value_ns,
387        );
388        Ok(0)
389    }

391    /// timer_gettime: report the time remaining until the next expiration and
392    /// the reload interval, both computed from the virtual clock.
393    pub async fn handle_timer_gettime<G: Guest<Self>>(
394        &self,
395        guest: &mut G,
396        call: syscalls::TimerGettime,
397    ) -> Result<i64, Error> {
398        let id = call.timerid();
399        let value_ptr = call.value().ok_or(Errno::EFAULT)?;
400        let now = thread_observe_time(guest).await;
401        let cur = {
402            let timers = guest.thread_state().posix_timers.lock().unwrap();
403            timers.gettime(id, now)
404        };
405        let (remaining_ns, interval_ns) = cur.ok_or(Errno::EINVAL)?;
406        let spec = libc::itimerspec {
407            it_interval: ns_to_timespec(interval_ns),
408            it_value: ns_to_timespec(remaining_ns),
409        };
410        guest.memory().write_value(value_ptr, &spec)?;
411        Ok(0)
412    }

414    /// timer_getoverrun: we never deliver expirations, so the overrun count is
415    /// always 0 for a live timer.
416    pub async fn handle_timer_getoverrun<G: Guest<Self>>(
417        &self,
418        guest: &mut G,
419        call: syscalls::TimerGetoverrun,
420    ) -> Result<i64, Error> {
421        let id = call.timerid();
422        let exists = guest
423            .thread_state()
424            .posix_timers
425            .lock()
426            .unwrap()
427            .contains(id);
428        if exists {
429            Ok(0)
430        } else {
431            Err(Errno::EINVAL.into())
432        }
433    }

435    /// timer_delete: destroy a timer created by `timer_create`.
436    pub async fn handle_timer_delete<G: Guest<Self>>(
437        &self,
438        guest: &mut G,
439        call: syscalls::TimerDelete,
440    ) -> Result<i64, Error> {
441        let id = call.timerid();
442        let existed = {
443            let mut timers = guest.thread_state().posix_timers.lock().unwrap();
444            timers.remove(id)
445        };
446        if existed {
447            detlog!(
448                "[dtid {}] timer_delete(id={})",
449                guest.thread_state().dettid,
450                id,
451            );
452            Ok(0)
453        } else {
454            Err(Errno::EINVAL.into())
455        }
456    }
