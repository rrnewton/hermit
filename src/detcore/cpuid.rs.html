9//! cpuid interception
10
11use raw_cpuid::CpuIdResult;
12
13#[derive(Debug, Clone, Copy)]
14pub struct InterceptedCpuid();
15
16impl InterceptedCpuid {
17    pub fn new() -> Self {
18        InterceptedCpuid()
19    }
20}
21
22impl InterceptedCpuid {
23    pub fn cpuid(&self, index: u32) -> Option<CpuIdResult> {
24        let request = index as usize;
25        if request >= 0x80000000 && request < 0x80000000 + EXTENDED_CPUIDS.len() {
26            Some(EXTENDED_CPUIDS[request - 0x80000000])
27        } else if request < CPUIDS.len() {
28            Some(CPUIDS[request])
29        } else {
30            None
31        }
32    }
33}
34
35const fn cpuid_result(eax: u32, ebx: u32, ecx: u32, edx: u32) -> CpuIdResult {
36    CpuIdResult { eax, ebx, ecx, edx }
37}
