11const EXPECTED_X86_64_SYSNO_COUNT: usize = 373;

13// `Sysno` is externally `#[non_exhaustive]`. These assertions make additions,
14// removals, or a changed table endpoint fail at compile time instead of silently
15// reaching the required final arm.
16const _: () = {
17    assert!(Sysno::count() == EXPECTED_X86_64_SYSNO_COUNT);
18    assert!(Sysno::last().id() == 461);
19};

21#[derive(Debug, Clone, Copy, Eq, PartialEq)]
22/// Detcore's execution policy for a named Linux syscall.
23pub(crate) enum SyscallClassification {
24    /// Detcore models the syscall or applies an explicit deterministic refusal policy.
25    Determinized,
26    /// The syscall is intentionally forwarded under documented container assumptions.
27    PassThrough,
28    /// The syscall retains the legacy fail-closed-or-forward policy pending investigation.
29    Unclassified,
30}

448mod tests {
449    use super::*;

451    #[test]
452    fn every_pinned_sysno_has_an_explicit_classification() {
453        let mut counts = [0usize; 3];
454        // syscalls 0.6.18 `Sysno::iter()` omits `last()` due its strict loop bound.
455        for sysno in Sysno::iter().chain(std::iter::once(Sysno::last())) {
456            match classify_syscall(sysno) {
457                SyscallClassification::Determinized => counts[0] += 1,
458                SyscallClassification::PassThrough => counts[1] += 1,
459                SyscallClassification::Unclassified => counts[2] += 1,
460            }
461        }
462
463        assert_eq!(counts, [109, 39, 225]);
464        assert_eq!(counts.iter().sum::<usize>(), EXPECTED_X86_64_SYSNO_COUNT);
465    }
