18bitflags! {
19    /// stx_attributes from statx, see linux/stat.h
20    #[derive(Serialize, Deserialize, PartialEq, Eq, PartialOrd, Ord, Hash, Debug, Clone, Copy)]
21    pub struct StatxAttributes: u64 {
22        /// [I] File is compressed by the fs
23        const STATX_ATTR_COMPRESSED = 0x4;
24        /// [I] File is marked immutable
25        const STATX_ATTR_IMMUTABLE = 0x00000010;
26        /// [I] File is append-only
27        const STATX_ATTR_APPEND	= 0x00000020;
28        /// [I] File is not to be dumped
29        const STATX_ATTR_NODUMP	= 0x00000040;
30        /// [I] File requires key to decrypt in fs
31        const STATX_ATTR_ENCRYPTED = 0x00000800;
32        /// Dir: Automount trigger
33        const STATX_ATTR_AUTOMOUNT = 0x00001000;
34        /// Root of a mount
35        const STATX_ATTR_MOUNT_ROOT	= 0x00002000;
36        /// [I] Verity protected file
37        const STATX_ATTR_VERITY	= 0x00100000;
38        /// File is currently in DAX state
39        const STATX_ATTR_DAX = 0x00200000;
40    }
41}

43/// Detcore stat/statx
44#[allow(missing_docs)]
45#[derive(Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
46pub struct DetStat {
47    /// stx_mask
48    pub mask: StatxMask,
49    /// stx_mode or st_mode
50    pub mode: u32,
51    /// uid
52    pub uid: u32,
53    /// gid
54    pub gid: u32,
55    /// stx_attributes
56    pub attributes: StatxAttributes,
57    /// stx_attributes_mask
58    pub attributes_mask: StatxAttributes,
59    /// stx_dev (maj << 32 | min)
60    pub dev: u64,
61    /// stx_rdev (maj << 32 | min)
62    pub rdev: u64,
63    /// inode
64    pub inode: u64,
65    /// stx_nlink
66    pub nlink: u64,
67    /// stx_size
68    pub size: i64,
69    /// stx_blksize,
70    pub blksize: i64,
71    /// stx_blocks
72    pub blocks: i64,
73    /// atime
74    pub atime: Timespec,
75    /// btime
76    pub btime: Timespec,
77    /// ctime
78    pub ctime: Timespec,
79    /// mtime
80    pub mtime: Timespec,
81}
82
83impl Default for DetStat {
84    fn default() -> Self {
85        let statx: libc::statx = unsafe { std::mem::zeroed() };
86        let mut stat: Self = statx.into();
87        stat.mask = StatxMask::STATX_BASIC_STATS;
88        stat.nlink = 1;
89        stat
90    }
91}

93impl fmt::Debug for DetStat {
94    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
95        f.debug_struct("DetStat")
96            .field("mode", &self.mode)
97            .field("uid", &self.uid)
98            .field("gid", &self.gid)
99            .field("dev", &self.dev)
100            .field("rdev", &self.rdev)
101            .field("inode", &self.inode)
102            .field("nlink", &self.nlink)
103            .field("size", &self.size)
104            .field("blksize", &self.blksize)
105            .field("blocks", &self.blocks)
106            .field("atime", &self.atime)
107            .field("btime", &self.btime)
108            .field("ctime", &self.ctime)
109            .field("mtime", &self.mtime)
110            .finish()
111    }
112}

114impl From<&libc::stat> for DetStat {
115    fn from(st: &libc::stat) -> Self {
116        DetStat {
117            mask: StatxMask::STATX_BASIC_STATS,
118            mode: st.st_mode,
119            uid: st.st_uid,
120            gid: st.st_gid,
121            attributes: StatxAttributes::empty(),
122            attributes_mask: StatxAttributes::empty(),
123            dev: st.st_dev,
124            rdev: st.st_rdev,
125            inode: st.st_ino,
126            nlink: st.st_nlink,
127            size: st.st_size,
128            blksize: st.st_blksize,
129            blocks: st.st_blocks,
130            atime: Timespec {
131                tv_sec: st.st_atime as _,
132                tv_nsec: st.st_atime_nsec as _,
133            },
134            btime: Timespec {
135                tv_sec: 0,
136                tv_nsec: 0,
137            },
138            ctime: Timespec {
139                tv_sec: st.st_ctime as _,
140                tv_nsec: st.st_ctime_nsec as _,
141            },
142            mtime: Timespec {
143                tv_sec: st.st_mtime as _,
144                tv_nsec: st.st_mtime_nsec as _,
145            },
146        }
147    }

155impl From<&DetStat> for libc::stat {
156    fn from(st: &DetStat) -> Self {
157        // libc::stat doesn't provide `impl Default`
158        let mut stat: libc::stat = unsafe { std::mem::zeroed() };
159        stat.st_dev = st.dev;
160        stat.st_ino = st.inode;
161        stat.st_nlink = st.nlink;
162        stat.st_mode = st.mode;
163        stat.st_uid = st.uid;
164        stat.st_gid = st.gid;
165        stat.st_rdev = st.rdev;
166        stat.st_size = st.size;
167        stat.st_blksize = st.blksize;
168        stat.st_blocks = st.blocks;
169        stat.st_atime = st.atime.tv_sec;
170        stat.st_atime_nsec = st.atime.tv_nsec;
171        stat.st_mtime = st.mtime.tv_sec;
172        stat.st_mtime_nsec = st.mtime.tv_nsec;
173        stat.st_ctime = st.ctime.tv_sec;
174        stat.st_ctime_nsec = st.ctime.tv_nsec;
175        stat
176    }
177}

186    fn from(st: &libc::statx) -> Self {
187        DetStat {
188            mask: StatxMask::from_bits_truncate(st.stx_mask),
189            mode: st.stx_mode as _,

191            gid: st.stx_gid,
192            attributes: StatxAttributes::from_bits_truncate(st.stx_attributes),
193            attributes_mask: StatxAttributes::from_bits_truncate(st.stx_attributes_mask),
194            dev: (st.stx_dev_major as u64) << 32 | st.stx_dev_minor as u64,
195            rdev: (st.stx_rdev_major as u64) << 32 | st.stx_rdev_minor as u64,
196            inode: st.stx_ino,

198            size: st.stx_size as i64,
199            blksize: st.stx_blksize as i64,
200            blocks: st.stx_blocks as i64,
201            atime: st.stx_atime.into(),
202            btime: st.stx_btime.into(),
203            ctime: st.stx_ctime.into(),
204            mtime: st.stx_mtime.into(),

215impl From<&DetStat> for libc::statx {
216    fn from(st: &DetStat) -> Self {
217        // libc::statx doesn't provide `impl Default`
218        let mut statx: libc::statx = unsafe { std::mem::zeroed() };
219        statx.stx_mask = st.mask.bits();
220        statx.stx_blksize = st.blksize as u32;
221        statx.stx_attributes = st.attributes.bits();
222        statx.stx_nlink = st.nlink as u32;
223        statx.stx_uid = st.uid;
224        statx.stx_gid = st.gid;
225        statx.stx_mode = st.mode as u16;
226        statx.stx_ino = st.inode;
227        statx.stx_size = st.size as u64;
228        statx.stx_blocks = st.blocks as u64;
229        statx.stx_attributes_mask = st.attributes_mask.bits();
230        statx.stx_atime = st.atime.into();
231        statx.stx_btime = st.btime.into();
232        statx.stx_ctime = st.ctime.into();
233        statx.stx_mtime = st.mtime.into();
234        statx.stx_rdev_major = (st.rdev >> 32) as u32;
235        statx.stx_rdev_minor = (st.rdev & 0xffffffffu64) as u32;
236        statx.stx_dev_major = (st.dev >> 32) as u32;
237        statx.stx_dev_minor = (st.dev & 0xffffffff) as u32;
238        statx
239    }
240}
