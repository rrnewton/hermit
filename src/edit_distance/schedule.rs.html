25#[derive(Debug, Copy, Clone, Serialize, Deserialize, Hash, Eq, PartialEq)]
26pub struct MiniSchedEvent {
27    #[serde(alias = "dettid")]
28    tid: u32,

36/// Counts the number of events for each thread.
37#[allow(dead_code)]
38fn count_thread_events(schedule: &Schedule) -> HashMap<u32, usize> {
39    let mut result = HashMap::new();
40
41    for event in schedule {
42        *result.entry(event.tid).or_insert(0) += event.count;
43    }
44
45    result
46}
47
48/// Counts the number of events for each thread.
49#[allow(dead_code)]
50fn partition_thread_events(schedule: Schedule) -> HashMap<u32, Schedule> {
51    let mut result = HashMap::<u32, Vec<_>>::new();
52
53    for event in schedule {
54        result.entry(event.tid).or_default().push(event);
55    }
56
57    result
58}
59
60/// Prunes the passing schedule such that it is the same length as the failing
61/// schedule. This is done by partitioning each schedule event by its thread ID.
62#[allow(dead_code)]
63pub fn prune_schedule(passing: Schedule, failing: &Schedule) -> Schedule {
64    assert!(passing.len() >= failing.len());
65
66    let mut result = Vec::with_capacity(failing.len());
67    let mut failing_count = count_thread_events(failing);
68
69    for event in passing {
70        if let Some(count) = failing_count.get_mut(&event.tid)
71            && *count > 0
72        {
73            result.push(event);
74            *count = count.saturating_sub(event.count);
75        }
76    }
77
78    result
79}
