38/// A Reverie tool that records syscalls. Note that only syscalls that cannot be
39/// made deterministic are forwarded to this tool.
40#[derive(Default, Serialize, Deserialize)]
41pub struct Recorder {
42    // TODO: We'll need to keep track of file descriptors here in order to
43    // determine if a file descriptor should be fully recorded or simply cached
44    // with a reflink. We can use `fstatfs` to figure out if the target file
45    // system supports reflinks or not. All other file systems will need their
46    // file interactions to be recorded on the syscall level.
47
48    // Keep track of the data directory. Each thread uses this path to open its
49    // event stream.
50    data: PathBuf,
51}

58    fn new(_pid: Pid, cfg: &<Self::GlobalState as GlobalTool>::Config) -> Self {

64    fn init_thread_state(
65        &self,
66        child: Tid,
67        _parent: Option<(Tid, &Self::ThreadState)>,
68    ) -> Self::ThreadState {
69        // We have to unwrap because there is no way to handle errors here.
70        EventWriter::create(&self.data, child).unwrap_or_else(|err| {
71            panic!(
72                "Failed to create {:?} for thread {}: {}",
73                self.data, child, err
74            )
75        })
76    }

148    async fn handle_syscall_event<G: Guest<Self>>(
149        &self,
150        guest: &mut G,
151        syscall: Syscall,
152    ) -> Result<i64, Error> {
153        self.record_raw_syscall(guest, syscall);
154        self.record_exec_path(guest, syscall);
155
156        Ok(match syscall {
157            // We must let through execve without any modification. Recording
158            // events for these is hard because execve only returns upon
159            // failure.
160            Syscall::Execve(_) => guest.inject(syscall).await,
161            Syscall::Execveat(_) => guest.inject(syscall).await,
162            Syscall::Brk(_) => self.let_through(guest, syscall).await,
163            Syscall::Mprotect(_) => self.let_through(guest, syscall).await,
164            Syscall::ArchPrctl(_) => {
165                // To properly handle arch_prctl, we should prevent calls from
166                // using ARCH_SET_CPUID since we already do that for the
167                // tracees. However, it is rare for programs to use
168                // ARCH_SET_CPUID. For all other arch_prctl subfunctions, we
169                // should let it through.
170                self.let_through(guest, syscall).await
171            }
172            Syscall::Read(syscall) => self.handle_read(guest, syscall).await,
173            Syscall::Pread64(syscall) => self.handle_pread64(guest, syscall).await,
174            Syscall::Readv(syscall) => {
175                self.handle_readv_family(
176                    guest,
177                    syscall.iov().map(|a| a.as_raw()),
178                    syscall.len(),
179                    syscall.into(),
180                )
181                .await
182            }
183            Syscall::Preadv(syscall) => {
184                self.handle_readv_family(
185                    guest,
186                    syscall.iov().map(|a| a.as_raw()),
187                    syscall.iov_len(),
188                    syscall.into(),
189                )
190                .await
191            }
192            Syscall::Preadv2(syscall) => {
193                self.handle_readv_family(
194                    guest,
195                    syscall.iov().map(|a| a.as_raw()),
196                    syscall.iov_len() as usize,
197                    syscall.into(),
198                )
199                .await
200            }
201            Syscall::Recvfrom(syscall) => self.handle_recvfrom(guest, syscall).await,
202            Syscall::Recvmsg(syscall) => self.handle_recvmsg(guest, syscall).await,
203            Syscall::Write(syscall) => self.handle_write_family(guest, syscall.into()).await,
204            Syscall::Pwrite64(syscall) => self.handle_write_family(guest, syscall.into()).await,
205            Syscall::Writev(syscall) => self.handle_write_family(guest, syscall.into()).await,
206            Syscall::Pwritev(syscall) => self.handle_write_family(guest, syscall.into()).await,
207            Syscall::Pwritev2(syscall) => self.handle_write_family(guest, syscall.into()).await,
208            Syscall::Access(_) => self.handle_simple(guest, syscall).await,
209            Syscall::Lseek(_) => self.handle_simple(guest, syscall).await,
210            Syscall::Stat(syscall) => self.handle_stat_family(guest, syscall.into()).await,
211            Syscall::Fstat(syscall) => self.handle_stat_family(guest, syscall.into()).await,
212            Syscall::Lstat(syscall) => self.handle_stat_family(guest, syscall.into()).await,
213            Syscall::Newfstatat(syscall) => self.handle_stat_family(guest, syscall.into()).await,
214            Syscall::Statfs(syscall) => {
215                self.handle_statfs(guest, syscall.into(), syscall.buf())
216                    .await
217            }
218            Syscall::Fstatfs(syscall) => {
219                self.handle_statfs(guest, syscall.into(), syscall.buf())
220                    .await
221            }
222            Syscall::Statx(syscall) => self.handle_statx(guest, syscall).await,
223            Syscall::Getdents(syscall) => self.handle_getdents(guest, syscall).await,
224            Syscall::Getdents64(syscall) => self.handle_getdents64(guest, syscall).await,
225            Syscall::Mmap(syscall) => self.handle_mmap(guest, syscall).await,
226            Syscall::Munmap(_) => self.let_through(guest, syscall).await,
227            Syscall::Open(_) => self.handle_simple(guest, syscall).await,
228            Syscall::Openat(_) => self.handle_simple(guest, syscall).await,
229            Syscall::Close(_) => self.handle_simple(guest, syscall).await,
230            Syscall::Fchdir(_) => self.handle_simple(guest, syscall).await,
231            Syscall::Fadvise64(_) => self.handle_simple(guest, syscall).await,
232            Syscall::Flock(_) => self.handle_simple(guest, syscall).await,
233            Syscall::Ftruncate(_) => self.handle_simple(guest, syscall).await,
234            Syscall::Dup(_) => self.handle_simple(guest, syscall).await,
235            Syscall::Dup2(_) => self.handle_simple(guest, syscall).await,
236            Syscall::Dup3(_) => self.handle_simple(guest, syscall).await,
237            Syscall::Ioctl(syscall) => self.handle_ioctl(guest, syscall).await,
238            Syscall::Socket(_) => self.handle_simple(guest, syscall).await,
239            Syscall::ClockGettime(syscall) => self.handle_clock_gettime(guest, syscall).await,
240            Syscall::Gettimeofday(syscall) => self.handle_gettimeofday(guest, syscall).await,
241            Syscall::Settimeofday(_) => self.handle_simple(guest, syscall).await,
242            Syscall::Time(syscall) => self.handle_time(guest, syscall).await,
243            Syscall::Setsockopt(_) => self.handle_simple(guest, syscall).await,
244            // FIXME: Not all fcntl cases are simple.
245            Syscall::Fcntl(_) => self.handle_simple(guest, syscall).await,
246            Syscall::Connect(_) => self.handle_simple(guest, syscall).await,
247            Syscall::Sendto(_) => self.handle_simple(guest, syscall).await,
248            Syscall::Sendmsg(_) => self.handle_simple(guest, syscall).await,
249            Syscall::Poll(syscall) => self.handle_poll(guest, syscall).await,
250            Syscall::Ppoll(syscall) => self.handle_ppoll(guest, syscall).await,
251            Syscall::EpollWait(syscall) => self.handle_epoll_wait(guest, syscall).await,
252            Syscall::Getsockopt(syscall) => self.handle_sockopt_family(guest, syscall.into()).await,
253            Syscall::Getpeername(syscall) => {
254                self.handle_sockopt_family(guest, syscall.into()).await
255            }
256            Syscall::Getsockname(syscall) => {
257                self.handle_sockopt_family(guest, syscall.into()).await
258            }
259            Syscall::Getrandom(syscall) => self.handle_getrandom(guest, syscall).await,
260            Syscall::Readlink(syscall) => self.handle_readlink(guest, syscall).await,
261            Syscall::Mkdir(_) => self.handle_simple(guest, syscall).await,
262            Syscall::Unlink(_) => self.handle_simple(guest, syscall).await,
263            Syscall::Unlinkat(_) => self.handle_simple(guest, syscall).await,
264            unsupported => return Ok(guest.inject(unsupported).await?),
265        }?)
266    }

268    async fn handle_rdtsc_event<G: Guest<Self>>(
269        &self,
270        guest: &mut G,
271        request: Rdtsc,
272    ) -> Result<RdtscResult, Errno> {
273        let result = RdtscResult::new(request);
274        self.record_event(guest, Ok(SyscallEvent::Rdtsc(result)));
275        Ok(result)
276    }

279impl Recorder {
280    fn record_raw_syscall<G: Guest<Self>>(&self, guest: &mut G, syscall: Syscall) {
281        let debug_event = DebugEvent::new(syscall, &guest.memory());
282        guest
283            .thread_state_mut()
284            .push_debug_event(debug_event)
285            .unwrap();
286    }
287
288    /// Records the absolute path of any executable that the guest execs, so that
289    /// the replayer can make the same binary available inside its chroot.
290    ///
291    /// Without this, a guest process that forks and execs another binary (for
292    /// example a shell running an external command) would desynchronize on
293    /// replay: the injected `execve` fails with `ENOENT` inside the chroot,
294    /// causing the guest to take a different code path than it did while
295    /// recording.
296    fn record_exec_path<G: Guest<Self>>(&self, guest: &mut G, syscall: Syscall) {
297        let path = match syscall {
298            Syscall::Execve(call) => call.path().map(|p| p.read(&guest.memory())),
299            // Only AT_FDCWD execveat calls carry a path we can resolve without
300            // reconstructing the guest's fd table; dirfd-relative execs are rare
301            // and are skipped (best effort).
302            Syscall::Execveat(call) if call.dirfd() == libc::AT_FDCWD => {
303                call.path().map(|p| p.read(&guest.memory()))
304            }
305            _ => return,
306        };
307        let Some(Ok(path)) = path else {
308            return;
309        };
310        // Only absolute paths can be reproduced in the chroot without also
311        // knowing the guest's working directory at exec time.
312        if !path.is_absolute() {
313            return;
314        }
315        if let Err(err) = self.append_exec_path(&path) {
316            tracing::warn!("Failed to record exec path {:?}: {}", path, err);
317        }
318    }
319
320    /// Appends a single executable path to the recording's `exec_paths` file.
321    ///
322    /// The file is opened in append mode for every exec so that concurrent guest
323    /// threads each contribute their targets; a single small `write` is atomic
324    /// under `O_APPEND` on Linux. Duplicates are deduplicated by the replayer.
325    fn append_exec_path(&self, path: &std::path::Path) -> std::io::Result<()> {
326        use std::io::Write;
327        use std::os::unix::ffi::OsStrExt;
328
329        let bytes = path.as_os_str().as_bytes();
330        // Paths are newline-delimited in the manifest; skip the (pathological)
331        // case of an embedded newline rather than corrupt the file.
332        if bytes.contains(&b'\n') {
333            return Ok(());
334        }
335
336        let mut file = std::fs::OpenOptions::new()
337            .create(true)
338            .append(true)
339            .open(self.data.join(crate::consts::EXEC_PATHS_NAME))?;
340        let mut line = Vec::with_capacity(bytes.len() + 1);
341        line.extend_from_slice(bytes);
342        line.push(b'\n');
343        file.write_all(&line)
344    }
345
346    fn record_event<G: Guest<Self>>(&self, guest: &mut G, event: Result<SyscallEvent, Errno>) {
347        // Record the event.
348        guest
349            .thread_state_mut()
350            .push_event(Event { event })
351            // TODO: Log errors instead of panicking.
352            .unwrap();
353    }
354
355    /// Called for syscalls to explicitly let through. This should only be called
356    /// for syscalls that cannot be recorded and are necessary for the program to
357    /// function correctly. Examples of syscalls that fall into this category are
358    /// ones that help with memory management (e.g., `brk`, `mprotect`, `mmap`,
359    /// or `munmap`) or process management (e.g., `fork`, `vfork`, `clone`).
360    ///
361    /// For these syscalls, we don't really need to record anything, but we
362    /// record their arguments to detect any desynchronization.
363    async fn let_through<G: Guest<Self>>(
364        &self,
365        guest: &mut G,
366        syscall: Syscall,
367    ) -> Result<i64, Errno> {
368        guest.inject(syscall).await
369    }
370
371    /// Handles a syscall whose only value we care about is the return value
372    /// (i.e., simple syscalls).
373    ///
374    /// For recording, this means we only record the return value of the syscall.
375    /// For replay, this means we substitute the return value in lieu of actually
376    /// performing the injection.
377    ///
378    /// The syscall must have two properties satisfied for this to be called:
379    ///  1. The syscall must only have "input" arguments. That is, all arguments
380    ///     must either be values or const pointers.
381    ///  2. The execution of the program must not depend on anything else other
382    ///     than the return value of the syscall. For example, `mmap` would violate
383    ///     this rule since it affects later memory access.
384    ///
385    /// There are many syscalls who satisfy these two requirements.
386    async fn handle_simple<G: Guest<Self>>(
387        &self,
388        guest: &mut G,
389        syscall: Syscall,
390    ) -> Result<i64, Errno> {
391        let result = guest.inject(syscall).await;
392
393        self.record_event(guest, result.map(SyscallEvent::Return));
394
395        result
396    }
