32/// Read the first `length` output bytes of a vectored read from the guest's
33/// `iovec` array, flattened in read order. `length` is the syscall return value,
34/// which may be smaller than the total iovec capacity (a short read), so we stop
35/// once `length` bytes have been collected.
36fn read_iovec_output<M: MemoryAccess>(
37    memory: &M,
38    iov_addr: Option<usize>,
39    iovcnt: usize,
40    length: usize,
41) -> Result<Vec<u8>, Errno> {
42    let mut buf = vec![0u8; length];
43    if length == 0 {
44        return Ok(buf);
45    }
46    let addr = iov_addr
47        .and_then(Addr::<libc::iovec>::from_raw)
48        .ok_or(Errno::EFAULT)?;
49    let mut iovecs = vec![
50        libc::iovec {
51            iov_base: std::ptr::null_mut(),
52            iov_len: 0,
53        };
54        iovcnt
55    ];
56    memory.read_values(addr, &mut iovecs)?;
57
58    let mut filled = 0;
59    for iovec in iovecs {
60        if filled == length {
61            break;
62        }
63        let take = (length - filled).min(iovec.iov_len);
64        if take == 0 {
65            continue;
66        }
67        let src = Addr::<u8>::from_raw(iovec.iov_base as usize).ok_or(Errno::EFAULT)?;
68        memory.read_exact(src, &mut buf[filled..filled + take])?;
69        filled += take;
70    }
71    Ok(buf)
72}

74impl Recorder {
75    /// Records the vectored read family (`readv`/`preadv`/`preadv2`). Writes only
76    /// need their return count (see `handle_write_family`), but vectored reads
77    /// scatter output across guest `iovec` buffers, so we capture the exact
78    /// returned bytes flattened in read order.
79    pub(super) async fn handle_readv_family<G: Guest<Self>>(
80        &self,
81        guest: &mut G,
82        iov_addr: Option<usize>,
83        iovcnt: usize,
84        syscall: Syscall,
85    ) -> Result<i64, Errno> {
86        let result = guest.inject(syscall).await;
87
88        self.record_event(
89            guest,
90            result.and_then(|length| {
91                let buf = read_iovec_output(&guest.memory(), iov_addr, iovcnt, length as usize)?;
92                Ok(SyscallEvent::Readv(buf))
93            }),
94        );
95
96        result
97    }

99    pub(super) async fn handle_read<G: Guest<Self>>(
100        &self,
101        guest: &mut G,
102        syscall: Read,
103    ) -> Result<i64, Errno> {
104        let result = guest.inject(syscall).await;
105
106        self.record_event(
107            guest,
108            result.and_then(|length| {
109                let mut buf = vec![0; length as usize];
110                let addr = syscall.buf().ok_or(Errno::EFAULT)?;
111                guest.memory().read_exact(addr, &mut buf)?;
112                Ok(SyscallEvent::Bytes(buf))
113            }),
114        );
115
116        result
117    }

151    pub(super) async fn handle_stat_family<G: Guest<Self>>(
152        &self,
153        guest: &mut G,
154        syscall: StatFamily,
155    ) -> Result<i64, Errno> {
156        let result = guest.inject(Syscall::from(syscall)).await;
157
158        self.record_event(
159            guest,
160            result.and_then(|ret| {
161                debug_assert_eq!(ret, 0);
162                let statbuf = syscall.stat().ok_or(Errno::EFAULT)?.read(&guest.memory())?;
163                Ok(SyscallEvent::Stat(StatEvent { statbuf }))
164            }),
165        );
166
167        result
168    }

170    pub(super) async fn handle_statfs<G: Guest<Self>>(
171        &self,
172        guest: &mut G,
173        syscall: Syscall,
174        buf: Option<AddrMut<'_, libc::statfs>>,
175    ) -> Result<i64, Errno> {
176        let result = guest.inject(syscall).await;
177
178        self.record_event(
179            guest,
180            result.and_then(|ret| {
181                debug_assert_eq!(ret, 0);
182                let mut bytes = vec![0; std::mem::size_of::<libc::statfs>()];
183                guest
184                    .memory()
185                    .read_exact(buf.ok_or(Errno::EFAULT)?.cast(), &mut bytes)?;
186                Ok(SyscallEvent::Statfs(bytes))
187            }),
188        );
189
190        result
191    }

215    /// ioctl is a beast of a syscall. We try to handle the common cases here.
216    pub(super) async fn handle_ioctl<G: Guest<Self>>(
217        &self,
218        guest: &mut G,
219        syscall: Ioctl,
220    ) -> Result<i64, Errno> {
221        let request = syscall.request();
222
223        if let Some(error) = deterministic_ioctl_error(&request) {
224            self.record_event(guest, Err(error));
225            return Err(error);
226        }
227
228        let ret = guest.inject(syscall).await.inspect_err(|&err| {
229            self.record_event(guest, Err(err));
230        })?;
231
232        if matches!(
233            request,
234            ioctl::Request::FIOCLEX | ioctl::Request::FIONCLEX | ioctl::Request::FIONBIO(_)
235        ) {
236            self.record_event(guest, Ok(SyscallEvent::Return(ret)));
237        } else if let Some(output) = request.read_output(&guest.memory()).transpose() {
238            // This ioctl request has an associated output.
239            self.record_event(guest, output.map(SyscallEvent::Ioctl));
240        } else {
241            self.record_event(guest, Ok(SyscallEvent::Return(ret)));
242        }
243
244        Ok(ret)
245    }
