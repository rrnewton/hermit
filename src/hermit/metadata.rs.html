21/// Hermit record version. Recorded as part of hermit-record, hermit-replay
22/// will check this version and will fail if hermit-record version is newer.
23#[derive(
24    Debug,
25    Clone,
26    Copy,
27    Eq,
28    PartialEq,
29    Ord,
30    PartialOrd,
31    Deserialize,
32    Serialize
33)]
34#[repr(transparent)]
35pub struct RecordVersion(u32);
36impl RecordVersion {
37    /// Check if the recorder/replayer version is compatible with a given
38    /// recording (trace).
39    pub fn compatible_with(&self, other: &RecordVersion) -> bool {
40        self.0 >= other.0 && other.0 >= MIN_COMPATIBLE_RECORD_VERSION.0
41    }
42}
43
44/// hermit record/replay version.
45// NB: Increase the version number when there's any breaking changes, i.e.:
46// when new syscalls are added.
47const MIN_COMPATIBLE_RECORD_VERSION: RecordVersion = RecordVersion(0x102);
48pub(crate) const RECORD_VERSION: RecordVersion = RecordVersion(0x102);
49
50/// Metadata associated with the recording. This is serialized as a JSON file.
51#[derive(Debug, Serialize, Deserialize)]
52pub struct Metadata {
53    /// The real path to the program.
54    pub exe: PathBuf,
55    /// The name of the program.
56    pub program: String,
57    /// The first argument passed to the program.
58    pub arg0: String,
59    /// Program arguments (not including arg0).
60    pub args: Vec<String>,
61    /// The working directory of the program.
62    pub current_dir: PathBuf,
63    /// The hostname in the UTS namespace used by the program.
64    pub hostname: Option<String>,
65    /// The domainname in the UTS namespace used by the program.
66    pub domainname: Option<String>,
67    /// The environment variables used by the program.
68    pub envs: BTreeMap<String, String>,
69    /// Hermit record/replay version.
70    pub version: RecordVersion,
71}
72
73impl Metadata {
74    /// Creates a new metadata object, populating it with information about a
75    /// command.
76    pub fn new(command: &Command) -> Result<Self, Error> {
77        let exe = command.find_program()?;
78
79        let program = command.get_program().to_string_lossy().into_owned();
80        let arg0 = command.get_arg0().to_string_lossy().into_owned();
81
82        let args = command
83            .get_args()
84            .map(|arg| arg.to_string_lossy().into_owned())
85            .collect();
86
87        let envs = command
88            .get_captured_envs()
89            .into_iter()
90            .map(|(k, v)| {
91                (
92                    k.to_string_lossy().into_owned(),
93                    v.to_string_lossy().into_owned(),
94                )
95            })
96            .collect();
97
98        let current_dir = command
99            .get_current_dir()
100            .map_or_else(|| env::current_dir().unwrap(), ToOwned::to_owned);
101
102        let hostname = command
103            .get_hostname()
104            .map(|s| s.to_string_lossy().into_owned());
105        let domainname = command
106            .get_domainname()
107            .map(|s| s.to_string_lossy().into_owned());
108
109        Ok(Self {
110            exe,
111            program,
112            arg0,
113            args,
114            current_dir,
115            hostname,
116            domainname,
117            envs,
118            version: RECORD_VERSION,
119        })
120    }
121
122    /// Constructs a command from the metadata.
123    pub fn command(&self) -> Command {
124        // NOTE: We bypass the normal $PATH search here by passing in the
125        // absolute path to the program directly.
126        let mut command = Command::new(&self.exe);
127        command.arg0(&self.arg0);
128        command.args(&self.args);
129        command.env_clear();
130        command.envs(&self.envs);
131        command.current_dir(&self.current_dir);
132
133        if let Some(hostname) = &self.hostname {
134            command.hostname(hostname);
135        }
136
137        if let Some(domainname) = &self.domainname {
138            command.domainname(domainname);
139        }
140
141        command
142    }
143}

146pub fn record_or_replay_config(data: &Path) -> detcore::Config {
147    // NOTE: Record and replay should use the exact same detcore
148    // configuration. Otherwise, the behavior of the program could diverge
149    // during replay.
150    let default_config: detcore::Config = Default::default();
151    let mut config = detcore::Config {
152        panic_on_unsupported_syscalls: false,
153        sequentialize_threads: true,
154        runs_post_fork: default_config.runs_post_fork,
155        // Record/replay keeps partial Detcore subscription; madvise policy semantics
156        // begin in v0x102.
157        passthru_opt: true,
158        deterministic_io: false,
159        virtualize_time: false,
160        virtualize_metadata: false,
161        virtualize_cpuid: true,
162        cpuid_virtualized_by_backend: false,
163        backend_supports_madvise: true,
164        has_uts_namespace: true,
165        // The path to the directory where syscalls will be recorded.
166        replay_data: Some(data.to_path_buf()),
167        clock_multiplier: None,
168        epoch: default_config.epoch,
169        gdbserver: false,
170        gdbserver_port: default_config.gdbserver_port,
171        kill_daemons: default_config.kill_daemons,
172        preemption_timeout: default_config.preemption_timeout,
173        seed: default_config.seed,
174        rng_seed: default_config.rng_seed,
175        imprecise_timers: false,
176        chaos: false,
177        sigint_instakill: false,
178        warn_non_zero_binds: false,
179        sched_heuristic: Default::default(),
180        sched_seed: default_config.sched_seed,
181        recordreplay_modes: true,
182        record_preemptions: false,
183        record_preemptions_to: None,
184        replay_preemptions_from: None,
185        replay_schedule_from: None,
186        replay_exhausted_panic: false,
187        die_on_desync: true,
188        stacktrace_event: Vec::new(),
189        stacktrace_signal: None,
190        preemption_stacktrace: false,
191        preemption_stacktrace_log_file: None,
192        stop_after_turn: None,
193        stop_after_iter: None,
194        debug_externalize_sockets: false,
195        debug_futex_mode: BlockingMode::Precise,
196        sched_sticky_random_param: 0.0,
197        no_rcb_time: false,
198        detlog_heap: false,
199        detlog_stack: false,
200        sysinfo_uptime_offset: 120,
201        memory: 1024 * 1024 * 1024,
202        interrupt_at: vec![],
203        fuzz_futexes: false,
204        chaos_target_races: false,
205        fuzz_seed: None,
206    };
207    if config.preemption_timeout.is_some() && !reverie_ptrace::is_perf_supported() {
208        tracing::warn!(
209            "Hardware perf counters are not supported on this machine. Records/Replays may randomly fail!"
210        );
211        config.preemption_timeout = None;
212    }
213    config
214}

227        assert!(RECORD_VERSION.compatible_with(&RecordVersion(0x102)));
228        assert!(!RECORD_VERSION.compatible_with(&RecordVersion(0x101)));
229        assert!(!RECORD_VERSION.compatible_with(&RecordVersion(0x103)));
