9use std::fs;
10use std::io;
11use std::os::unix;
12use std::path::Path;
13use std::path::PathBuf;
14
15use tempfile::TempDir;
16
17/// Represents a temporary chroot environment. This shall only contain the files
18/// necessary to reproduce an execution.
19pub struct TempChroot {
20    dir: TempDir,
21}
22
23impl TempChroot {
24    /// Creates a temporary directory that the tracee will chroot to. It is here
25    /// where we can hard link files needed by the tracee.
26    pub fn new_in(data_dir: &Path) -> io::Result<Self> {
27        let dir = TempDir::new_in(data_dir)?;
28        Ok(Self { dir })
29    }
30
31    /// Returns the path to the chroot directory.
32    pub fn path(&self) -> &Path {
33        self.dir.path()
34    }
35
36    /// Returns a path relative to the chroot directory.
37    pub fn relpath(&self, path: &Path) -> PathBuf {
38        self.path().join(path.strip_prefix("/").unwrap_or(path))
39    }
40
41    /// Hard link a file path into the chroot directory. Any intermediate
42    /// directory that the path has will also be created. `link` must be an
43    /// absolute path.
44    ///
45    /// NOTE: The `path` should be a path on the same file system. Otherwise this
46    /// will fail. This should be the case if the path being hard linked has
47    /// already been copied to the recording directory.
48    ///
49    /// NOTE: This should only be used for paths we are certain will not be
50    /// modified by a replay. Otherwise, we could end up currupting a previously
51    /// recorded file.
52    ///
53    /// NOTE: This does not handle symlinks. If `path` is a symlink, the real
54    /// path is not hard linked, just the symlink itself.
55    pub fn hard_link(&self, original: &Path, link: &Path) -> io::Result<()> {
56        let link = self.relpath(link);
57
58        if let Some(dir) = link.parent() {
59            fs::create_dir_all(dir)?;
60        }
61
62        fs::hard_link(original, link)
63    }
64
65    /// Create symlink a file path into the chroot directory. Any intermediate
66    /// directory that the path has will also be created. `link` must be an
67    /// absolute path.
68    ///
69    /// NOTE: original path must exist in chroot directory, otherwise this
70    /// could create a dead sym link.
71    pub fn symlink(&self, original: &Path, link: &Path) -> io::Result<()> {
72        let link = self.relpath(link);
73
74        if let Some(dir) = link.parent() {
75            fs::create_dir_all(dir)?;
76        }
77
78        unix::fs::symlink(original, link)
79    }
80
81    /// Copies a file into the chroot.
82    pub fn copy(&self, from: &Path, to: &Path) -> io::Result<()> {
83        let to = self.relpath(to);
84
85        if let Some(dir) = to.parent() {
86            fs::create_dir_all(dir)?;
87        }
88
89        fs::copy(from, to)?;
90        Ok(())
91    }
92
93    /// Like `copy`, but copies `path` to `{chroot}/{path}`.
94    pub fn copy_same(&self, path: &Path) -> io::Result<()> {
95        self.copy(path, path)
96    }
97
98    /// Creates a directory in the chroot.
99    pub fn create_dir_all(&self, path: &Path) -> io::Result<()> {
100        fs::create_dir_all(self.relpath(path))
101    }
102
103    /// Deletes the chroot. The `Drop` impl will also take care of this, but
104    /// calling this explicitly gives a chance to catch any errors.
105    pub fn remove(self) -> io::Result<()> {
106        self.dir.close()
107    }
108}
