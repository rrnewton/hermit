21const SIOCETHTOOL: usize = 0x8946;
22
23/// Returns the stable error used for legacy ioctls whose nested output cannot
24/// be represented by the currently pinned Reverie decoder.
25pub(crate) fn deterministic_ioctl_error(request: &ioctl::Request<'_>) -> Option<Errno> {
26    match request {
27        // SIOCETHTOOL stores its output behind the data pointer nested in an
28        // ifreq. Treating it as an opaque request would lose guest-visible
29        // memory updates, so reject it identically during record and replay.
30        ioctl::Request::SIOCETHTOOL(_) | ioctl::Request::Other(SIOCETHTOOL, _) => {
31            Some(Errno::ENODEV)
32        }
33        _ => None,
34    }
35}

37/// An event. This contains everything needed to verify and reproduce the
38/// execution of a syscall.
39#[derive(Debug, Serialize, Deserialize)]
40pub struct Event {
41    /// The event that we use to reconstruct the outputs of the original syscall.
42    /// This is `Some` if need to record this syscall. If the syscall is already
43    /// deterministic, then this is `None`.
44    ///
45    /// If a recorded syscall failed, then this is `Some(Err(Errno))`. That is,
46    /// the failure should be reproduced during replay.
47    pub event: Result<SyscallEvent, Errno>,
48}
49
50/// A `SyscallEvent` contains all the necessary information to replay a system
51/// call.
52///
53/// Note that we only need a small amount of information to replay a syscall. The
54/// only side effects observable by the user are:
55///  1. Mutable pointers
56///  2. Return values.
57///
58/// No registers are modified by the kernel except for `rax` (the return value).
59/// Therefore, registers themselves do not need to be recorded since they are
60/// strictly inputs. However, any arguments that are pointers that point to
61/// mutable data expected to be modified by the kernel need to be recorded. If
62/// this rule is applied uniformly for all syscalls, then we should be able to
63/// implement full record and replay.
64#[derive(Debug, Serialize, Deserialize)]
65pub enum SyscallEvent {
66    Bytes(Vec<u8>),
67    /// The flattened output bytes of a vectored read (`readv`/`preadv`/
68    /// `preadv2`). The bytes are stored contiguously in read order; on replay
69    /// they are scattered back across the guest's `iovec` buffers. The length of
70    /// the vector is exactly the return value of the syscall.
71    Readv(Vec<u8>),
72    Write(i64),
73    Mmap(MmapEvent),
74    Recvmsg(RecvmsgEvent),
75    /// A syscall whose only value we care about is the return value. For many
76    /// syscalls, this is often the only output of the syscall and thus it is the
77    /// only piece of information that needs to be recorded.
78    Return(i64),
79    Stat(StatEvent),
80    Statfs(Vec<u8>),
81    Statx(StatxBuf),
82    Rdtsc(RdtscResult),
83    Ioctl(ioctl::Output),
84    Timespec(TimespecEvent),
85    Timeofday((Timeval, Timezone)),
86    Poll(PollEvent),
87    SockOpt(SockOptEvent),
88    EpollWait(EpollWaitEvent),
89}

122/// Records the guest-visible outputs of a `poll` or `ppoll` call. Both syscalls
123/// have identical output semantics (the updated `pollfd` array plus a return
124/// count), so they share this event. `ppoll`'s temporary signal mask only
125/// affects which signals can interrupt the wait; a resulting `EINTR` (or a
126/// timeout returning 0) is captured by the enclosing `Event`'s `Result` and
127/// return count respectively, so no extra fields are needed here.
128#[derive(Serialize, Deserialize, Debug)]
129pub struct PollEvent {
130    /// The complete list of file descriptors. Note that only the `revents` field
131    /// in `pollfd` is an output of the syscall. Technically, we only need to
132    /// store the `revents` field, but it is easier to store everything for
133    /// replay purposes (only one simple call to `process_vm_writev` is needed).
134    /// It is possible to do a vectored write, skipping the other fields, but
135    /// that is a little more complicated. For programs that need to wait on many
136    /// file descriptors at once, they should be using `epoll` instead.
137    pub fds: Vec<PollFd>,
138
139    /// The return value (i.e., the number of items in the above list that have
140    /// been updated).
141    ///
142    /// A value of 0 indicates that the call timed out and no file descriptors
143    /// were ready.
144    pub updated: usize,
145}

163}

165#[cfg(test)]
166mod tests {
167    use super::*;
168
169    #[test]
170    fn siocethtool_has_a_deterministic_error() {
171        let request = ioctl::Request::SIOCETHTOOL(None);
172
173        assert_eq!(deterministic_ioctl_error(&request), Some(Errno::ENODEV));
174
175        let legacy_request = ioctl::Request::Other(SIOCETHTOOL, 0x1234);
176        assert_eq!(
177            deterministic_ioctl_error(&legacy_request),
178            Some(Errno::ENODEV)
179        );
180    }
181
182    #[test]
183    fn neighboring_unknown_ioctl_is_not_rejected() {
184        let request = ioctl::Request::Other(SIOCETHTOOL - 1, 0x1234);
185
186        assert_eq!(deterministic_ioctl_error(&request), None);
187    }
