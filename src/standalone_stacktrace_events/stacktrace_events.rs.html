9//! Test that --stacktrace-event actually prints the right event.
10
11use std::fs;
12use std::path::PathBuf;
13use std::process::Command;
14
15use detcore::preemptions::PreemptionReader;
16use detcore::preemptions::PreemptionRecord;
17use regex::Regex;
18use tempfile::NamedTempFile;
19
20fn main() {
21    let allargs: Vec<String> = std::env::args().collect();
22    let hermit = &allargs[1];
23    let prog = &allargs[2];
24
25    let sched_file = NamedTempFile::new().unwrap();
26    let sched_path = sched_file.path();
27    let events = {
28        eprintln!(":: Recording schedule to {}", sched_path.display());
29        eprintln!(
30            ":: Running program under hermit to record schedule: {}",
31            prog
32        );
33        let output = Command::new(hermit)
34            .arg("run")
35            .arg("--base-env=minimal")
36            .arg("--tmp=/tmp")
37            .arg("--record-preemptions-to")
38            .arg(sched_path)
39            .arg(prog)
40            .output()
41            .expect("hermit run to succeed");
42        eprintln!(":: First run exited with status: {:?}", output.status);
43
44        let preempts: PreemptionRecord = PreemptionReader::new(sched_path).load_all();
45        preempts.into_global()
46    };
47    eprintln!(":: Loaded schedule, length {}", events.len());
48
49    // Points where the event after is a different thread:
50    let mut switch_points: Vec<usize> = Vec::new();
51    {
52        let mut expected_pairs: Vec<(i32, i32)> = Vec::new();
53        for ix in 0..events.len() - 1 {
54            let tid1 = events[ix].dettid;
55            let tid2 = events[ix + 1].dettid;
56            // Backtraces are not available for events without rips, so exclude switch points
57            // to and from such events. TODO(T138829292): Perhaps this test should instead not
58            // expect backtraces?
59            if tid1 != tid2 && events[ix].end_rip.is_some() && events[ix + 1].end_rip.is_some() {
60                switch_points.push(ix);
61                expected_pairs.push((tid1.as_raw(), tid2.as_raw()));
62            }
63        }
64        eprintln!(
65            ":: Context switch points, at event indices: {:?}",
66            switch_points
67        );
68        eprintln!(":: Tids before/after switch points: {:?}", expected_pairs);
69    }
70
71    let mut before_after_switch: Vec<usize> = Vec::new();
72    for ix in &switch_points {
73        before_after_switch.push(*ix);
74        before_after_switch.push(*ix + 1);
75    }
76    before_after_switch.dedup();
77
78    eprintln!(
79        ":: Indices of interest, before/after switches, with dedup: {:?}",
80        before_after_switch
81    );
82    let expected_tids: Vec<i32> = before_after_switch
83        .iter()
84        .map(|ix| events[*ix].dettid.as_raw())
85        .collect();
86
87    let go = |args: Vec<&str>| {
88        let stacktrace_paths: Vec<PathBuf> = before_after_switch
89            .iter()
90            .map(|_ix| NamedTempFile::new().unwrap().path().into())
91            .collect();
92        let output = {
93            let mut rerun_cmd = Command::new(hermit);
94            let mut rerun = &mut rerun_cmd;
95            rerun = rerun
96                .arg("--log=info")
97                .arg("run")
98                .arg("--base-env=minimal")
99                .arg("--tmp=/tmp")
100                .args(args);
101            for (i, ix) in before_after_switch.iter().enumerate() {
102                rerun = rerun.arg(format!(
103                    "--stacktrace-event={},{}",
104                    ix,
105                    stacktrace_paths[i].display()
106                ));
107            }
108            eprintln!(":: Running command: {:?}", rerun);
109            rerun.arg(prog).output().expect("hermit run to succeed")
110        };
111        eprintln!(":: Final output status: {:?}", output.status);
112        let stderr = String::from_utf8_lossy(&output.stderr);
113
114        let event_ixs: Vec<usize> = {
115            let re = Regex::new(r"Now output stack trace for scheduled event #(\d+)").unwrap();
116            stderr
117                .lines()
118                .filter(|l| re.is_match(l))
119                .map(|l| {
120                    let cap = re.captures(l);
121                    cap.unwrap()
122                        .get(1)
123                        .unwrap()
124                        .as_str()
125                        .parse::<usize>()
126                        .unwrap()
127                })
128                .collect()
129        };
130
131        let tids: Vec<i32> = stacktrace_paths
132            .iter()
133            .filter_map(|path| {
134                let stack = fs::read_to_string(path).ok()?;
135                let re = Regex::new(r#""thread_id":(\d+)"#).unwrap();
136                re.captures(&stack)
137                    .and_then(|captures| captures.get(1))
138                    .and_then(|tid| tid.as_str().parse::<i32>().ok())
139            })
140            .collect();
141
142        eprintln!(
143            ":: Sched event indices successfully printed ({}): {:?}",
144            event_ixs.len(),
145            event_ixs
146        );
147        eprintln!(":: Tids at stack traces ({}): {:?}", tids.len(), tids);
148
149        assert_eq!(
150            event_ixs, before_after_switch,
151            "Printed stacktraces should be from the expected indices"
152        );
153        assert_eq!(tids, expected_tids, "Tids during expected events matched.");
154    };
155    eprintln!("\n:: Running again, with recording, for stack traces...");
156    go(vec!["--record-preemptions"]);
157
158    eprintln!("\n:: Running again, with replay to print stack traces again...");
159    go(vec!["--replay-schedule-from", sched_path.to_str().unwrap()]);
160
161    eprintln!(":: Test Succeeded!");
162}
