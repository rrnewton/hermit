35fn test_trace_level() -> String {
36    std::env::var("DETCORE_TEST_RUST_LOG").unwrap_or_else(|_| {
37        // This is a compromise. We don't want to slow things down too much, but it's nice
38        // if we print some logs for failures on Sandcastle.
39        "detcore=info".into()
40    })
41}

43fn install_global_test_subscriber<W>(trace_level: &str, writer: W) -> bool
44where
45    W: for<'writer> MakeWriter<'writer> + Send + Sync + 'static,
46{
47    let collector = tracing_subscriber::fmt()
48        .with_env_filter(trace_level)
49        .with_writer(writer)
50        .finish();
51    tracing::subscriber::set_global_default(collector).is_ok()
52}

54static GLOBAL_TEST_SUBSCRIBER: LazyLock<()> =
55    LazyLock::new(|| _ = install_global_test_subscriber(&test_trace_level(), std::io::stderr));

238/// A basic oracle, which expects a success exit code.
239pub fn expect_success(o: &Output, _s: <Detcore as Tool>::GlobalState) {
240    if o.status != reverie::ExitStatus::Exited(0) {
241        eprintln!("Expected successful exit code, instead tracee output was:");
242        print_tracee_output(o);
243        panic!("Guest exited with non-zero status.",);
244    }
245}

247#[macro_export]
248macro_rules! make_det_test_variants {
249    ( $fn:path ) => {
250        $crate::make_det_test_variants!($fn, "all");
251    };
252    ( $fn:path, "all" ) => {
253        $crate::make_det_test_variants!(@variants ["bottom" "middle" "default" "top"] $fn);
254    };
255    ( $fn:path, $($various:tt),*  ) => {
256        $crate::make_det_test_variants!(@variants [$($various)*] $fn);
257    };
258
259    (@variants [ ] $fn:path ) => {
260    };
261
262    // Here we use good-old Lisp-style lists with no commas:
263    (@variants [ $first:tt $($rest:tt)* ] $fn:path ) => {
264        $crate::make_det_test_variants!(@one_variant $first, $fn);
265        $crate::make_det_test_variants!(@variants [ $($rest)* ] $fn);
266    };
267
268    ( @one_variant "bottom", $fn:path ) => {
269        #[test]
270        fn bottom_detcore() {
271            $fn(& $crate::BOTTOM_CFG);
272        }
273    };
274    ( @one_variant "middle", $fn:path ) => {
275        #[test]
276        fn middle_detcore() {
277            $fn(& $crate::MIDDLE_CFG);
278        }
279    };
280    ( @one_variant "default", $fn:path ) => {
281        #[test]
282        fn default_detcore() {
283            $fn(& ::core::default::Default::default());
284        }
285    };
286    ( @one_variant "top", $fn:path ) => {
287        #[test]
288        fn top_detcore() {
289            $fn(& $crate::TOP_CFG);
290        }
291    };
292}

294/// A convenient way to wrap a function to test multiple detcore execution variants, each
295/// as a separate unit test. This is in contrast with `det_test_all_configs` which runs
296/// multiple modes sequentially under one test target.
297///
298/// Arguments are `basic_det_test(function, predicate, modes...)`, where:
299///   - `function` is the procedure under test.
300///   - `predicate` is a function that accepts a Config and returns true if the test should
301///     run deterministically in that configuration.
302///
303/// This generates calls to `det_test_fn_with_config`.
304#[macro_export]
305macro_rules! basic_det_test {
306    ( $fn:path ) => {
307        $crate::basic_det_test!($fn, |_| true);
308    };
309    ( $fn:path, $f:expr ) => {
310        $crate::basic_det_test!(@gendef $fn, $f);
311        $crate::make_det_test_variants!(detcore, "all");
312    };
313    ( $fn:path, $f:expr, $($variants:tt),+ ) => {
314        $crate::basic_det_test!(@gendef $fn, $f);
315        $crate::make_det_test_variants!(detcore $(,$variants)* );
316    };
317
318    (@gendef $fn:path, $f:expr ) => {
319        fn detcore(cfg: & ::detcore::Config) {
320            $crate::det_test_fn_with_config(
321                $f(cfg),
322                $fn,
323                cfg.clone(),
324                $crate::expect_success,
325            );
326        }
327    }
328}

330/// Runs a test across MULTIPLE configurations.  This combines the multiple variants into
331/// a single test execution (running them one after another).  This can be convenient at
332/// times, but, when debugging, you may want to split these apart into their own test
333/// targets.
334///
335/// The first function argument returns `true` iff the test *should* be deterministic
336/// under a given configuration.
337pub fn det_test_all_configs<C, T, O>(check: C, test: T, oracle: O)
338where
339    C: Fn(&Config) -> bool,
340    T: Fn(&Config),
341    O: Fn(&Output, <Detcore as Tool>::GlobalState) + Clone,
342{
343    let do_cfg = |cfg: Config| {
344        let cfg2 = cfg.clone();
345        let test2 = || test(&cfg);
346        println!("Full config: {:?}", cfg);
347        println!(
348            "================================================================================"
349        );
350        let oracle2 = |output: &Output, state: <Detcore as Tool>::GlobalState| {
351            (oracle.clone())(output, state);
352        };
353        det_test_fn_with_config(check(&cfg), test2, cfg2, oracle2);
354    };
355    println!("\nDEFAULT mode");
356    do_cfg(DEFAULT_CFG.clone());
357    println!("\n\"Bottom\", least-strict configuration");
358    do_cfg(BOTTOM_CFG.clone());
359    println!("\n\"Middle\", medium-strict configuration");
360    do_cfg(MIDDLE_CFG.clone());
361    println!("\n\"Top\", most-strict configuration");
362    do_cfg(TOP_CFG.clone());
363}

365/// The log messages produced by the tracer while executing the tracee, split into lines.
366type TracerLogs = Vec<String>;

368struct BufWriter {
369    buf: Arc<Mutex<Vec<u8>>>,
370}
371
372impl Clone for BufWriter {
373    fn clone(&self) -> Self {
374        BufWriter {
375            buf: self.buf.clone(),
376        }
377    }
378}
379
380impl BufWriter {
381    fn new() -> BufWriter {
382        BufWriter {
383            buf: Arc::new(Mutex::new(Vec::new())),
384        }
385    }
386
387    fn buf(&self) -> MutexGuard<'_, Vec<u8>> {
388        self.buf.lock().unwrap()
389    }
390
391    fn get_strings(&self) -> Vec<String> {
392        let mut b = self.buf();
393        let s = String::from_utf8_lossy(&b[..]).to_string();
394        b.clear();
395        s.lines().map(String::from).collect()
396    }
397}
398
399impl io::Write for BufWriter {
400    fn write(&mut self, msg: &[u8]) -> io::Result<usize> {
401        let _ = std::io::stderr().write(msg);
402        self.buf().write(msg)
403    }
404
405    fn flush(&mut self) -> std::io::Result<()> {
406        let _ = std::io::stderr().flush();
407        self.buf().flush()
408    }
409}
410
411impl MakeWriter<'_> for BufWriter {
412    type Writer = BufWriter;
413
414    fn make_writer(&self) -> Self::Writer {
415        BufWriter {
416            buf: self.buf.clone(),
417        }
418    }
419}

421/// Similar to Reverie's `test_fn_with_config` that captures and returns the tracer's
422/// logs as well.
423fn test_fn_with_logs<T, F>(
424    f: F,
425    config: <T::GlobalState as GlobalTool>::Config,
426    capture_output: bool,
427) -> Result<(Output, T::GlobalState, TracerLogs), Error>
428where
429    T: Tool + 'static,
430    F: FnOnce(),
431{
432    LazyLock::force(&GLOBAL_TEST_SUBSCRIBER);
433    let trace_level = test_trace_level();
434    let bufwriter = BufWriter::new();
435    let collector = tracing_subscriber::fmt()
436        .with_env_filter(trace_level)
437        .with_writer(bufwriter.clone())
438        .finish();
439
440    // Wrap function with an allocator reset for deterministic in-guest
441    // allocations. This only has an effect if the global allocator has been set
442    // to test_allocator.
443    let f = move || {
444        test_allocator::GLOBAL
445            // Try to skip to an arbitrary fixed offset that's likely to be far
446            // past all the memory we've allocated so far.
447            .skip_to_offset_if_init(test_allocator::GLOBAL_MAX_OFFSET / 2)
448            .unwrap();
449        f()
450    };
451
452    // Here we have to keep the collector tightly scoped to this test, because
453    // we want to recapture the output of this test and no other.
454    let (out, state) = tracing::subscriber::with_default(collector, || {
455        let rt = tokio::runtime::Builder::new_current_thread()
456            .enable_io()
457            .build()
458            .unwrap();
459        rt.block_on(async move {
460            let tracee = spawn_fn_with_config::<T, _>(f, config, capture_output).await?;
461            tracee.wait_with_output().await
462        })
463    })?;
464    Ok((out, state, bufwriter.get_strings()))
465}

467/// Runs a function as a guest with Detcore's test log filter applied both to
468/// the calling thread and to tracer work performed on other threads.
469pub fn test_fn_with_config<T, F>(
470    f: F,
471    config: <T::GlobalState as GlobalTool>::Config,
472    capture_output: bool,
473) -> Result<(Output, T::GlobalState), Error>
474where
475    T: Tool + 'static,
476    F: FnOnce(),
477{
478    test_fn_with_logs::<T, F>(f, config, capture_output)
479        .map(|(output, state, _logs)| (output, state))
480}
481
482/// Runs a function as a guest with Detcore's test log filter and requires a
483/// successful guest exit status.
484pub fn check_fn_with_config<T, F>(
485    f: F,
486    config: <T::GlobalState as GlobalTool>::Config,
487    capture_output: bool,
488) -> T::GlobalState
489where
490    T: Tool + 'static,
491    F: FnOnce(),
492{
493    let (output, state) = test_fn_with_config::<T, F>(f, config, capture_output).unwrap();
494    if output.status != ExitStatus::Exited(0) {
495        print_tracee_output(&output);
496        panic!("Got exit status {:?}", output.status);
497    }
498    state
499}

514where

521/// Like [`det_test_fn_with_config`], but runs and compares exactly `repetitions` times.
522pub fn det_test_fn_with_config_repetitions<F, O>(
523    repetitions: u64,
524    isdet: bool,
525    f: F,
526    config: Config,
527    oracle: O,
528) where
529    F: Fn(),
530    O: Fn(&Output, <Detcore as Tool>::GlobalState),
531{
532    assert!(repetitions > 0, "at least one test repetition is required");
533
534    if isdet {
535        println!("Expecting determinism:");
536    } else {
537        println!("Not expecting determinism, but still performing multiple test runs:");
538    }
539    let mut dts = DetTestState::default();
540    for ix in 1..repetitions {
541        println!("Test Run {}:", ix);
542        let (output, state, logs) =
543            test_fn_with_logs::<Detcore, _>(&f, config.clone(), true).unwrap();
544        println!("({} log lines captured.)", logs.len());
545        oracle(&output, state);
546        println!("Oracle passed.");
547        if isdet {
548            println!("Comparing against prior run, if any.");
549            check_output(&output, logs, &mut dts);
550        }
551    }
552    println!("Test Run {}:", repetitions);
553    let (output, state, logs) = test_fn_with_logs::<Detcore, _>(f, config, true).unwrap();
554    println!("({} log lines captured.)", logs.len());
555    oracle(&output, state);
556    println!("Oracle passed. Full stderr:");
557    println!("{}", String::from_utf8_lossy(&output.stderr));
558    println!("Full stdout:");
559    println!("{}", String::from_utf8_lossy(&output.stdout));
560    if isdet {
561        println!("Comparing against prior run, if any.");
562        check_output(&output, logs, &mut dts);
563    }
564}

575/// Like `det_test_cmd`, but allows passing in a non-default configuration.
576pub fn det_test_cmd_with_config<O>(program: &str, args: &[&str], oracle: O, config: Config)
577where
578    O: Fn(&Output, <Detcore as Tool>::GlobalState),
579{
580    LazyLock::force(&GLOBAL_TEST_SUBSCRIBER);
581    let mut dts = DetTestState::default();
582    for _ in 1..(TEST_REPS - 1) {
583        let (output, state) =
584            test_cmd_with_config::<Detcore>(program, args, config.clone()).unwrap();
585        oracle(&output, state);
586        println!("Oracle passed. Comparing against prior run, if any.");
587        check_output(&output, Vec::new(), &mut dts);
588    }
589
590    let (output, state) = test_cmd_with_config::<Detcore>(program, args, config).unwrap();
591    oracle(&output, state);
592    println!("Oracle passed. Comparing against prior run, if any.");
593    check_output(&output, Vec::new(), &mut dts);
594}

596/// Checks the output of a run to ensure there is no difference from the last
597/// run.
598fn check_output(output: &Output, logs: Vec<String>, dts: &mut DetTestState) {
599    dts.test_run_num += 1;
600
601    match dts.last_exit {
602        None => {}
603        Some(x) => assert_eq!(
604            x, output.status,
605            "\n  Consecutive runs of test had different exit status: {:?} {:?}",
606            x, output.status
607        ),
608    }
609    dts.last_exit = Some(output.status);
610
611    let stdout_str = String::from_utf8(output.stdout.clone()).unwrap_or_else(|_| {
612        panic!(
613            "Test produced stdout that is not valid utf8: {:?}",
614            output.stdout
615        );
616    });
617    let stderr_str = String::from_utf8(output.stderr.clone()).unwrap_or_else(|_| {
618        panic!(
619            "Test produced stderr that is not valid utf8: {:?}",
620            output.stderr
621        );
622    });
623
624    // TODO: use some kind of diffing framework to concisely print only the
625    // diffs, as well as some quantitative info about the extent of the diff.
626    match &dts.last_stdout {
627        None => {}
628        Some(x) => assert_eq!(
629            x, &stdout_str,
630            "\n  Consecutive runs of test had different stdout, run1:\n{:?}\n  Run2:\n{:?}",
631            x, &stdout_str
632        ),
633    }
634    dts.last_stdout = Some(stdout_str);
635
636    match &dts.last_stderr {
637        None => {}
638        Some(x) => assert_eq!(
639            x, &stderr_str,
640            "\n  Consecutive runs of test had different stderr, run1:\n{:?}\n  Run2:\n{:?}",
641            x, &stderr_str
642        ),
643    }
644    dts.last_stderr = Some(stderr_str);
645
646    let filtered: Vec<String> = logs
647        .iter()
648        .filter(|l| l.contains("COMMIT turn"))
649        .map(|s| {
650            let vec: Vec<&str> = s.split("COMMIT").collect();
651            if let [_pref, suffix] = vec[..] {
652                suffix.to_string()
653            } else {
654                panic!("Unexpected form to COMMIT log line: {}", s);
655            }
656        })
657        .collect();
658
659    match &dts.last_log {
660        None => {}
661        Some(x) => {
662            if x.len() != filtered.len() {
663                eprintln!(
664                    "Differing number of commit lines! ({} vs {})",
665                    x.len(),
666                    filtered.len()
667                )
668            }
669            for ix in 0..x.len().min(filtered.len()) {
670                let str_a = detcore::logdiff::strip_log_entry(&x[ix]);
671                let str_b = detcore::logdiff::strip_log_entry(&filtered[ix]);
672                assert_eq!(
673                    str_a, str_b,
674                    "\n  Consecutive runs of test had different COMMITs #{}, run1:\n{:?}\n  Run2:\n{:?}",
675                    ix, str_a, str_b,
676                )
677            }
678            if x.len() != filtered.len() {
679                panic!(
680                    "All present lines matched, but different number of COMMIT lines across two runs."
681                )
682            }
683        }
684    }
685    dts.last_log = Some(filtered);
686}
